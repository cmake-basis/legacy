50  * @brief Determine if this is the built or the installed executable.
51  *
52  * @returns Whether the executable was executed from within the build tree.
53  */
54 static inline bool executing_in_build_tree()
55 {
56  // get executable path relative to top directory of build tree
57  string rel_path = os::path::relpath(
58  // directory of this executable
59  os::exedir(),
60  // The following path is the absolute path of the top directory of the build
61  // tree in which the software was built. It can be used to determine whether
62  // an executable is executing from within the build tree rather than an
63  // installation. Given that we tend to build software in a directory with the
64  // prefix "-build", but never choose such directory for the installation prefix,
65  // a check whether or not the location of the executable file is inside this
66  // build tree is enough to know whether or not it is executed from within the
67  // build tree or an installation.
68  os::path::realpath("/Users/as12312/Software/SBIA/BASIS/build-clang-5.0-release"));
69  // return whether executable directory is inside the build tree or not
70  return !(rel_path == "" || rel_path == "." ||
71  (rel_path.substr(0, 2) == ".." && (rel_path.size() == 2 || rel_path[2] == '/')));
72 }

165 /**
166  * @brief Get absolute path of installation directory.
167  *
168  * This function returns the absolute path of the installation prefix path.
169  * If the installation was not moved after a "make install", the returned
170  * directory corresponds to the value of the INSTALL_PREFIX CMake variable
171  * as specified during the build of the executable file. Note, however,
172  * that even when the installation tree was moved after the configuration
173  * and build of the software, the correct path is returned as long as the
174  * relative directory structure of the installation tree is maintained.
175  * This is because the path is determined relative to the directory of the
176  * executable itself, knowning in which path this executable is located
177  * relative to the INSTALL_PREFIX.
178  *
179  * @note If the executable is executed from within the build tree, the
180  * returned path will not be correct. Therefore, only use this
181  * function when executing_in_build_tree() returns false. Otherwise,
182  * the configured absolute paths which are valid for the build tree
183  * have to be used. Note that the build tree is not supposed to be
184  * relocatable in any case as it is only a temporary directory tree
185  * and CMake requires it to be not moved anywhere else.
186  *
187  * @returns Absolute path of top directory of installation tree.
188  */
189 static string install_prefix()
190 {
191 #ifdef LIBEXEC
192  return os::path::join(os::exedir(), "..");
193 #else
194  return os::path::join(os::exedir(), "..");
195 #endif
196 }

320 string ExecutableTargetInfo::targetuid(const string& target) const
321 {
322  // empty(, invalid) target name remains unchanged
323  if (target.empty()) return "";
324  // in case of a leading namespace separator, do not modify target name
325  if (target[0] == '.') return target;
326  // project namespace
327  string prefix = string("basis");
328  // try prepending namespace or parts of it until target is known
329  for (;;) {
330  if (_exec_names.find(prefix + "." + target) != _exec_names.end()) {
331  return prefix + "." + target;
332  }
333  string::size_type pos = prefix.rfind('.');
334  if (pos == string::npos) break;
335  prefix = prefix.substr(0, pos);
336  }
337  // otherwise, return target name unchanged
338  return target;
339 }

352 string ExecutableTargetInfo::basename(const string& target) const
353 {
354  if (target.empty()) return "";
355  string uid;
356  if (target[0] == '.') uid = target.substr(1);
357  else uid = targetuid(target);
358  MapIterator it = _exec_names.find(uid);
359  if (it == _exec_names.end ()) return "";
360  return it->second;
361 }

363 // ---------------------------------------------------------------------------
364 string ExecutableTargetInfo::dirname(const string& target) const
365 {
366  if (target.empty()) return "";
367  string uid;
368  if (target[0] == '.') uid = target.substr(1);
369  else uid = targetuid(target);
370  if (executing_in_build_tree()) {
371  MapIterator it = _build_dirs.find(uid);
372  if (it == _build_dirs.end()) return "";
373  #if defined CMAKE_INTDIR
374  // this subdirectory is used by CMake generators for IDEs such as Visual Studio
375  return os::path::join(it->second, CMAKE_INTDIR);
376  #else
377  return it->second;
378  #endif
379  } else {
380  MapIterator it = _install_dirs.find(uid);
381  if (it == _install_dirs.end()) return "";
382  return os::path::join(install_prefix(), it->second);
383  }
384 }
