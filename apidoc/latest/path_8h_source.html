54 /**
55  * @brief Normalize path, i.e., remove occurences of "./", duplicate slashes,...
56  *
57  * This function removes single periods enclosed by slashes (or backslashes),
58  * duplicate slashes (or backslashes), and further tries to reduce the
59  * number of parent directory references. Moreover, on Windows, it replaces
60  * slashes with backslashes.
61  *
62  * For example, on Windows, "../bla//.//.\bla\\\\\bla/../.." is convert to "..\bla".
63  * On Unix, it is converted to "..". Note that on Unix, ".\bla\\\\\bla" is a
64  * directory name.
65  *
66  * @param [in] path Path.
67  *
68  * @returns Normalized path.
69  */
70 std::string normpath(const std::string& path);
71

77  *
78  * @param [in] path Path.

106 /**
107  * @brief Split path into two parts.
108  *
109  * This function splits a path into its head and tail. Trailing slashes are
110  * stripped from head unless it is the root. See the following table for an
111  * illustration:
112  @verbatim
113  path | head | tail
114  -------------------------+-------+---------------+-------------
115  "/" | "/" | ""
116  "/usr/bin" | "/usr" | "bin"
117  "/home/user/info.txt" | "/home/user" | "info.txt"
118  "word.doc" | "" | "word.doc"
119  "../word.doc" | ".." | "word.doc"
120  "C:/" | "C:/" | ""
121  "C:" | "C:" | ""
122  "C:/WINDOWS/regedit.exe" | "C:/WINDOWS" | "regedit.exe"
123  "d:\data" | "d:/" | "data"
124  "/usr/local/" | "/usr/local" | ""
125  @endverbatim
126  * In all cases, join(head, tail) returns a path to the same location as path
127  * (but the strings may differ).
128  *
129  * @param [in] path Path.
130  * @param [out] head Head part of path.
131  * @param [out] tail Tail part of path.
132  *
133  * @sa splitdrive()
134  * @sa splitext()
135  * @sa dirname()
136  * @sa basename()
137  */
138 void split(const std::string& path, std::string& head, std::string& tail);
139

168 /**
169  * @brief Get file name extension.
170  *
171  * @param [in] path Path.
172  * @param [out] head Remaining path without extension.
173  * @param [out] ext Extension (including leading dot).
174  * @param [in] exts Set of recognized extensions. Note that the given set
175  * can contain extensions with dots (.) as part of the
176  * extension, e.g., ".nii.gz". If NULL is given, the part
177  * after the last dot (including the dot) is considered to
178  * be the file name extension. Otherwise, the longest extension
179  * from the given set which is equal to the end of the file
180  * path is returned. If no specified extension matched, an
181  * empty string is returned as extension.
182  * @param [in] icase Whether to ignore the case of the extensions.
183  */
184 void splitext(const std::string& path, std::string& head, std::string& ext,
185  const std::set<std::string>* exts = NULL, bool icase = false);
