23 ## @brief Clean path, i.e., remove occurences of "./", duplicate slashes,...
24 #
25 # This function removes single periods (.) enclosed by slashes or backslashes,
26 # duplicate slashes (/) or backslashes (\), and further tries to reduce the
27 # number of parent directory references.
28 #
29 # For example, "../bla//.//.\bla\\\\\bla/../.." is convert to "../bla".
30 #
31 # @param [in] path Path.
32 #
33 # @return Cleaned path.
34 normpath()
35 {
36  local _basis_cp_path="$1"
37  # GNU bash, version 3.00.15(1)-release (x86_64-redhat-linux-gnu)
38  # turns the array into a single string value if local is used
39  if [ ${BASH_VERSION_MAJOR} -gt 3 ] || [ ${BASH_VERSION_MAJOR} -eq 3 -a ${BASH_VERSION_MINOR} -gt 0 ]; then
40  local _basis_cp_dirs=()
41  else
42  _basis_cp_dirs=()
43  fi
44  # split path into parts, discarding redundant slashes
45  while [ -n "${_basis_cp_path}" ]; do
46  if [ ${#_basis_cp_dirs[@]} -eq 0 ]; then
47  _basis_cp_dirs=("`basename -- "${_basis_cp_path}"`")
48  else
49  _basis_cp_dirs=("`basename -- "${_basis_cp_path}"`" "${_basis_cp_dirs[@]}")
50  fi
51  _basis_cp_path="`dirname -- "${_basis_cp_path}"`"
52  if [ "${_basis_cp_path}" == '/' ]; then
53  _basis_cp_path=''
54  fi
55  done
56  # build up path again from the beginning,
57  # discarding dots ('.') and stepping one level up for each '..'
58  local _basis_cp_i=0
59  while [ ${_basis_cp_i} -lt ${#_basis_cp_dirs[@]} ]; do
60  if [ "${_basis_cp_dirs[${_basis_cp_i}]}" != '.' ]; then
61  if [ "${_basis_cp_dirs[${_basis_cp_i}]}" == '..' ]; then
62  _basis_cp_path=`dirname -- "${_basis_cp_path}"`
63  else
64  _basis_cp_path="${_basis_cp_path}/${_basis_cp_dirs[${_basis_cp_i}]}"
65  fi
66  fi
67  let _basis_cp_i++
68  done
69  # return
70  echo -n "${_basis_cp_path}"
71 }

74 ## @brief Get absolute path given a relative path.
75 #
76 # This function converts a relative path to an absolute path. If the given
77 # path is already absolute, this path is passed through unchanged.
78 #
79 # @param [in] path Absolute or relative path.
80 #
81 # @return Absolute path.
82 abspath()
83 {
84  local _basis_tap_base="$1"
85  local _basis_tap_path="$2"
86  if [ "${_basis_tap_base:0:1}" != '/' ]; then
87  _basis_tap_base="`pwd`/${_basis_tap_base}"
88  fi
89  if [ "${_basis_tap_path:0:1}" != '/' ]; then
90  _basis_tap_path="${_basis_tap_base}/${_basis_tap_path}"
91  fi
92  normpath "${_basis_tap_path}"
93 }

96 ## @brief Get canonical file path.
97 #
98 # This function resolves symbolic links and returns a cleaned path.
99 #
100 # @param [in] path Path.
101 #
102 # @return Canonical file path without duplicate slashes, ".", "..",
103 # and symbolic links.
104 realpath()
105 {
106  # make path absolute and resolve '..' references
107  local _basis_grp_path=`abspath "$1"`
108  if ! [ -e "${_basis_grp_path}" ]; then echo -n "${_basis_grp_path}"; return; fi
109  # resolve symbolic links within path
110  _basis_grp_path=`cd -P -- $(dirname -- "${_basis_grp_path}") && pwd -P`/`basename -- "${_basis_grp_path}"`
111  # if path itself is a symbolic link, follow it
112  local _basis_grp_i=0
113  local _basis_grp_cur="${_basis_grp_path}"
114  while [ -h "${_basis_grp_cur}" ] && [ ${_basis_grp_i} -lt 100 ]; do
115  _basis_grp_dir=`dirname -- "${_basis_grp_cur}"`
116  _basis_grp_cur=`readlink -- "${_basis_grp_cur}"`
117  _basis_grp_cur=`cd "${_basis_grp_dir}" && cd $(dirname -- "${_basis_grp_cur}") && pwd`/`basename -- "${_basis_grp_cur}"`
118  let _basis_grp_i++
119  done
120  # If symbolic link could entirely be resolved in less than 100 iterations,
121  # return the obtained canonical file path. Otherwise, return the original
122  # link which could not be resolved due to some probable cycle.
123  if [ ${_basis_grp_i} -lt 100 ]; then _basis_grp_path="${_basis_grp_cur}"; fi
124  # return
125  echo -n "${_basis_grp_path}"
126 }
