103 ## @brief Get UID of build target.
104 #
105 # The UID of a build target is its name prepended by a namespace identifier
106 # which should be unique for each project.
107 #
108 # @param [in] name Name of build target.
109 # @param [in] prefix Common prefix of targets belonging to this project.
110 # @param [in] targets Dictionary mapping target UIDs to executable paths.
111 #
112 # @returns UID of named build target.
113 def targetuid(name, prefix=None, targets=None):
114  # handle invalid arguments
115  if not name: return None
116  # in case of a leading namespace separator, do not modify target name
117  if name.startswith('.'): return name
118  # common target UID prefix of project
119  if prefix is None or not targets: return name
120  # try prepending namespace or parts of it until target is known
121  separator = '.'
122  while True:
123  if separator.join([prefix, name]) in targets:
124  return separator.join([prefix, name])
125  parts = prefix.split(separator, 1)
126  if len(parts) == 1: break
127  prefix = parts[0]
128  # otherwise, return target name unchanged
129  return name

145 # ----------------------------------------------------------------------------
146 ## @brief Get absolute path of executable file.
147 #
148 # This function determines the absolute file path of an executable. If no
149 # arguments are given, the absolute path of this executable is returned.
150 # If the command names a known executable build target, the absolute path to
151 # the corresonding built (and installed) executable file is returned.
152 # Otherwise, the named command is searched in the system @c PATH and its
153 # absolute path returned if found. If the executable is not found, @c None
154 # is returned.
155 #
156 # @param [in] name Name of command or @c None.
157 # @param [in] prefix Common prefix of targets belonging to this project.
158 # @param [in] targets Dictionary mapping target UIDs to executable paths.
159 # @param [in] base Base directory for relative paths in @p targets.
160 #
161 # @returns Absolute path of executable or @c None if not found.
162 # If @p name is @c None, the path of this executable is returned.
163 def exepath(name=None, prefix=None, targets=None, base='.'):
164  path = None
165  if name is None:
166  path = os.path.realpath(sys.argv[0])
167  elif istarget(name, prefix=prefix, targets=targets):
168  uid = targetuid(name, prefix=prefix, targets=targets)
169  if uid.startswith('.'): uid = uid[1:]
170  path = os.path.normpath(os.path.join(os.path.join(_MODULE_DIR, base), targets[uid]))
171  if '$(IntDir)' in path:
172  for intdir in ['Release', 'Debug', 'RelWithDebInfo', 'MinSizeRel']:
173  tmppath = path.replace('$(IntDir)', intdir)
174  if os.path.isfile(tmppath):
175  path = tmppath
176  break
177  path = path.replace('$(IntDir)', '')
178  else:
179  try:
180  path = which.which(name)
181  except which.WhichError:
182  pass
183  return path

232 # ----------------------------------------------------------------------------
233 ## @brief Convert array of arguments to quoted string.
234 #
235 # @param [in] args Array of arguments.
236 #
237 # @returns Double quoted string, i.e., string where arguments are separated
238 # by a space character and surrounded by double quotes if necessary.
239 # Double quotes within an argument are escaped with a backslash.
240 #
241 # @sa split_quoted_string()
242 def tostring(args):
243  qargs = []
244  re_quote_or_not = re.compile(r"'|\s|^$")
245  for arg in args:
246  # escape double quotes
247  arg = arg.replace('"', '\\"')
248  # surround element by double quotes if necessary
249  if re_quote_or_not.search(arg): qargs.append(''.join(['"', arg, '"']))
250  else: qargs.append(arg)
251  return ' '.join(qargs)

264 # ----------------------------------------------------------------------------
265 ## @brief Execute command as subprocess.
266 #
267 # @param [in] args Command with arguments given either as quoted string
268 # or array of command name and arguments. In the latter
269 # case, the array elements are converted to strings
270 # using the built-in str() function. Hence, any type
271 # which can be converted to a string is permitted.
272 # The first argument must be the name or path of the
273 # executable of the command.
274 # @param [in] quiet Turns off output of @c stdout of child process to
275 # stdout of parent process.
276 # @param [in] stdout Whether to return the command output.
277 # @param [in] allow_fail If true, does not raise an exception if return
278 # value is non-zero. Otherwise, a @c SubprocessError is
279 # raised by this function.
280 # @param [in] verbose Verbosity of output messages.
281 # Does not affect verbosity of executed command.
282 # @param [in] simulate Whether to simulate command execution only.
283 # @param [in] prefix Common prefix of targets belonging to this project.
284 # @param [in] targets Dictionary mapping target UIDs to executable paths.
285 # @param [in] base Base directory for relative paths in @p targets.
286 #
287 # @return The exit code of the subprocess if @p stdout is false (the default).
288 # Otherwise, if @p stdout is true, a tuple consisting of exit code
289 # command output is returned. Note that if @p allow_fail is false,
290 # the returned exit code will always be 0.
291 #
292 # @throws SubprocessError If command execution failed. This exception is not
293 # raised if the command executed with non-zero exit
294 # code but @p allow_fail set to @c True.
295 def execute(args, quiet=False, stdout=False, allow_fail=False, verbose=0, simulate=False,
296  prefix=None, targets=None, base='.'):
297  # convert args to list of strings
298  if type(args) is list: args = [str(i) for i in args]
299  elif type(args) in (str, unicode): args = qsplit(args);
300  else: raise Exception("execute(): Argument args must be either list or string, but %s given" % type(args))
301  if len(args) == 0: raise Exception("execute(): No command specified for execution")
302  # get absolute path of executable
303  path = exepath(args[0], prefix=prefix, targets=targets, base=base)
304  if not path: raise SubprocessError(args[0] + ": Command not found")
305  args[0] = path
306  # some verbose output
307  if verbose > 0 or simulate:
308  sys.stdout.write('$ ')
309  sys.stdout.write(tostring(args))
310  if simulate: sys.stdout.write(' (simulated)')
311  sys.stdout.write('\n')
312  # execute command
313  status = 0
314  output = ''
315  if not simulate:
316  try:
317  # open subprocess
318  process = subprocess.Popen(args, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
319  # read stdout until EOF
320  for line in process.stdout:
321  if stdout:
322  output = ''.join([output, line])
323  if not quiet:
324  print line.rstrip()
325  sys.stdout.flush()
326  # wait until subprocess terminated and set exit code
327  (out, err) = process.communicate()
328  # print error messages of subprocess
329  for line in err: sys.stderr.write(line);
330  # get exit code
331  status = process.returncode
332  except OSError, e:
333  raise SubprocessError(args[0] + ': ' + str(e))
334  except Exception, e:
335  msg = "Exception while executing \"" + args[0] + "\"!\n"
336  msg += "\tArguments: " + tostring(args[1:]) + '\n'
337  msg += '\t' + str(e)
338  raise SubprocessError(msg)
339  # if command failed, throw an exception
340  if status != 0 and not allow_fail:
341  raise SubprocessError("** Failed: " + tostring(args))
342  # return
343  if stdout: return (status, output)
344  else: return status
