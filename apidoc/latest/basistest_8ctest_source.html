296  # get compiler information from CMake cache
297  if (EXISTS "${CTEST_BINARY_DIRECTORY}/CMakeCache.txt")

309  set (COMPILER "${CMAKE_BUILD_TOOL}")
310  get_filename_component (COMPILER_NAME "${CMAKE_BUILD_TOOL}" ${NAME_COMPONENT})
311  endif ()
312  if (COMPILER_NAME MATCHES "msdev")
313  set (COMPILER_NAME "vs60")
314  elseif (COMPILER_NAME MATCHES "devenv")
315  read_cmake_cache_value (CMAKE_GENERATOR)
316  string (REGEX REPLACE "Visual Studio ([0-9][0-9]?)($|.*)" "\\1" NUMBER "${CMAKE_GENERATOR}")
317  if ("${CMAKE_GENERATOR}" MATCHES "Visual Studio 7 .NET 2003")
318  set (COMPILER_NAME "vs71")
319  else ()
320  set (COMPILER_NAME "vs${NUMBER}")
321  endif ()
322  endif ()
323  else ()
324  set (COMPILER "unknown")
325  set (COMPILER_NAME "unknown")
326  endif ()
327  # replace COMPILER_NAME in build name
328  string (CONFIGURE "${CTEST_BUILD_NAME}" BUILDNAME @ONLY)
329  message ("Change build name to ${BUILDNAME}")
330  # get tag, i.e., subdirectory name required by update_metadata()
331  if (EXISTS "${CTEST_BINARY_DIRECTORY}/Testing/TAG")
332  file (STRINGS "${CTEST_BINARY_DIRECTORY}/Testing/TAG" TAG LIMIT_COUNT 1)
333  else ()
334  set (TAG ".")
335  endif ()
336  # modify submission files
337  if (NOT ARGV)
338  file (
339  GLOB_RECURSE
340  ARGV
341  RELATIVE
342  "${CTEST_BINARY_DIRECTORY}/Testing/${TAG}"
343  "${CTEST_BINARY_DIRECTORY}/Testing/${TAG}/*.xml"
344  )
345  endif ()
346  foreach (SUBMISSION_FILE IN LISTS ARGV)
347  set (SUBMISSION_FILE "${CTEST_BINARY_DIRECTORY}/Testing/${TAG}/${SUBMISSION_FILE}")
348  # read submission file
349  file (READ "${SUBMISSION_FILE}" XML)
350  # set compiler name if necessary

352  string (REPLACE "<CompilerName></CompilerName>" "<CompilerName>${COMPILER}</CompilerName>" XML "${XML}")
353  # replace intermediate build name

355  string (REPLACE "<BuildName></BuildName>" "<BuildName>${BUILDNAME}</BuildName>" XML "${XML}")
356  string (REPLACE "BuildName=\"${CTEST_BUILD_NAME}\"" "BuildName=\"${BUILDNAME}\"" XML "${XML}")
357  string (REPLACE "<BuildName>${CTEST_BUILD_NAME}</BuildName>" "<BuildName>${BUILDNAME}</BuildName>" XML "${XML}")
358  # write modified submission file
359  file (WRITE "${SUBMISSION_FILE}" "${XML}")
360  endforeach ()
