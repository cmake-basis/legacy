53 sub _source
54 {
55  my ($self, $unused, $module) = @_;
56  $module =~ s/^\s*[\"']?//;
57  $module =~ s/[\"']?\s*$//;
58  $module =~ s/\/.\//\//g;
59  $module =~ s/^\${_\w+_DIR}\///;
60  $module =~ s/^\$\(exedir\)\///;
61  $module =~ s/^\${?exec_dir}?\///;
62  $self->_append("#include \"$module\"");
63 }

66 sub _constant
67 {
68  my ($self, $unused1, $unused2, $name, $value) = @_;
69  if (not $name =~ m/^_/) {
70  $value =~ s/^\s*[\"']?//;
71  $value =~ s/[\"']?\s*$//;
72  if ($value =~ /^[+-]?[0-9]+$/) {
73  $self->_append("int $name = $value;");
74  } elsif ($value =~ /^[+-]?[0-9]+[.][0-9]+$/) {
75  $self->_append("float $name = $value;");
76  } else {
77  $self->_append("string $name = \"$value\";");
78  }
79  }
80 }

83 sub _fndef
84 {
85  my ($self, $unused, $name1, $name2) = @_;
86  my $name = $name1 ? $name1 : $name2;
87  if (not $name =~ m/^_/) {
88  my @params = ();
89  foreach my $paramdoc (@{$self->{'params'}}) {
90  push @params, $paramdoc->{'dir'} . " " . $paramdoc->{'name'};
91  }
92  $self->_append("/// \@returns Nothing.") if not $self->{'returndoc'};
93  $self->_append("function $name(" . join(', ', @params) . ");");
94  }
