229 # ----------------------------------------------------------------------------
230 sub _fndef_append
231 {
232  my ($self, $line) = @_;
233  if ($line =~ /^\s*(macro|function)\s*\(\s*(\w+)(\s+[^\)]*)?\)\s*(#.*)?$/) {
234  my $type = $1;
235  my $name = $2;
236  my $params = $3;
237  my @params = ();
238  if ($params) {
239  chomp $params;
240  $params =~ s/^\s+//;
241  @params = split /\s+/, $params;
242  for (my $i = 0; $i <= $#params; $i++) {
243  my $dir = 'in';
244  foreach my $paramdoc (@{$self->{'params'}}) {
245  if ($paramdoc->{'name'} eq $params[$i]) {
246  $dir = $paramdoc->{'dir'};
247  last;
248  }
249  }
250  $params[$i] = "$dir $params[$i]";
251  }
252  }
253  foreach my $paramdoc (@{$self->{'params'}}) {
254  push @params, $paramdoc->{'dir'} . " " . $paramdoc->{'name'}
255  if $paramdoc->{'name'} =~ /^ARG(N|V[0-9])$/;
256  }
257  $self->_append("/// \@returns Nothing.") if not $self->{'returndoc'};
258  $self->_append("$type $name(" . join(', ', @params) . ");");
259  }
260 }
