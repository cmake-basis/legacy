144 # determine python version string
145 if(PYTHON_EXECUTABLE)
146  execute_process(COMMAND "${PYTHON_EXECUTABLE}" -E -c
147  "import sys; sys.stdout.write(';'.join([str(x) for x in sys.version_info[:3]]))"
148  OUTPUT_VARIABLE _VERSION
149  RESULT_VARIABLE _PYTHON_VERSION_RESULT
150  ERROR_QUIET)
151  if(NOT _PYTHON_VERSION_RESULT)
152  string(REPLACE ";" "." PYTHON_VERSION_STRING "${_VERSION}")
153  # just to make sure as some programs (i.e. pydoc) do not return a non-zero
154  # exit code even though called with unknown option. This happened by mistake
155  # once when pydoc was specified instead of python as PYTHON_EXECUTABLE
156  if(NOT PYTHON_VERSION_STRING MATCHES "^[0-9]+\\.[0-9]+\\.[0-9]+.*")
157  message(WARNING "Version string returned by Python executable\n"
158  "\tPYTHON_EXECUTABLE = ${PYTHON_EXECUTABLE}"
159  "\tPYTHON_VERSION_STRING = ${PYTHON_VERSION_STRING}"
160  "\ndoes not match the expected format: <major>.<minor>.<patch>")
161  endif()
162  list(GET _VERSION 0 PYTHON_VERSION_MAJOR)
163  list(GET _VERSION 1 PYTHON_VERSION_MINOR)
164  list(GET _VERSION 2 PYTHON_VERSION_PATCH)
165  if(PYTHON_VERSION_PATCH EQUAL 0)
166  # it's called "Python 2.7", not "2.7.0"
167  string(REGEX REPLACE "\\.0$" "" PYTHON_VERSION_STRING "${PYTHON_VERSION_STRING}")
168  endif()
169  else()
170  # sys.version predates sys.version_info, so use that
171  execute_process(COMMAND "${PYTHON_EXECUTABLE}" -E -c "import sys; sys.stdout.write(sys.version)"
172  OUTPUT_VARIABLE _VERSION
173  RESULT_VARIABLE _PYTHON_VERSION_RESULT
174  ERROR_QUIET)
175  if(NOT _PYTHON_VERSION_RESULT)
176  string(REGEX REPLACE " .*" "" PYTHON_VERSION_STRING "${_VERSION}")
177  string(REGEX REPLACE "^([0-9]+)\\.[0-9]+.*" "\\1" PYTHON_VERSION_MAJOR "${PYTHON_VERSION_STRING}")
178  string(REGEX REPLACE "^[0-9]+\\.([0-9])+.*" "\\1" PYTHON_VERSION_MINOR "${PYTHON_VERSION_STRING}")
179  if(PYTHON_VERSION_STRING MATCHES "^[0-9]+\\.[0-9]+\\.[0-9]+.*")
180  string(REGEX REPLACE "^[0-9]+\\.[0-9]+\\.([0-9]+).*" "\\1" PYTHON_VERSION_PATCH "${PYTHON_VERSION_STRING}")
181  else()
182  set(PYTHON_VERSION_PATCH "0")
183  endif()
184  else()
185  # sys.version was first documented for Python 1.5, so assume
186  # this is older.
187  set(PYTHON_VERSION_STRING "1.4")
188  set(PYTHON_VERSION_MAJOR "1")
189  set(PYTHON_VERSION_MINOR "4")
190  set(PYTHON_VERSION_PATCH "0")
191  endif()
192  endif()
193  unset(_PYTHON_VERSION_RESULT)
194  unset(_VERSION)
195 endif(PYTHON_EXECUTABLE)
