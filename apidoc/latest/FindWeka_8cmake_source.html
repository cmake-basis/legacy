88 # @brief Determine if this Weka version has a package manager.
89 #
90 # @returns This macro sets the Weka_HAS_PACKAGE_MANAGER variable.
91 macro (weka_has_package_manager)
92  if (Weka_VERSION_MAJOR AND Weka_VERSION_MINOR)
93  if (Weka_VERSION_MAJOR EQUAL 3)
94  if (Weka_VERSION_MINOR GREATER 7 OR (Weka_VERSION_MINOR EQUAL 7 AND Weka_VERSION_PATCH GREATER 1))
95  set (Weka_HAS_PACKAGE_MANAGER TRUE)
96  else ()
97  set (Weka_HAS_PACKAGE_MANAGER FALSE)
98  endif ()
99  elseif (Weka_VERSION_MAJOR GREATER 2)
100  set (Weka_HAS_PACKAGE_MANAGER TRUE)
101  else ()
102  set (Weka_HAS_PACKAGE_MANAGER FALSE)
103  endif ()
104  else ()
105  set (Weka_HAS_PACKAGE_MANAGER TRUE)
106  endif ()
107 endmacro ()

110 ## @brief Get list of Weka packages.
111 #
112 # @param [out] PACKAGES Name of variable storing the list of packages.
113 # @param [in] WHICH Argument to -list-packages option of Weka package
114 # manager, i.e., "all", "available", or "installed".
115 function (weka_list_packages PACKAGES WHICH)
116  string (TOLOWER "${WHICH}" WHICH) # ignore case of argument
117  set (PKGS)
118  if (NOT JAVA_EXECUTABLE)
119  set (JAVA_EXECUTABLE java)
120  endif ()
121  if (Weka_VERSION_MAJOR AND Weka_VERSION_MINOR)
122  if (Weka_VERSION_MAJOR EQUAL 3)
123  if (Weka_VERSION_MINOR GREATER 7 OR (Weka_VERSION_MINOR EQUAL 7 AND Weka_VERSION_PATCH GREATER 1))
124  set (HAS_PACKAGE_MANAGER TRUE)
125  else ()
126  set (HAS_PACKAGE_MANAGER FALSE)
127  endif ()
128  elseif (Weka_VERSION_MAJOR GREATER 2)
129  set (HAS_PACKAGE_MANAGER TRUE)
130  else ()
131  set (HAS_PACKAGE_MANAGER FALSE)
132  endif ()
133  else ()
134  set (HAS_PACKAGE_MANAGER TRUE)
135  endif ()
136  if (HAS_PACKAGE_MANAGER)
137  if (Weka_CLASSPATH)
138  execute_process (
139  COMMAND "${JAVA_EXECUTABLE}"
140  -classpath "${Weka_CLASSPATH}"
141  weka.core.WekaPackageManager
142  -list-packages "${WHICH}"
143  RESULT_VARIABLE STATUS
144  OUTPUT_VARIABLE STDOUT
145  ERROR_VARIABLE STDERR
146  )
147  if (STATUS EQUAL 0)
148  string (REPLACE ";" "," STDOUT "${STDOUT}")
149  string (REPLACE "\n" ";" STDOUT "${STDOUT}")
150  foreach (LINE IN LISTS STDOUT)
151  if (LINE MATCHES "^(-+|[0-9]+(\\.[0-9]+(\\.[0-9]+)))[ \t]+[0-9]+(\\.[0-9]+(\\.[0-9]+))[ \t]+([a-zA-Z_-]+):")
152  list (APPEND PKGS "${CMAKE_MATCH_6}")
153  endif ()
154  endforeach ()
155  else ()
156  message (WARNING "Failed to retrieve list of ${WHICH} Weka packages! The command\n"
157  "\"${JAVA_EXECUTABLE}\" -classpath \"${Weka_CLASSPATH}\" weka.core.WekaPackageManager -list-packages ${WHICH}\n"
158  "failed with the error:\n${STDERR}")
159  endif ()
160  else ()
161  message (WARNING "Cannot retrieve list of Weka packages because Weka_CLASSPATH is not set!")
162  endif ()
163  endif ()
164  set (${PACKAGES} "${PKGS}" PARENT_SCOPE)
165 endfunction ()

168 ## @brief Get Weka version.
169 #
170 # @param [out] VERSION Version string of Weka or 0.0.0.
171 # @param [out] MAJOR Major version of Weka or 0.
172 # @param [out] MINOR Minor version of Weka or 0.
173 # @param [out] PATCH Patch number of Weka or 0.
174 function (weka_get_version VERSION MAJOR MINOR PATCH)
175  set (VERSION_STRING "0.0.0")
176  set (VERSION_MAJOR 0)
177  set (VERSION_MINOR 0)
178  set (VERSION_PATCH 0)
179  if (NOT JAVA_EXECUTABLE)
180  set (JAVA_EXECUTABLE java)
181  endif ()
182  if (Weka_CLASSPATH)
183  execute_process (
184  COMMAND "${JAVA_EXECUTABLE}" -classpath "${Weka_CLASSPATH}" weka.core.Version
185  RESULT_VARIABLE RETVAL
186  OUTPUT_VARIABLE STDOUT
187  ERROR_VARIABLE STDERR
188  )
189  if (STDOUT MATCHES "^([0-9]+)\\.([0-9]+)\\.([0-9]+)")
190  set (VERSION_STRING "${CMAKE_MATCH_0}")
191  set (VERSION_MAJOR "${CMAKE_MATCH_1}")
192  set (VERSION_MINOR "${CMAKE_MATCH_2}")
193  set (VERSION_PATCH "${CMAKE_MATCH_3}")
194  else ()
195  message (WARNING "Failed to version of Weka installation. The command\n"
196  "\"${JAVA_EXECUTABLE}\" -classpath \"${Weka_CLASSPATH}\" weka.core.Version\n"
197  "failed with the error:\n${STDERR}")
198  endif ()
199  endif ()
200  set (${VERSION} "${VERSION_STRING}" PARENT_SCOPE)
201  set (${MAJOR} "${VERSION_MAJOR}" PARENT_SCOPE)
202  set (${MINOR} "${VERSION_MINOR}" PARENT_SCOPE)
203  set (${PATCH} "${VERSION_PATCH}" PARENT_SCOPE)
204 endfunction ()
