21 # ----------------------------------------------------------------------------
22 # find BASH executable
23 find_program (BASH_EXECUTABLE bash)
24 mark_as_advanced (BASH_EXECUTABLE)
25 
26 # ----------------------------------------------------------------------------
27 # get version of found BASH executable
28 if (BASH_EXECUTABLE)
29  execute_process (COMMAND "${BASH_EXECUTABLE}" --version OUTPUT_VARIABLE _BASH_STDOUT ERROR_VARIABLE _BASH_STDERR)
30  if (_BASH_STDOUT MATCHES "version ([0-9]+)\\.([0-9]+)\\.([0-9]+)")
31  set (BASH_VERSION_MAJOR "${CMAKE_MATCH_1}")
32  set (BASH_VERSION_MINOR "${CMAKE_MATCH_2}")
33  set (BASH_VERSION_PATCH "${CMAKE_MATCH_3}")
34  set (BASH_VERSION_STRING "${BASH_VERSION_MAJOR}.${BASH_VERSION_MINOR}.${BASH_VERSION_PATCH}")
35  else ()
36  message (WARNING "Failed to determine version of Bash interpreter (${BASH_EXECUTABLE})! Error:\n${_BASH_STDERR}")
37  endif ()
38  unset (_BASH_STDOUT)
39  unset (_BASH_STDERR)
40 endif ()
