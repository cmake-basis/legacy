2 # @file TopologicalSort.cmake
3 # @brief CMake implementation of topological sorting algorithm.
4 #
5 # Perform a reverse topological sort on the given LIST.
6 #
7 # topological_sort(my_list "MY_" "_EDGES")
8 #
9 # LIST is the name of a variable containing a list of elements to be
10 # sorted in reverse topological order. Each element in the list has a
11 # set of outgoing edges (for example, those other list elements that
12 # it depends on). In the resulting reverse topological ordering
13 # (written back into the variable named LIST), an element will come
14 # later in the list than any of the elements that can be reached by
15 # following its outgoing edges and the outgoing edges of any vertices
16 # they target, recursively. Thus, if the edges represent dependencies
17 # on build targets, for example, the reverse topological ordering is
18 # the order in which one would build those targets.
19 #
20 # For each element E in this list, the edges for E are contained in
21 # the variable named ${PREFIX}${E}${SUFFIX}. If no such variable
22 # exists, then it is assumed that there are no edges. For example, if
23 # my_list contains a, b, and c, one could provide a dependency graph
24 # using the following variables:
25 #
26 # MY_A_EDGES b
27 # MY_B_EDGES
28 # MY_C_EDGES a b
29 #
30 # With the involcation of topological_sort shown above and these
31 # variables, the resulting reverse topological ordering will be b, a, c.
32 #
33 # @verbatim

73 function(topological_sort LIST PREFIX SUFFIX)
74  # Clear the stack and output variable
75  set(VERTICES "${${LIST}}")
76  set(STACK)
77  set(${LIST})
78 
79  # Loop over all of the vertices, starting the topological sort from
80  # each one.
81  foreach(VERTEX ${VERTICES})
82 
83  # If we haven't already processed this vertex, start a depth-first
84  # search from where.
85  if (NOT FOUND_${VERTEX})
86  # Push this vertex onto the stack with all of its outgoing edges
87  string(REPLACE ";" " " NEW_ELEMENT
88  "${VERTEX};${${PREFIX}${VERTEX}${SUFFIX}}")
89  list(APPEND STACK ${NEW_ELEMENT})
90 
91  # We've now seen this vertex
92  set(FOUND_${VERTEX} TRUE)
93 
94  # While the depth-first search stack is not empty
95  list(LENGTH STACK STACK_LENGTH)
96  while(STACK_LENGTH GREATER 0)
97  # Remove the vertex and its remaining out-edges from the top
98  # of the stack
99  list(GET STACK -1 OUT_EDGES)
100  list(REMOVE_AT STACK -1)
101 
102  # Get the source vertex and the list of out-edges
103  separate_arguments(OUT_EDGES)
104  list(GET OUT_EDGES 0 SOURCE)
105  list(REMOVE_AT OUT_EDGES 0)
106 
107  # While there are still out-edges remaining
108  list(LENGTH OUT_EDGES OUT_DEGREE)
109  while (OUT_DEGREE GREATER 0)
110  # Pull off the first outgoing edge
111  list(GET OUT_EDGES 0 TARGET)
112  list(REMOVE_AT OUT_EDGES 0)
113 
114  if (NOT FOUND_${TARGET})
115  # We have not seen the target before, so we will traverse
116  # its outgoing edges before coming back to our
117  # source. This is the key to the depth-first traversal.
118 
119  # We've now seen this vertex
120  set(FOUND_${TARGET} TRUE)
121 
122  # Push the remaining edges for the current vertex onto the
123  # stack
124  string(REPLACE ";" " " NEW_ELEMENT
125  "${SOURCE};${OUT_EDGES}")
126  list(APPEND STACK ${NEW_ELEMENT})
127 
128  # Setup the new source and outgoing edges
129  set(SOURCE ${TARGET})
130  set(OUT_EDGES
131  ${${PREFIX}${SOURCE}${SUFFIX}})
132  endif(NOT FOUND_${TARGET})
133 
134  list(LENGTH OUT_EDGES OUT_DEGREE)
135  endwhile (OUT_DEGREE GREATER 0)
136 
137  # We have finished all of the outgoing edges for
138  # SOURCE; add it to the resulting list.
139  list(APPEND ${LIST} ${SOURCE})
140 
141  # Check the length of the stack
142  list(LENGTH STACK STACK_LENGTH)
143  endwhile(STACK_LENGTH GREATER 0)
144  endif (NOT FOUND_${VERTEX})
145  endforeach(VERTEX)
146 
147  set(${LIST} ${${LIST}} PARENT_SCOPE)
148 endfunction(topological_sort)
