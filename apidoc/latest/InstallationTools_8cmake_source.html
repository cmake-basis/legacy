35 # ----------------------------------------------------------------------------
36 ## @brief Install content of source directory excluding typical files.
37 #
38 # Files which are excluded are typical backup files, system files, files
39 # from revision control systems, and CMakeLists.txt files.
40 #
41 # Example:
42 # @code
43 # basis_install_directory("${INSTALL_DATA_DIR}")
44 # basis_install_directory(. "${INSTALL_DATA_DIR}")
45 # basis_install_directory("${CMAKE_CURRENT_SOURCE_DIR}" "${INSTALL_DATA_DIR}")
46 # basis_install_directory(images "${INSTALL_DATA_DIR}/images")
47 # @endcode
48 #
49 # @param [in] ARGN The first two arguments are extracted from the beginning
50 # of this list in the named order (without option name),
51 # and the remaining arguments are passed on to CMake's
52 # <a href="http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:install">
53 # <tt>install(DIRECTORY)</tt></a> command.
54 # @par
55 # <table border="0">
56 # <tr>
57 # @tp @b SOURCE @endtp
58 # <td>Source directory. Defaults to current source directory
59 # if only one argument, the @p DESTINATION, is given./td>
60 # </tr>
61 # <tr>
62 # @tp @b DESTINATION @endtp
63 # <td>Destination directory.</td>
64 # </tr>
65 # </table>
66 #
67 # @sa http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:install
68 #
69 # @ingroup CMakeAPI
70 function (basis_install_directory)
71  if (ARGC EQUAL 1)
72  set (SOURCE "${CMAKE_CURRENT_SOURCE_DIR}")
73  set (DESTINATION "${ARGV0}")
74  set (OPTIONS "${ARGV}")
75  list (REMOVE_AT OPTIONS 0)
76  elseif (ARGC GREATER 1)
77  set (SOURCE "${ARGV0}")
78  set (DESTINATION "${ARGV1}")
79  set (OPTIONS "${ARGV}")
80  list (REMOVE_AT OPTIONS 0 1)
81  else ()
82  message (FATAL_ERROR "Too few arguments given!")
83  endif ()
84  # check arguments
85  if (NOT IS_ABSOLUTE "${SOURCE}")
86  get_filename_component (SOURCE "${SOURCE}" ABSOLUTE)
87  endif ()
88  string (REGEX REPLACE "/+$" "" SOURCE "${SOURCE}")
89  if (NOT IS_ABSOLUTE "${DESTINATION}")
90  set (DESTINATION "${CMAKE_INSTALL_PREFIX}/${DESTINATION}")
91  endif ()
92  basis_sanitize_for_regex (PROJECT_SOURCE_DIR_RE "${PROJECT_SOURCE_DIR}")
93  if ("${DESTINATION}" MATCHES "^${PROJECT_SOURCE_DIR_RE}(/|$)")
94  message (FATAL_ERROR "Installation directory ${DESTINATION} is inside the project source tree!")
95  endif ()
96  # parse options
97  set (MATCH)
98  set (EXCLUDE)
99  set (COMPONENT)
100  set (CONFIGURATIONS)
101  set (RE)
102  set (OPT)
103  set (NUM)
104  foreach (O IN LISTS OPTIONS)
105  if (O MATCHES "^(FILES_MATCHING|.*PERMISSIONS|OPTIONAL|DESTINATION|DIRECTORY)$")
106  message (FATAL_ERROR "Option ${O} not supported by basis_install_directory()!")
107  endif ()
108  if (OPT MATCHES "^(PATTERN|REGEX)$")
109  if (NUM EQUAL 1)
110  if (O MATCHES "^EXCLUDE$")
111  list (APPEND EXCLUDE "${RE}")
112  else ()
113  list (APPEND MATCH "${RE}")
114  endif ()
115  set (OPT)
116  set (RE)
117  else ()
118  if (OPT MATCHES "PATTERN")
119  string (REGEX REPLACE "(^|[^\\])\\*" "&#9734" O "${O}")
120  basis_sanitize_for_regex (O "${O}")
121  string (REPLACE "&#9734" ".*" O "${O}")
122  endif ()
123  set (RE "${O}")
124  set (NUM 1)
125  endif ()
126  elseif (OPT MATCHES "^COMPONENT$")
127  set (COMPONENT "${O}")
128  set (OPT)
129  elseif (OPT MATCHES "^CONFIGURATIONS$")
130  if (O MATCHES "^(PATTERN|REGEX|COMPONENT|CONFIGURATIONS)$")
131  set (OPT)
132  else ()
133  list (APPEND CONFIGURATIONS "${O}")
134  math (EXPR NUM "${NUM} + 1")
135  endif ()
136  endif ()
137  if (O MATCHES "^(PATTERN|REGEX|COMPONENT|CONFIGURATIONS)$")
138  if (OPT)
139  message (FATAL_ERROR "basis_install_directory(): Option ${OPT} is missing an argument!")
140  endif ()
141  set (OPT ${O})
142  set (NUM 0)
143  endif ()
144  endforeach ()
145  if (OPT MATCHES "CONFIGURATIONS" AND NOT NUM GREATER 0)
146  message (FATAL_ERROR "basis_install_directory(): Missing argument(s) for ${OPT} option!")
147  elseif (OPT MATCHES "PATTERN|REGEX")
148  if (NUM EQUAL 1)
149  list (APPEND MATCH "${RE}")
150  else ()
151  message (FATAL_ERROR "basis_install_directory(): Missing argument(s) for ${OPT} option!")
152  endif ()
153  elseif (OPT MATCHES "COMPONENT")
154  message (FATAL_ERROR "basis_install_directory(): Missing argument for ${OPT} option!")
155  endif ()
156  list (APPEND EXCLUDE "CMakeLists.txt$" "/.svn/" "/.git/" ".DS_Store$" ".*~$") # default excludes
157  basis_list_to_delimited_string (MATCH "|" NOAUTOQUOTE ${MATCH})
158  basis_list_to_delimited_string (EXCLUDE "|" NOAUTOQUOTE ${EXCLUDE})
159  string (REPLACE "\\" "\\\\" MATCH "${MATCH}")
160  string (REPLACE "\"" "\\\"" MATCH "${MATCH}")
161  string (REPLACE "\\" "\\\\" EXCLUDE "${EXCLUDE}")
162  string (REPLACE "\"" "\\\"" EXCLUDE "${EXCLUDE}")
163  # Add installation instructions. Note that install(DIRECTORY) is here not
164  # used to avoid the generation of empty directories
165  set (OPTIONS)
166  if (CONFIGURATIONS)
167  list (APPEND OPTIONS CONFIGURATIONS ${CONFIGURATIONS})
168  endif ()
169  if (COMPONENT)
170  list (APPEND OPTIONS COMPONENT ${COMPONENT})
171  endif ()
172  install (CODE
173  "# -----------------------------------------------------------------------
174  # basis_install_directory(): ${SOURCE}
175  set (BASIS_INSTALL_DIRECTORY_FILES)
176  set (BASIS_INSTALL_DIRECTORY_SOURCE \"${SOURCE}\")
177  set (BASIS_INSTALL_DIRECTORY_DESTINATION \"\$ENV{DESTDIR}${DESTINATION}\")
178  set (BASIS_INSTALL_DIRECTORY_MATCH \"${MATCH}\")
179  set (BASIS_INSTALL_DIRECTORY_EXCLUDE \"${EXCLUDE}\")
180  file (GLOB_RECURSE BASIS_INSTALL_DIRECTORY_ALL_FILES \"\${BASIS_INSTALL_DIRECTORY_SOURCE}/*\")
181  foreach (BASIS_INSTALL_DIRECTORY_FILE IN LISTS BASIS_INSTALL_DIRECTORY_ALL_FILES)
182  if (NOT BASIS_INSTALL_DIRECTORY_MATCH OR
183  BASIS_INSTALL_DIRECTORY_FILE MATCHES \"\${BASIS_INSTALL_DIRECTORY_MATCH}\" AND
184  NOT BASIS_INSTALL_DIRECTORY_FILE MATCHES \"\${BASIS_INSTALL_DIRECTORY_EXCLUDE}\")
185  list (APPEND BASIS_INSTALL_DIRECTORY_FILES \"\${BASIS_INSTALL_DIRECTORY_FILE}\")
186  endif ()
187  endforeach ()
188  foreach (BASIS_INSTALL_DIRECTORY_FILE IN LISTS BASIS_INSTALL_DIRECTORY_FILES)
189  file (RELATIVE_PATH BASIS_INSTALL_DIRECTORY_FILE \"\${BASIS_INSTALL_DIRECTORY_SOURCE}\" \"\${BASIS_INSTALL_DIRECTORY_FILE}\")
190  execute_process (
191  COMMAND \"${CMAKE_COMMAND}\" -E compare_files
192  \"\${BASIS_INSTALL_DIRECTORY_SOURCE}/\${BASIS_INSTALL_DIRECTORY_FILE}\"
193  \"\${BASIS_INSTALL_DIRECTORY_DESTINATION}/\${BASIS_INSTALL_DIRECTORY_FILE}\"
194  RESULT_VARIABLE RC
195  OUTPUT_QUIET
196  ERROR_QUIET
197  )
198  if (RC EQUAL 0)
199  message (STATUS \"Up-to-date: \${BASIS_INSTALL_DIRECTORY_DESTINATION}/\${BASIS_INSTALL_DIRECTORY_FILE}\")
200  else ()
201  message (STATUS \"Installing: \${BASIS_INSTALL_DIRECTORY_DESTINATION}/\${BASIS_INSTALL_DIRECTORY_FILE}\")
202  execute_process (
203  COMMAND \"${CMAKE_COMMAND}\" -E copy_if_different
204  \"\${BASIS_INSTALL_DIRECTORY_SOURCE}/\${BASIS_INSTALL_DIRECTORY_FILE}\"
205  \"\${BASIS_INSTALL_DIRECTORY_DESTINATION}/\${BASIS_INSTALL_DIRECTORY_FILE}\"
206  RESULT_VARIABLE RC
207  OUTPUT_QUIET
208  ERROR_QUIET
209  )
210  if (RC EQUAL 0)
211  list (APPEND CMAKE_INSTALL_MANIFEST_FILES \"\${BASIS_INSTALL_DIRECTORY_DESTINATION}/\${BASIS_INSTALL_DIRECTORY_FILE}\")
212  else ()
213  message (STATUS \"Failed to install \${BASIS_INSTALL_DIRECTORY_DESTINATION}/\${BASIS_INSTALL_DIRECTORY_FILE}\")
214  endif ()
215  endif ()
216  endforeach ()
217  # -----------------------------------------------------------------------"
218  ${OPTIONS}
219  )
220 endfunction ()

240 ## @brief Add installation rule to create a symbolic link.
241 #
242 # Note that the installation rule will only be effective on a Unix-like
243 # system, i.e., one which supports the creation of a symbolic link.
244 #
245 # @param [in] OLD The value of the symbolic link.
246 # @param [in] NEW The name of the symbolic link.
247 #
248 # @returns Adds installation rule to create the symbolic link @p NEW.
249 #
250 # @ingroup CMakeAPI
251 function (basis_install_link OLD NEW)
252  # Attention: CMAKE_INSTALL_PREFIX must be used instead of CMAKE_INSTALL_PREFIX.
253  set (CMD_IN
254  "
255  set (OLD \"@OLD@\")
256  set (NEW \"@NEW@\")
257 
258  if (NOT IS_ABSOLUTE \"\${OLD}\")
259  set (OLD \"\$ENV{DESTDIR}\${CMAKE_INSTALL_PREFIX}/\${OLD}\")
260  endif ()
261  if (NOT IS_ABSOLUTE \"\${NEW}\")
262  set (NEW \"\$ENV{DESTDIR}\${CMAKE_INSTALL_PREFIX}/\${NEW}\")
263  endif ()
264 
265  if (IS_SYMLINK \"\${NEW}\")
266  file (REMOVE \"\${NEW}\")
267  endif ()
268 
269  if (EXISTS \"\${NEW}\")
270  message (STATUS \"Skipping: \${NEW} -> \${OLD}\")
271  else ()
272  message (STATUS \"Installing: \${NEW} -> \${OLD}\")
273 
274  get_filename_component (SYMDIR \"\${NEW}\" PATH)
275 
276  file (RELATIVE_PATH OLD \"\${SYMDIR}\" \"\${OLD}\")
277 
278  if (NOT EXISTS \${SYMDIR})
279  file (MAKE_DIRECTORY \"\${SYMDIR}\")
280  endif ()
281 
282  execute_process (
283  COMMAND \"${CMAKE_COMMAND}\" -E create_symlink \"\${OLD}\" \"\${NEW}\"
284  RESULT_VARIABLE RETVAL
285  )
286 
287  if (NOT RETVAL EQUAL 0)
288  message (ERROR \"Failed to create (symbolic) link \${NEW} -> \${OLD}\")
289  else ()
290  list (APPEND CMAKE_INSTALL_MANIFEST_FILES \"\${NEW}\")
291  endif ()
292  endif ()
293  "
294  )
295 
296  string (CONFIGURE "${CMD_IN}" CMD @ONLY)
297  install (CODE "${CMD}")
298 endfunction ()

300 # ----------------------------------------------------------------------------
301 ## @brief Adds installation rules to create default symbolic links.
302 #
303 # This function creates for each main executable a symbolic link directly
304 # in the directory @c INSTALL_PREFIX/bin if @c INSTALL_SINFIX is TRUE and the
305 # software is installed on a Unix-like system, i.e., one which
306 # supports the creation of symbolic links.
307 #
308 # @returns Adds installation command for creation of symbolic links in the
309 # installation tree.
310 function (basis_install_links)
311  if (NOT UNIX)
312  return ()
313  endif ()
314 
315  # main executables
316  basis_get_project_property (TARGETS PROPERTY TARGETS)
317  foreach (TARGET_UID ${TARGETS})
318  get_target_property (IMPORTED ${TARGET_UID} "IMPORTED")
319 
320  if (NOT IMPORTED)
321  get_target_property (BASIS_TYPE ${TARGET_UID} "BASIS_TYPE")
322  get_target_property (LIBEXEC ${TARGET_UID} "LIBEXEC")
323  get_target_property (TEST ${TARGET_UID} "TEST")
324 
325  if (BASIS_TYPE MATCHES "EXECUTABLE" AND NOT LIBEXEC AND NOT TEST)
326  get_target_property (SYMLINK_NAME ${TARGET_UID} "SYMLINK_NAME")
327  if (NOT "${SYMLINK_NAME}" MATCHES "^none$|^None$|^NONE$")
328  get_target_property (SYMLINK_PREFIX ${TARGET_UID} "SYMLINK_PREFIX")
329  get_target_property (SYMLINK_SUFFIX ${TARGET_UID} "SYMLINK_SUFFIX")
330  get_target_property (INSTALL_DIR ${TARGET_UID} "RUNTIME_INSTALL_DIRECTORY")
331  if (NOT INSTALL_DIR)
332  get_target_property (INSTALL_DIR ${TARGET_UID} "INSTALL_DIRECTORY")
333  endif ()
334 
335  basis_get_target_location (OUTPUT_NAME ${TARGET_UID} NAME)
336 
337  if (NOT SYMLINK_NAME)
338  set (SYMLINK_NAME "${OUTPUT_NAME}")
339  endif ()
340  if (SYMLINK_PREFIX)
341  set (SYMLINK_NAME "${SYMLINK_PREFIX}${SYMLINK_NAME}")
342  endif ()
343  if (SYMLINK_SUFFIX)
344  set (SYMLINK_NAME "${SYMLINK_NAME}${SYMLINK_SUFFIX}")
345  endif ()
346 
347  # avoid creation of symbolic link if there would be a conflict with
348  # the subdirectory in bin/ where the actual executables are installed
349  if (INSTALL_SINFIX AND "${SYMLINK_NAME}" STREQUAL "${BASIS_INSALL_SINFIX}")
350  message (STATUS \"Skipping: ${INSTALL_DIR}/${OUTPUT_NAME} -> ${INSTALL_PREFIX}/bin/${SYMLINK_NAME}\")
351  else ()
352  basis_install_link (
353  "${INSTALL_DIR}/${OUTPUT_NAME}"
354  "bin/${SYMLINK_NAME}"
355  )
356  endif ()
357  endif ()
358  endif ()
359  endif ()
360  endforeach ()
361 endfunction ()

368 ## @brief Register installed package with CMake.
369 #
370 # This function adds an entry to the CMake registry for packages with the
371 # path of the directory where the package configuration file is located in
372 # order to help CMake find the package.
373 #
374 # The uninstaller whose template can be found in cmake_uninstaller.cmake.in
375 # is responsible for removing the registry entry again.
376 function (basis_register_package)
377  set (PKGDIR "${CMAKE_INSTALL_PREFIX}/${INSTALL_CONFIG_DIR}")
378  set (PKGUID "${BASIS_PROJECT_PACKAGE_UID}")
379  if (WIN32)
380  install (CODE
381  "execute_process (
382  COMMAND reg add \"HKCU\\\\Software\\\\Kitware\\\\CMake\\\\Packages\\\\${PROJECT_PACKAGE_CONFIG_PREFIX}\" /v \"${PKGUID}\" /d \"${PKGDIR}\" /t REG_SZ /f
383  RESULT_VARIABLE RT
384  ERROR_VARIABLE ERR
385  OUTPUT_QUIET
386  )
387  if (RT EQUAL 0)
388  message (STATUS \"Register: Added HKEY_CURRENT_USER\\\\Software\\\\Kitware\\\\CMake\\\\Packages\\\\${PROJECT_PACKAGE_CONFIG_PREFIX}\\\\${PKGUID}\")
389  else ()
390  string (STRIP \"\${ERR}\" ERR)
391  message (STATUS \"Register: Failed to add registry entry: \${ERR}\")
392  endif ()"
393  )
394  elseif (IS_DIRECTORY "$ENV{HOME}")
395  file (WRITE "${PROJECT_BINARY_DIR}/${PROJECT_PACKAGE_CONFIG_PREFIX}RegistryFile" "${PKGDIR}")
396  install (
397  FILES "${PROJECT_BINARY_DIR}/${PROJECT_PACKAGE_CONFIG_PREFIX}RegistryFile"
398  DESTINATION "$ENV{HOME}/.cmake/packages/${PROJECT_PACKAGE_CONFIG_PREFIX}"
399  RENAME "${PKGUID}"
400  )
401  endif ()
402 endfunction ()
