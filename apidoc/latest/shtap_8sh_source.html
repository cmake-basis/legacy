12 # Plan:
13 # @code
14 # plan_no_plan
15 # plan_skip_all [REASON]
16 # plan_tests NB_TESTS
17 # @endcode
18 #
19 # Test:
20 # @code
21 # ok RESULT [NAME]
22 # okx COMMAND
23 # is RESULT EXPECTED [NAME]
24 # isnt RESULT EXPECTED [NAME]
25 # like RESULT PATTERN [NAME]

27 # pass [NAME]
28 # fail [NAME]
29 # @endcode
30 #
31 # Skip:
32 # @code
33 # skip [CONDITION] [REASON] [NB_TESTS=1]
34 #
35 # skip $feature_not_present "feature not present" 2 ||
36 # {
37 # is $a "a"
38 # is $b "b"
39 # }
40 # @endcode
41 #
42 # Specify TODO mode by setting the TODO variable:
43 # @code
44 # TODO="not implemented yet"
45 # ok $result "some not implemented test"

47 # @endcode
48 #
49 # Other:
50 # @code
51 # diag MSG
52 # @endcode
53 #
54 # Example:
55 # @code
56 # #! /usr/bin/env bash
57 #
58 # source shtap.sh
59 #
60 # plan_tests 7
61 #
62 # # test identity
63 # {
64 # me=${USER}
65 # is ${USER} ${me} "I am myself"

68 # }
69 #

76 # }

224 # ----------------------------------------------------------------------------
225 ## @brief Evaluate test expression and fail if it does not evaluate to 0 or
226 # check return value of function.
227 #
228 # This is the workhorse method that actually prints the tests result.
229 #
230 # @param [in] expression Test expression or return value.
231 # @param [in] name Name of test.
232 function ok
233 {
234  local expression=${1:?}
235  local name=${2:-''}
236 
237  [ ${_shtap_plan_set} -eq 0 ] && _shtap_die "You tried to run a test without a plan! Gotta have a plan."
238 
239  (( _shtap_executed_tests++ ))
240 
241  if [ -n "${name}" ]; then
242  if match "${name}" "^[0-9]+$"; then
243  diag " You named your test '${name}'. You shouldn't use numbers for your test names."
244  diag " Very confusing."
245  fi
246  fi
247 
248  local match=`expr "${expression}" : '\([0-9]*\)'`
249  local result=0
250  if [ -z "${expression}" ]; then
251  result=1
252  elif [ -n "${match}" -a "${expression}" = "${match}" ]; then
253  [ ${expression} -ne 0 ] && result=1
254  else
255  ( eval ${expression} ) >/dev/null 2>&1
256  [ $? -ne 0 ] && result=1
257  fi
258 
259  if [ ${result} -ne 0 ]; then
260  echo -n "not "
261  (( _shtap_failed_tests++ ))
262  fi
263  echo -n "ok ${_shtap_executed_tests}"
264 
265  if [ -n "${name}" ]; then
266  local ename=${name//\#/\\#}
267  echo -n " - ${ename}"
268  fi
269 
270  if [ -n "${TODO}" ]; then
271  echo -n " # TODO ${TODO}" ;
272  if [ ${result} -ne 0 ]; then
273  (( _shtap_failed_tests-- ))
274  fi
275  fi
276 
277  echo
278  if [ ${result} -ne 0 ]; then
279  local file='_SHTAP_FILENAME'
280  local func=
281  local line=
282 
283  local i=0
284  local bt=$(caller ${i})
285  while match "${bt}" "${_SHTAP_FILENAME}$"; do
286  (( i++ ))
287  bt=$(caller ${i})
288  done
289  local backtrace=
290  eval $(caller ${i} | (read line func file; echo "backtrace=\"${file}:${func}() at line ${line}.\""))
291 
292  local t=
293  [ -n "${TODO}" ] && t="(TODO) "
294 
295  if [ -n "${name}" ]; then
296  diag " Failed ${t}test '${name}'"
297  diag " in ${backtrace}"
298  else
299  diag " Failed ${t}test in ${backtrace}"
300  fi
301  fi
302 
303  return ${result}
304 }

541 # ----------------------------------------------------------------------------
542 ## @brief Cleaning up after execution of tests and see if plan was fulfilled.
543 function _shtap_cleanup
544 {
545  local rc=0 # return code
546 
547  if [ ${_shtap_plan_set} -eq 0 ]; then
548  diag "Looks like your test died before it could output anything."
549  return ${rc}
550  fi
551 
552  if [ ${_shtap_test_died} -ne 0 ]; then
553  diag "Looks like your test died just after ${_shtap_executed_tests}."
554  return ${rc}
555  fi
556 
557  if [ ${_shtap_skip_all} -eq 0 -a ${_shtap_no_plan} -ne 0 ]; then
558  _shtap_print_plan ${_shtap_executed_tests}
559  fi
560 
561  local s=
562  if [ ${_shtap_no_plan} -eq 0 -a ${_shtap_expected_tests} -lt ${_shtap_executed_tests} ]; then
563  s=''
564  [ ${_shtap_expected_tests} -gt 1 ] && s='s'
565  local extra=$(( _shtap_executed_tests - _shtap_expected_tests ))
566  diag "Looks like you planned ${_shtap_expected_tests} test${s} but ran ${extra} extra."
567  rc=-1 ;
568  fi
569 
570  if [ ${_shtap_no_plan} -eq 0 -a ${_shtap_expected_tests} -gt ${_shtap_executed_tests} ]; then
571  s=''
572  [ ${_shtap_expected_tests} -gt 1 ] && s='s'
573  diag "Looks like you planned ${_shtap_expected_tests} test${s} but only ran ${_shtap_executed_tests}."
574  fi
575 
576  if [ ${_shtap_failed_tests} -gt 0 ]; then
577  s=''
578  [ ${_shtap_failed_tests} -gt 1 ] && s='s'
579  diag "Looks like you failed ${_shtap_failed_tests} test${s} of ${_shtap_executed_tests}."
580  fi
581 
582  return ${rc}
583 }

587 function _shtap_exit_status
