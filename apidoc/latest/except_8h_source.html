30 /**
31  * @brief Throw exception with given message.
32  *
33  * Example:
34  * @code
35  * void func(int i) {
36  * if (i < 0) BASIS_THROW(std::invalid_argument, "Argument i (= " << i << ") must be positive");
37  * }
38  * @endcode
39  *
40  * @param [in] type The type of the exception. Note that this exception
41  * type has to implement a constructor with one std::string
42  * as argument, the exception message.
43  * @param [in] msg The exception message. The given argument is streamed
44  * into a std::ostringstream.
45  */
46 #define BASIS_THROW(type, msg) \
47  { \
48  ::std::ostringstream oss; \
49  oss << msg; \
50  throw type(oss.str().c_str()); \
51  }
