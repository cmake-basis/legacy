39  /**
40  * @brief Handles lists of Arg's that are to be XOR'd on the command-line.
41  *
42  * This subclass of the TCLAP::XorHandler overloads the check() method such
43  * that XOR'd arguments where none of the arguments is required are handled
44  * correctly. The TCLA::XorHandler and TCLAP::CmdLine implementations imply
45  * that all XOR'd arguments are required, i.e., that one of the mutual
46  * exclusive arguments need to be specified. The sbia::basis::XorHandler and
47  * sbia::basis::CmdLine implementations, on the other side, do not require
48  * that any of the XOR'd arguments be given on the command-line if none of
49  * the XOR'd arguments is required.
50  */
51  class XorHandler : public TCLAP::XorHandler
52  {
53  public:
54 
55  /**
56  * @brief Constructor.
57  */
58  XorHandler() {}
59 
60  /**
61  * @brief Checks whether the specified Arg is in one of the xor lists.
62  *
63  * If the argument does match one, this function returns the size of the xor
64  * list that the Arg matched if the Arg is required. If the Arg matches,
65  * then it also sets the rest of the Arg's in the list.
66  *
67  * @param a The Arg to be checked.
68  */
69  int check(const Arg* a)
70  {
71  int n = TCLAP::XorHandler::check(a);
72  return a->isRequired() ? n : 0;
73  }
