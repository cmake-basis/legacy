64 #elif WINDOWS
65  LPTSTR buffer = NULL;

69

71  newbuf = static_cast<LPTSTR>(realloc(buffer, buflen * sizeof(TCHAR)));
72  if (!newbuf) break;
73  buffer = newbuf;

169 // common implementation of mkdir() and makedirs()
170 static inline bool makedir(const string& path, bool parent)
171 {
172  if (path.empty()) return true; // cwd already exists
173  if (path::isfile(path)) return false;
174  vector<string> dirs;
175  string dir(path);
176  if (parent) {
177  while (!dir.empty() && !path::exists(dir)) {
178  dirs.push_back(dir);
179  dir = path::dirname(dir);
180  }
181  } else if (!path::exists(dir)) {
182  dirs.push_back(dir);
183  }
184  for (vector<string>::reverse_iterator it = dirs.rbegin(); it != dirs.rend(); ++it) {
185 #if WINDOWS
186  if (CreateDirectory(it->c_str(), NULL) == FALSE) return false;
187 #else
188  if (::mkdir(it->c_str(), 0755) != 0) return false;
189 #endif
190  }
191  return true;
192 }

234 bool emptydir(const string& path)
235 {
236  bool ok = true;
237  string subpath; // either subdirectory or file path
238 
239 #if WINDOWS
240  WIN32_FIND_DATA info;
241  HANDLE hFile = ::FindFirstFile(path::join(path, "*.*").c_str(), &info);
242  if (hFile != INVALID_HANDLE_VALUE) {
243  do {
244  // skip '.' and '..'
245  if (strncmp(info.cFileName, ".", 2) == 0 || strncmp(info.cFileName, "..", 3) == 0) {
246  continue;
247  }
248  // remove subdirectory or file, respectively
249  subpath = path::join(path, info.cFileName);
250  if(info.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) {
251  if (!removedir(subpath, true)) ok = false;
252  } else {
253  if (::SetFileAttributes(subpath.c_str(), FILE_ATTRIBUTE_NORMAL) == FALSE ||
254  ::DeleteFile(subpath.c_str()) == FALSE) ok = false;
255  }
256  } while (::FindNextFile(hFile, &info) == TRUE);
257  ::FindClose(hFile);
258  }
259 #else
260  struct dirent *p = NULL;
261  DIR *d = opendir(path.c_str());
262  if (d != NULL) {
263  while ((p = readdir(d)) != NULL) {
264  // skip '.' and '..'
265  if (strncmp(p->d_name, ".", 2) == 0 || strncmp(p->d_name, "..", 3) == 0) {
266  continue;
267  }
268  // remove subdirectory or file, respectively
269  subpath = path::join(path, p->d_name);
270  if (path::isdir(subpath)) {
271  if (!rmtree(subpath)) ok = false;
272  } else {
273  if (unlink(subpath.c_str()) != 0) ok = false;
274  }
275  }
276  closedir(d);
277  }
278 #endif
279  return ok;
280 }
