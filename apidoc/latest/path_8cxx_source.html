63 static inline string replace(string str, char from, char to)
64 {
65  string res(str.size(), '\0');
66  string::const_iterator in = str.begin();
67  string::iterator out = res.begin();
68  while (in != str.end()) {
69  if (*in == from) *out = to;
70  else *out = *in;
71  in++; out++;
72  }
73  return res;
74 }

77 string normpath(const string& path)
78 {
79  if (path.empty()) return "";
80  char drive[3] = {'\0', ':', '\0'};
81  size_t i = 0;
82  #if WINDOWS
83  if (path.size() > 1 && path[1] == ':') {
84  drive[0] = path[0];
85  i = 2;
86  }
87  #endif
88  string norm_path = drive;
89  bool abs = issep(path[i]);
90  if (abs) {
91  #if WINDOWS
92  while (i <= path.size() && issep(path[i])) {
93  norm_path += cSeparator;
94  i++;
95  }
96  #else
97  norm_path += cSeparator;
98  #endif
99  }
100  string current;
101  vector<string> parts;
102  while (i <= path.size()) {
103  if (issep(path[i]) || path[i] == '\0') {
104  if (current == "..") {
105  if (!abs && (parts.empty() || parts.back() == "..")) {
106  parts.push_back(current);
107  } else if (!parts.empty()) {
108  parts.pop_back();
109  }
110  } else if (current != "" && current != ".") {
111  parts.push_back(current);
112  }
113  current.clear();
114  } else {
115  current += path[i];
116  }
117  i++;
118  }
119  for (i = 0; i < parts.size(); i++) {
120  norm_path = join(norm_path, parts[i]);
121  }
122  return norm_path.empty() ? "." : norm_path;
123 }

160 void split(const string& path, string& head, string& tail)
161 {
162  size_t last = path.find_last_of(cSeparators);
163  if (last == string::npos) {
164  head = "";
165  tail = path;
166  } else {
167  size_t pos = last;
168  if (last > 0) pos = path.find_last_not_of(cSeparators, last - 1);
169  if (pos == string::npos) head = path.substr(0, last + 1);
170  else head = path.substr(0, pos + 1);
171  tail = path.substr(last + 1);
172  }
173 }

208 void splitext(const string& path, string& head, string& ext, const set<string>* exts, bool icase)
209 {
210  size_t pos = string::npos;
211  // test user supplied extensions only
212  if (exts) {
213  for (set<string>::const_iterator i = exts->begin(); i != exts->end(); ++i) {
214  if (path.size() < i->size()) continue;
215  size_t start = path.size() - i->size();
216  if (start < pos) { // longest match
217  if (icase) {
218  string str = path.substr(start);
219  string ext = *i;
220  std::transform(str.begin(), str.end(), str.begin(), ::toupper);
221  std::transform(ext.begin(), ext.end(), ext.begin(), ::toupper);
222  if (str == ext) pos = start;
223  } else if (path.compare(start, i->size(), *i) == 0) {
224  pos = start;
225  }
226  }
227  }
228  // otherwise, get position of last dot
229  } else {
230  pos = path.find_last_of('.');
231  // leading dot of file name in Posix indicates hidden file,
232  // not start of file extension
233  #if UNIX
234  if (pos != string::npos && (pos == 0 || issep(path[pos - 1]))) {
235  pos = string::npos;
236  }
237  #endif
238  }
239  // split extension
240  if (pos == string::npos) {
241  head = path;
242  ext = "";
243  } else {
244  // tmp variable used for the case that head references the same input
245  // string as path
246  string tmp = path.substr(0, pos);
247  ext = path.substr(pos);
248  head = tmp;
249  }
250 }

304 string relpath(const string& path, const string& base)
305 {
306  // if relative path is given just return it
307  if (!isabs(path)) return path;
308  // normalize paths
309  string norm_path = normpath(path);
310  string norm_base = normpath(join(getcwd(), base));
311  // check if paths are on same drive
312  #if WINDOWS
313  string drive = splitdrive(norm_path)[0];
314  string base_drive = splitdrive(norm_base)[0];
315  if (drive != base_drive) {
316  BASIS_THROW(invalid_argument,
317  "Path is on drive " << drive << ", start is on drive " << base_drive);
318  }
319  #endif
320  // find start of first path component in which paths differ
321  string::const_iterator b = norm_base.begin();
322  string::const_iterator p = norm_path.begin();
323  size_t pos = 0;
324  size_t i = 0;
325  while (b != norm_base.end() && p != norm_path.end()) {
326  if (issep(*p)) {
327  if (!issep(*b)) break;
328  pos = i;
329  } else if (*b != *p) {
330  break;
331  }
332  b++; p++; i++;
333  }
334  // set pos to i (in this case, the size of one of the paths) if the end
335  // of one path was reached, but the other path has a path separator
336  // at this position, this is required below
337  if ((b != norm_base.end() && issep(*b)) ||
338  (p != norm_path.end() && issep(*p))) pos = i;
339  // skip trailing separator of other path if end of one path reached
340  if (b == norm_base.end() && p != norm_path.end() && issep(*p)) p++;
341  if (p == norm_path.end() && b != norm_base.end() && issep(*b)) b++;
342  // if paths are the same, just return a period (.)
343  //
344  // Thanks to the previous skipping of trailing separators, this condition
345  // handles all of the following cases:
346  //
347  // base := "/usr/bin" path := "/usr/bin"
348  // base := "/usr/bin/" path := "/usr/bin/"
349  // base := "/usr/bin" path := "/usr/bin/"
350  // base := "/usr/bin/" path := "/usr/bin"
351  if (b == norm_base.end() && p == norm_path.end()) return ".";
352  // otherwise, pos is the index of the last slash for which both paths
353  // were identical; hence, everything that comes after in the original
354  // path is preserved and for each following component in the base path
355  // a "../" is prepended to the relative path
356  string rel_path;
357  // truncate base path with a separator as for each "*/" path component,
358  // a "../" will be prepended to the relative path
359  if (b != norm_base.end() && !issep(norm_base[norm_base.size() - 1])) {
360  // attention: This operation may invalidate the iterator b!
361  // Therefore, remember position of iterator and get a new one.
362  size_t pos = b - norm_base.begin();
363  norm_base += cSeparator;
364  b = norm_base.begin() + pos;
365  }
366  while (b != norm_base.end()) {
367  if (issep(*b)) {
368  rel_path += "..";
369  rel_path += cSeparator;
370  }
371  b++;
372  }
373  if (pos + 1 < norm_path.size()) rel_path += norm_path.substr(pos + 1);
374  // remove trailing path separator
375  if (issep(rel_path[rel_path.size() - 1])) {
376  rel_path.erase(rel_path.size() - 1);
377  }
378  return rel_path;
379 }

382 string realpath(const string& path)
383 {
384  string curr_path = join(getcwd(), path);
385  #if UNIX
386  // use stringstream and std::getline() to split absolute path at slashes (/)
387  stringstream ss(curr_path);
388  curr_path.clear();
389  string fname;
390  string prev_path;
391  string next_path;
392  char slash;
393  ss >> slash; // root slash
394  while (getline(ss, fname, '/')) {
395  // current absolute path
396  curr_path += '/';
397  curr_path += fname;
398  // if current path is a symbolic link, follow it
399  if (islink(curr_path)) {
400  // for safety reasons, restrict the depth of symbolic links followed
401  for (unsigned int i = 0; i < 100; i++) {
402  next_path = os::readlink(curr_path);
403  if (next_path.empty()) {
404  // if real path could not be determined because of permissions
405  // or invalid path, return the original path
406  break;
407  } else {
408  curr_path = join(prev_path, next_path);
409  if (!islink(next_path)) break;
410  }
411  }
412  // if real path could not be determined with the given maximum number
413  // of loop iterations (endless cycle?) or one of the symbolic links
414  // could not be read, just return original path as absolute path
415  if (islink(next_path)) {
416  return abspath(path);
417  }
418  }
419  // memorize previous path used as base for abspath()
420  prev_path = curr_path;
421  }
422  #endif
423  // normalize path after all symbolic links were resolved
424  return normpath(curr_path);
425 }
