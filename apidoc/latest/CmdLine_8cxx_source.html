330 void StdOutput::printUsage(ostream& os, bool heading) const
331 {
332  string exec_name = os::exename();
333  list<TCLAP::Arg*> args = _cmd->getArgList();
334  TCLAP::XorHandler& xorhandler = _cmd->getXorHandler();
335  vector< vector<TCLAP::Arg*> > xors = xorhandler.getXorList();
336 
337  // separate into argument groups
338  vector< vector<TCLAP::Arg*> > reqxors;
339  vector< vector<TCLAP::Arg*> > optxors;
340  for (int i = 0; static_cast<unsigned int>(i) < xors.size(); i++) {
341  if (xors[i].size() > 0) {
342  if (xors[i][0]->isRequired()) reqxors.push_back(xors[i]);
343  else optxors.push_back(xors[i]);
344  }
345  }
346  list<TCLAP::Arg*> reqargs, reqposargs;
347  list<TCLAP::Arg*> optargs, optposargs;
348  for (TCLAP::ArgListIterator it = args.begin(); it != args.end(); it++) {
349  if (_stdargs.find((*it)->getName()) == _stdargs.end()
350  && !xorhandler.contains((*it))) {
351  if (isUnlabeledArg(*it)) {
352  if ((*it)->isRequired()) {
353  (*it)->addToList(reqposargs);
354  } else {
355  (*it)->addToList(optposargs);
356  }
357  } else {
358  if ((*it)->isRequired()) {
359  (*it)->addToList(reqargs);
360  } else {
361  (*it)->addToList(optargs);
362  }
363  }
364  }
365  }
366 
367  // executable name
368  string s = exec_name;
369  string id;
370  // optional arguments with flags
371  for (int i = 0; static_cast<unsigned int>(i) < optxors.size(); i++) {
372  s += " [";
373  for (TCLAP::ArgVectorIterator it = optxors[i].begin();
374  it != optxors[i].end(); it++) {
375  id = getArgumentID(*it);
376  s += id;
377  if ((*it)->acceptsMultipleValues() && id.find("...") == string::npos) {
378  s += "...";
379  }
380  s += "|";
381  }
382  s[s.length() - 1] = ']';
383  }
384  for (TCLAP::ArgListIterator it = optargs.begin(); it != optargs.end(); it++) {
385  id = getArgumentID(*it);
386  s += " [";
387  s += id;
388  s += "]";
389  if ((*it)->acceptsMultipleValues() && id.find("...") == string::npos) {
390  s += "...";
391  }
392  }
393  // required arguments with flags
394  for (int i = 0; static_cast<unsigned int>(i) < reqxors.size(); i++) {
395  s += " (";
396  for (TCLAP::ArgVectorIterator it = reqxors[i].begin();
397  it != reqxors[i].end(); it++) {
398  id = getArgumentID(*it);
399  s += id;
400  if ((*it)->acceptsMultipleValues() && id.find("...") == string::npos) {
401  s += "...";
402  }
403  s += "|";
404  }
405  s[s.length() - 1] = ')';
406  }
407  for (TCLAP::ArgListIterator it = reqargs.begin(); it != reqargs.end(); it++) {
408  id = getArgumentID(*it);
409  s += " ";
410  s += id;
411  if ((*it)->acceptsMultipleValues() && id.find("...") == string::npos) {
412  s += "...";
413  }
414  }
415  // required positional arguments
416  for (TCLAP::ArgListIterator it = reqposargs.begin(); it != reqposargs.end(); it++) {
417  id = getArgumentID(*it);
418  s += " ";
419  s += id;
420  if ((*it)->acceptsMultipleValues() && id.find("...") == string::npos) {
421  s += "...";
422  }
423  }
424  // optional positional arguments
425  for (TCLAP::ArgListIterator it = optposargs.begin(); it != optposargs.end(); it++) {
426  id = getArgumentID(*it);
427  s += " [";
428  s += id;
429  s += "]";
430  if ((*it)->acceptsMultipleValues() && id.find("...") == string::npos) {
431  s += "...";
432  }
433  }
434 
435  // print usage with proper number of columns
436  // if the program name is too long, then adjust the second line offset
437  if (heading) {
438  os << "SYNOPSIS" << endl;
439  }
440  int offset = static_cast<int>(exec_name.length()) + 1;
441  if (offset > _columns / 2) offset = 8;
442  print_wrapped(os, s, _columns, 4, offset);
443  print_wrapped(os, exec_name + " [-h|--help|--helpshort|--helpxml|--helpman|--version]", _columns, 4, offset);
444 }

457 void StdOutput::printArguments(ostream& os, bool all) const
458 {
459  list<TCLAP::Arg*> args = _cmd->getArgList();
460  TCLAP::XorHandler& xorhandler = _cmd->getXorHandler();
461  vector< vector<TCLAP::Arg*> > xors = xorhandler.getXorList();
462 
463  // separate into argument groups
464  vector< vector<TCLAP::Arg*> > reqxors;
465  vector< vector<TCLAP::Arg*> > optxors;
466  for (int i = 0; static_cast<unsigned int>(i) < xors.size(); i++) {
467  if (xors[i].size() > 0) {
468  if (xors[i][0]->isRequired()) reqxors.push_back(xors[i]);
469  else optxors.push_back(xors[i]);
470  }
471  }
472  list<TCLAP::Arg*> reqargs, reqposargs;
473  list<TCLAP::Arg*> optargs, optposargs;
474  list<TCLAP::Arg*> stdargs;
475  for (TCLAP::ArgListIterator it = args.begin(); it != args.end(); it++) {
476  if (_stdargs.find((*it)->getName()) != _stdargs.end()) {
477  (*it)->addToList(stdargs);
478  } else if (!xorhandler.contains((*it))) {
479  if (isUnlabeledArg(*it)) {
480  if ((*it)->isRequired()) {
481  (*it)->addToList(reqposargs);
482  } else {
483  (*it)->addToList(optposargs);
484  }
485  } else {
486  if ((*it)->isRequired()) {
487  (*it)->addToList(reqargs);
488  } else {
489  (*it)->addToList(optargs);
490  }
491  }
492  }
493  }
494 
495  // return if command has no arguments
496  if (xors.empty() && reqargs.empty() && optargs.empty()) {
497  return;
498  }
499 
500  os << endl;
501  os << "OPTIONS" << endl;
502 
503  // required arguments
504  if (!reqxors.empty() || !reqargs.empty() || !reqposargs.empty()) {
505  os << " Required arguments:" << endl;
506  for (TCLAP::ArgListIterator it = reqposargs.begin(); it != reqposargs.end(); it++) {
507  if (it != reqposargs.begin()) os << endl;
508  printArgumentHelp(os, *it);
509  }
510  for (int i = 0; static_cast<unsigned int>(i) < reqxors.size(); i++) {
511  if (i > 0 || !reqposargs.empty()) os << endl;
512  for (TCLAP::ArgVectorIterator it = reqxors[i].begin();
513  it != reqxors[i].end(); it++) {
514  if (it != reqxors[i].begin()) {
515  os << " or ";
516  printArgumentHelp(os, *it, false);
517  } else {
518  printArgumentHelp(os, *it);
519  }
520  }
521  }
522  for (TCLAP::ArgListIterator it = reqargs.begin(); it != reqargs.end(); it++) {
523  if (!reqxors.empty() || it != reqargs.begin()) os << endl;
524  printArgumentHelp(os, *it);
525  }
526  }
527 
528  // optional arguments
529  if (!optxors.empty() || !optargs.empty()) {
530  if (!reqxors.empty() || !reqargs.empty() || !reqposargs.empty()) {
531  os << endl;
532  }
533  os << " Optional arguments:" << endl;
534  for (TCLAP::ArgListIterator it = optposargs.begin(); it != optposargs.end(); it++) {
535  if (it != optposargs.begin()) os << endl;
536  printArgumentHelp(os, *it);
537  }
538  for (int i = 0; static_cast<unsigned int>(i) < optxors.size(); i++) {
539  if (i > 0 || !optposargs.empty()) os << endl;
540  for (TCLAP::ArgVectorIterator it = optxors[i].begin();
541  it != optxors[i].end(); it++) {
542  if (it != optxors[i].begin()) {
543  os << " or ";
544  printArgumentHelp(os, *it, false);
545  } else {
546  printArgumentHelp(os, *it);
547  }
548  }
549  }
550  for (TCLAP::ArgListIterator it = optargs.begin(); it != optargs.end(); it++) {
551  if (!optxors.empty() || it != optargs.begin()) os << endl;
552  printArgumentHelp(os, *it);
553  }
554  }
555 
556  // standard arguments
557  if (all && !stdargs.empty()) {
558  if (!xors.empty() || !reqargs.empty() || !optargs.empty()) {
559  os << endl;
560  }
561  os << " Standard arguments:" << endl;
562  for (TCLAP::ArgListIterator it = stdargs.begin(); it != stdargs.end(); it++) {
563  if (it != stdargs.begin()) os << endl;
564  printArgumentHelp(os, *it);
565  }
566  }
567 }

570 void StdOutput::printExample(ostream& os) const
571 {
572  const string exec_name = os::exename();
573  const vector<string>& examples = _cmd->getExamples();
574 
575  if (!examples.empty()) {
576  os << endl;
577  os << "EXAMPLE" << endl;
578  for (vector<string>::const_iterator it = examples.begin();
579  it != examples.end(); ++it) {
580  if (it != examples.begin()) os << endl;
581  string example = *it;
582  string::size_type pos;
583  // backwards compatibility
584  pos = 0;
585  while ((pos = example.find("EXECNAME", pos)) != string::npos) {
586  example.replace(pos, 8, exec_name);
587  }
588  // desired placeholder as it relates to the exename() function
589  pos = 0;
590  while ((pos = example.find("EXENAME", pos)) != string::npos) {
591  example.replace(pos, 7, exec_name);
592  }
593  print_wrapped(os, example, _columns, 4, 4);
594  }

940 // -----------------------------------------------------------------------
941 void CmdLine::parse(vector<string>& args)
942 {
943  bool shouldExit = false;
944  int estat = 0;
945 
946  try {
947  _progName = os::exename();
948  args.erase(args.begin());
949 
950  int requiredCount = 0;
951  for (int i = 0; static_cast<unsigned int>(i) < args.size(); i++) {
952  bool matched = false;
953  for (TCLAP::ArgListIterator it = _argList.begin(); it != _argList.end(); it++) {
954  if ((*it)->processArg(&i, args)) {
955  requiredCount += _xorHandler.check(*it);
956  matched = true;
957  break;
958  }
959  }
960  if (!matched && _emptyCombined(args[i])) matched = true;
961  if (!matched && !TCLAP::Arg::ignoreRest()) {
962  throw TCLAP::CmdLineParseException("Couldn't find match for argument", args[i]);
963  }
964  }
965 
966  if (requiredCount < _numRequired) {
967  string args;
968  for (TCLAP::ArgListIterator it = _argList.begin(); it != _argList.end(); it++) {
969  if ((*it)->isRequired() && !(*it)->isSet()) {
970  args += (*it)->getName();
971  args += ", ";
972  }
973  }
974  args = args.substr(0, args.length() - 2);
975  string msg = string("Not all required arguments specified, missing: ") + args;
976  throw CmdLineParseException(msg);
977  }
978  if (requiredCount > _numRequired) {
979  throw TCLAP::CmdLineParseException("Too many arguments given!");
980  }
981 
982  } catch (TCLAP::ArgException& e) {
983  if (!_handleExceptions) throw;
984  try {
985  _output->failure(*this, e);
986  } catch (TCLAP::ExitException& ee) {
987  estat = ee.getExitStatus();
988  shouldExit = true;
989  }
990  } catch (TCLAP::ExitException& ee) {
991  if (!_handleExceptions) throw;
992  estat = ee.getExitStatus();
993  shouldExit = true;
994  }
995 
996  if (shouldExit) exit(estat);
997 }
