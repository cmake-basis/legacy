33 ## @brief Set properties on a target.
34 #
35 # This function replaces CMake's
36 # <a href="http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:set_target_properties">
37 # set_target_properties()</a> command and extends its functionality.
38 # In particular, it maps the given target names to the corresponding target UIDs.
39 #
40 # @note Due to a bug in CMake (http://www.cmake.org/Bug/view.php?id=12303),
41 # except of the first property given directly after the @c PROPERTIES keyword,
42 # only properties listed in @c BASIS_PROPERTIES_ON_TARGETS can be set.
43 #
44 # @param [in] ARGN List of arguments. See
45 # <a href="http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:set_target_properties">
46 # set_target_properties()</a>.
47 #
48 # @returns Sets the specified properties on the given target.
49 #
50 # @sa http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:set_target_properties
51 #
52 # @ingroup CMakeAPI
53 function (basis_set_target_properties)
54  # convert target names to UIDs
55  set (TARGET_UIDS)
56  list (LENGTH ARGN N)
57  if (N EQUAL 0)
58  message (FATAL_ERROR "basis_set_target_properties(): Missing arguments!")
59  endif ()
60  list (GET ARGN 0 ARG)
61  while (NOT ARG MATCHES "^PROPERTIES$")
62  basis_get_target_uid (TARGET_UID "${ARG}")
63  list (APPEND TARGET_UIDS "${TARGET_UID}")
64  list (REMOVE_AT ARGN 0)
65  list (LENGTH ARGN N)
66  if (N EQUAL 0)
67  break ()
68  else ()
69  list (GET ARGN 0 ARG)
70  endif ()
71  endwhile ()
72  if (NOT ARG MATCHES "^PROPERTIES$")
73  message (FATAL_ERROR "Missing PROPERTIES argument!")
74  elseif (NOT TARGET_UIDS)
75  message (FATAL_ERROR "No target specified!")
76  endif ()
77  # remove PROPERTIES keyword
78  list (REMOVE_AT ARGN 0)
79  math (EXPR N "${N} - 1")
80  # set targets properties
81  #
82  # Note: By iterating over the properties, the empty property values
83  # are correctly passed on to CMake's set_target_properties()
84  # command, while
85  # _set_target_properties(${TARGET_UIDS} PROPERTIES ${ARGN})
86  # (erroneously) discards the empty elements in ARGN.
87  if (BASIS_DEBUG)
88  message ("** basis_set_target_properties:")
89  message ("** Target(s): ${TARGET_UIDS}")
90  message ("** Properties: [${ARGN}]")
91  endif ()
92  while (N GREATER 1)
93  list (GET ARGN 0 PROPERTY)
94  list (GET ARGN 1 VALUE)
95  list (REMOVE_AT ARGN 0 1)
96  list (LENGTH ARGN N)
97  # The following loop is only required b/c CMake's ARGV and ARGN
98  # lists do not support arguments which are themselves lists.
99  # Therefore, we need a way to decide when the list of values for a
100  # property is terminated. Hence, we only allow known properties
101  # to be set, except for the first property where the name follows
102  # directly after the PROPERTIES keyword.
103  while (N GREATER 0)
104  list (GET ARGN 0 ARG)
105  if (ARG MATCHES "${BASIS_PROPERTIES_ON_TARGETS_RE}")
106  break ()
107  endif ()
108  list (APPEND VALUE "${ARG}")
109  list (REMOVE_AT ARGN 0)
110  list (LENGTH ARGN N)
111  endwhile ()
112  if (BASIS_DEBUG)
113  message ("** -> ${PROPERTY} = [${VALUE}]")
114  endif ()
115  # check property name
116  if (PROPERTY MATCHES "^$") # remember: STREQUAL is buggy and evil!
117  message (FATAL_ERROR "Empty property name given!")
118  endif ()
119  # set target property
120  _set_target_properties (${TARGET_UIDS} PROPERTIES ${PROPERTY} "${VALUE}")
121  endwhile ()
122  # make sure that every property had a corresponding value
123  if (NOT N EQUAL 0)
124  message (FATAL_ERROR "No value given for target property ${ARGN}")
125  endif ()
126 endfunction ()

390 # ----------------------------------------------------------------------------
391 ## @brief Add link dependencies to build target.
392 #
393 # This function replaces CMake's
394 # <a href="http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:target_link_libraries">
395 # target_link_libraries()</a> command.
396 #
397 # The main reason for replacing this function is to treat libraries such as
398 # MEX-files which are supposed to be compiled into a MATLAB executable added
399 # by basis_add_executable() special. In this case, these libraries are added
400 # to the LINK_DEPENDS property of the given MATLAB Compiler target. Similarly,
401 # executable scripts and modules written in a scripting language may depend
402 # on other modules.
403 #
404 # Another reason is the mapping of build target names to fully-qualified
405 # build target names as used by BASIS (see basis_get_target_uid()).
406 #
407 # Example:
408 # @code
409 # basis_add_library (MyMEXFunc MEX myfunc.c)
410 # basis_add_executable (MyMATLABApp main.m)
411 # basis_target_link_libraries (MyMATLABApp MyMEXFunc OtherMEXFunc.mexa64)
412 # @endcode
413 #
414 # @param [in] TARGET_NAME Name of the target.
415 # @param [in] ARGN Link libraries.
416 #
417 # @returns Adds link dependencies to the specified build target.
418 # For custom targets, the given libraries are added to the
419 # @c LINK_DEPENDS property of these targets, in particular.
420 #
421 # @sa http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:target_link_libraries
422 #
423 # @ingroup CMakeAPI
424 function (basis_target_link_libraries TARGET_NAME)
425  basis_get_target_uid (TARGET_UID "${TARGET_NAME}")
426  if (NOT TARGET "${TARGET_UID}")
427  message (FATAL_ERROR "basis_target_link_libraries(): Unknown target ${TARGET_UID}.")
428  endif ()
429  # get type of named target
430  get_target_property (BASIS_TYPE ${TARGET_UID} BASIS_TYPE)
431  # substitute non-fully qualified target names
432  set (ARGS)
433  foreach (ARG ${ARGN})
434  basis_get_target_uid (UID "${ARG}")
435  if (TARGET "${UID}")
436  if (UID MATCHES "^${TARGET_UID}$")
437  message (FATAL_ERROR "Cannot add link library as dependency of itself!")
438  endif ()
439  list (APPEND ARGS "${UID}")
440  else ()
441  list (APPEND ARGS "${ARG}")
442  endif ()
443  endforeach ()
444  # get current link libraries
445  if (BASIS_TYPE MATCHES "^EXECUTABLE$|^(SHARED|STATIC|MODULE)_LIBRARY$")
446  get_target_property (DEPENDS ${TARGET_UID} BASIS_LINK_DEPENDS)
447  else ()
448  get_target_property (DEPENDS ${TARGET_UID} LINK_DEPENDS)
449  endif ()
450  if (NOT DEPENDS)
451  set (DEPENDS)
452  endif ()
453  # note that MCC does itself a dependency check and in case of scripts
454  # the basis_get_target_link_libraries() function is used
455  if (BASIS_TYPE MATCHES "MCC|SCRIPT")
456  list (APPEND DEPENDS ${ARGS})
457  # otherwise
458  else ()
459  list (APPEND DEPENDS ${ARGS})
460  # pull implicit dependencies (e.g., ITK uses this)
461  set (DEPENDENCY_ADDED 1)
462  while (DEPENDENCY_ADDED)
463  set (DEPENDENCY_ADDED 0)
464  foreach (LIB IN LISTS DEPENDS)
465  foreach (LIB_DEPEND IN LISTS ${LIB}_LIB_DEPENDS)
466  if (NOT LIB_DEPEND MATCHES "^$|^general$")
467  string (REGEX REPLACE "^-l" "" LIB_DEPEND "${LIB_DEPEND}")
468  list (FIND DEPENDS ${LIB_DEPEND} IDX)
469  if (IDX EQUAL -1)
470  list (APPEND DEPENDS ${LIB_DEPEND})
471  set (DEPENDENCY_ADDED 1)
472  endif ()
473  endif ()
474  endforeach ()
475  endforeach ()
476  endwhile ()
477  endif ()
478  # update LINK_DEPENDS
479  if (BASIS_TYPE MATCHES "^EXECUTABLE$|^(SHARED|STATIC|MODULE)_LIBRARY$")
480  _set_target_properties (${TARGET_UID} PROPERTIES BASIS_LINK_DEPENDS "${DEPENDS}")
481  target_link_libraries (${TARGET_UID} ${ARGS})
482  else ()
483  # FIXME cannot use LINK_DEPENDS for CMake targets as these depends are
484  # otherwise added as is to the Makefile. therefore, consider renaming
485  # LINK_DEPENDS in general to BASIS_LINK_DEPENDS.
486  _set_target_properties (${TARGET_UID} PROPERTIES LINK_DEPENDS "${DEPENDS}")
487  endif ()
488 endfunction ()

503 # ----------------------------------------------------------------------------
504 ## @brief Add executable target.
505 #
506 # This is the main function to add an executable target to the build system,
507 # where an executable can be a binary file or a script written in a scripting
508 # language. In general we refer to any output file which is part of the software
509 # (i.e., excluding configuration files) and which can be executed
510 # (e.g., a binary file in the ELF format) or interpreted (e.g., a Python script)
511 # directly, as executable file. Natively, CMake supports only executables built
512 # from C/C++ source code files. This function extends CMake's capabilities
513 # by adding custom build commands for non-natively supported programming
514 # languages and further standardizes the build of executable targets.
515 # For example, by default, it is not necessary to specify installation rules
516 # separately as these are added by this function already (see below).
517 #
518 # @par Programming languages
519 # Besides adding usual executable targets build by the set <tt>C/CXX</tt>
520 # language compiler, this function inspects the list of source files given and
521 # detects whether this list contains sources which need to be build using a
522 # different compiler. In particular, it supports the following languages:
523 # @n
524 # <table border="0">
525 # <tr>
526 # @tp @b CXX @endtp
527 # <td>The default behavior, adding an executable target build from C/C++
528 # source code. The target is added via CMake's add_executable() command.</td>
529 # </tr>
530 # <tr>
531 # @tp <b>PYTHON</b>|<b>JYTHON</b>|<b>PERL</b>|<b>BASH</b> @endtp
532 # <td>Executables written in one of the named scripting languages are built by
533 # configuring and/or copying the script files to the build tree and
534 # installation tree, respectively. During the build step, certain strings
535 # of the form \@VARIABLE\@ are substituted by the values set during the
536 # configure step. How these CMake variables are set is specified by a
537 # so-called script configuration, which itself is either a CMake script
538 # file or a string of CMake code set as value of the @c SCRIPT_DEFINITIONS
539 # property of the executable target.</td>
540 # </tr>
541 # <tr>
542 # @tp @b MATLAB @endtp
543 # <td>Standalone application built from MATLAB sources using the
544 # MATLAB Compiler (mcc). This language option is used when the list
545 # of source files contains one or more *.m files. A custom target is
546 # added which depends on custom command(s) that build the executable.</td>
547 # @n@n
548 # Attention: The *.m file with the entry point/main function of the
549 # executable has to be given before any other *.m file.
550 # </tr>
551 # </table>
552 #
553 # @par Helper functions
554 # If the programming language of the input source files is not specified
555 # explicitly by providing the @p LANGUAGE argument, the extensions of the
556 # source files and if necessary the first line of script files are inspected
557 # by the basis_get_source_language() function. Once the programming language is
558 # known, this function invokes the proper subcommand which adds the respective
559 # build target. In particular, it calls basis_add_executable_target() for C++
560 # sources (.cxx), basis_add_mcc_target() for MATLAB scripts (.m), and
561 # basis_add_script() for all other source files.
562 #
563 # @note DO NOT use the mentioned subcommands directly. Always use
564 # basis_add_executable() to add an executable target to your project.
565 # Only refer to the documentation of the subcommands to learn about the
566 # available options of the particular subcommand and considered target
567 # properties.
568 #
569 # @par Output directories
570 # The built executable file is output to the @c BINARY_RUNTIME_DIR or
571 # @c BINARY_LIBEXEC_DIR if the @p LIBEXEC option is given.
572 # If this function is used within the @c PROJECT_TESTING_DIR, however,

750 # written in a scripting language. In general we refer to any output file which
751 # is part of the software (i.e., excluding configuration files), but cannot be
752 # executed (e.g., a binary file in the ELF format) or interpreted

754 # libraries built from C/C++ source code files. This function extends CMake's
755 # capabilities by adding custom build commands for non-natively supported
756 # programming languages and further standardizes the build of library targets.
757 # For example, by default, it is not necessary to specify installation rules
758 # separately as these are added by this function already (see below).
759 #
760 # @par Programming languages
761 # Besides adding usual library targets built from C/C++ source code files,
762 # this function can also add custom build targets for libraries implemented
763 # in other programming languages. It therefore tries to detect the programming
764 # language of the given source code files and delegates the addition of the
765 # build target to the proper helper functions. It in particular supports the
766 # following languages:
767 # @n
768 # <table border="0">
769 # <tr>

771 # <td>Source files written in C/C++ are by default built into either

774 # the MEX script instead of using the default C++ compiler directly.</td>
775 # </tr>

777 # @tp <b>PYTHON</b>|<b>JYTHON</b>|<b>PERL</b>|<b>BASH</b> @endtp
778 # <td>Modules written in one of the named scripting languages are built similar
779 # to executable scripts except that the file name extension is preserved
780 # and no executable file permission is set on Unix. These modules are
781 # intended for import/inclusion in other modules or executables written
782 # in the particular scripting language only.</td>
783 # </tr>
784 # <tr>
785 # @tp @b MATLAB @endtp

812 # the <tt>BINARY_&lt;LANGUAGE&gt;_LIBRARY_DIR</tt> if defined. Otherwise,
813 # the built libraries are output to the @c BINARY_RUNTIME_DIR, @c BINARY_LIBRARY_DIR,

818 # @par Installation

826 # If this function is used within the @c PROJECT_TESTING_DIR, however, no

828 # as part of the build tree.

831 # @code

835 # basis_add_library (

1047 #
1048 # @note This function should not be used directly for executable scripts or
1049 # module libraries. Use basis_add_executable() or basis_add_library()
1050 # in such (most) cases instead.
1051 #
1052 # This function can be used to add a single arbitrary script file (i.e., any
1053 # text file which is input to a program), such as a CTest script for example,
1054 # to the build if neither basis_add_executable() nor basis_add_library() are
1055 # appropriate choices. In all other cases, either basis_add_executable() or
1056 # basis_add_library() should be used. Note that the script file is by default
1057 # not considered to be an executable. Instead it is assumed that the program
1058 # which interprets/processes the script must be executed explicitly with this
1059 # script as argument. Only scripts built with the @p EXECUTABLE or @p LIBEXEC
1060 # type option are treated as executable files, where in case of Unix a shebang
1061 # directive implicitly states the program used by the shell to interpret the
1062 # script and on Windows a Windows Command which imitates the behavior of Unix
1063 # shells is generated by BASIS. Do not use these type options, however, but
1064 # only use the default @p MODULE option. The basis_add_executable() function
1065 # should be used instead to add an executable script. The basis_add_script()
1066 # function shall only be used for none-executable arbitrary script files which
1067 # cannot be built by basis_add_executable() or basis_add_library().
1068 #
1069 # If the script name ends in <tt>.in</tt>, the <tt>.in</tt> suffix is removed
1070 # from the output name. Further, in case of executable scripts, the file name
1071 # extension is removed from the output file name. Instead, a shebang directive
1072 # is added on Unix to the built script. In order to enable the convenient
1073 # execution of Python and Perl scripts also on Windows without requiring the
1074 # user to setup a proper associate between the filename extension with the
1075 # corresponding interpreter executable, a few lines of Batch code are added at
1076 # the top and bottom of executable Python and Perl scripts. This Batch code
1077 # invokes the configured interpreter with the script file and the given script
1078 # arguments as command-line arguments. Note that both the original script source
1079 # code and the Batch code are stored within the single file. The file name
1080 # extension of such modified scripts is by default set to <tt>.cmd</tt>, the
1081 # common extension for Windows NT Command Scripts. Scripts in other languages
1082 # are not modified and the extension of the original scripts script file is
1083 # preserved on Windows in this case. In case of non-executable scripts, the
1084 # file name extension is kept in any case.
1085 #
1086 # Certain CMake variables within the source file are replaced during the
1087 # built of the script. See the
1088 # <a href="http://opensource.andreasschuh.com/cmake-basis/scripttargets/>
1089 # Build System Standard</a> for details.
1090 # Note, however, that source files are only configured if the file name
1091 # ends in the <tt>.in</tt> suffix.
1092 #
1093 # A custom CMake build target with the following properties is added by this
1094 # function to the build system. These properties are used by basis_build_script()
1095 # to generate a build script written in CMake code which is executed by a custom
1096 # CMake command. Before the invokation of basis_build_script(), the target
1097 # properties can be modified using basis_set_target_properties().
1098 #
1099 # @note Custom BASIS build targets are finalized by BASIS at the end of
1100 # basis_project_impl(), i.e., the end of the root CMake configuration file
1101 # of the (sub-)project.

1297  # parse arguments
1298  CMAKE_PARSE_ARGUMENTS (
1299  ARGN
1300  "MODULE;EXECUTABLE;LIBEXEC;NO_BASIS_UTILITIES;USE_BASIS_UTILITIES;EXPORT;NOEXPORT"
1301  "COMPONENT;DESTINATION;LANGUAGE"
1302  ""
1303  ${ARGN}
1304  )
1305  if (NOT ARGN_MODULE AND NOT ARGN_EXECUTABLE AND NOT ARGN_LIBEXEC)
1306  set (ARGN_MODULE TRUE)
1307  endif ()
1308  if (ARGN_MODULE)
1309  set (TYPE MODULE)
1310  else ()
1311  set (TYPE EXECUTABLE)
1312  endif ()
1313  string (TOLOWER "${TYPE}" type)
1314  # derive target name from file name if existing source file given as first argument
1315  get_filename_component (S "${TARGET_NAME}" ABSOLUTE)
1316  if (EXISTS "${S}" AND NOT IS_DIRECTORY "${S}" OR (NOT S MATCHES "\\.in$" AND EXISTS "${S}.in" AND NOT IS_DIRECTORY "${S}.in"))
1317  set (SOURCES "${S}")
1318  if (ARGN_MODULE)
1319  basis_get_source_target_name (TARGET_NAME "${TARGET_NAME}" NAME)
1320  else ()
1321  basis_get_source_target_name (TARGET_NAME "${TARGET_NAME}" NAME_WE)
1322  endif ()
1323  else ()
1324  set (SOURCES)
1325  set (OUTPUT_NAME "${TARGET_NAME}")
1326  endif ()
1327  # check target name
1328  basis_check_target_name ("${TARGET_NAME}")
1329  basis_make_target_uid (TARGET_UID "${TARGET_NAME}")
1330  message (STATUS "Adding ${type} script ${TARGET_UID}...")
1331  if (ARGN_MODULE AND TYPE MATCHES "EXECUTABLE")
1332  message (FATAL_ERROR "Target ${TARGET_UID}: MODULE and EXECUTABLE or LIBEXEC options are mutually exclusive!")
1333  endif ()
1334  # check/set parsed arguments
1335  basis_set_flag (ARGN EXPORT ${BASIS_EXPORT})
1336  if (ARGN_USE_BASIS_UTILITIES AND ARGN_NO_BASIS_UTILITIES)
1337  message (FATAL_ERROR "Options USE_BASIS_UTILITIES and NO_BASIS_UTILITIES are mutually exclusive!")
1338  endif ()
1339  list (LENGTH ARGN_UNPARSED_ARGUMENTS N)
1340  if (SOURCES)
1341  math (EXPR N "${N} + 1")
1342  endif ()
1343  if (N GREATER 1)
1344  if (NOT SOURCES)
1345  list (REMOVE_AT ARGN_UNPARSED_ARGUMENTS 0)
1346  endif ()
1347  message (FATAL_ERROR "Target ${TARGET_UID}: Too many or unrecognized arguments: ${ARGN_UNPARSED_ARGUMENTS}!\n"
1348  " Only one script can be built by each script target.")
1349  elseif (NOT SOURCES)
1350  set (SOURCES "${ARGN_UNPARSED_ARGUMENTS}")
1351  get_filename_component (SOURCES "${SOURCES}" ABSOLUTE)
1352  endif ()
1353  if (NOT EXISTS "${SOURCES}" AND NOT SOURCES MATCHES "\\.in$" AND EXISTS "${SOURCES}.in")
1354  set (SOURCES "${SOURCES}.in")
1355  endif ()
1356  if (NOT EXISTS "${SOURCES}")
1357  string (REGEX REPLACE "\\.in$" "" SOURCES "${SOURCES}")
1358  message (FATAL_ERROR "Target ${TARGET_UID}: Source file ${SOURCES}[.in] does not exist!")
1359  endif ()

1361  set (BUILD_DIR "${CMAKE_CURRENT_BINARY_DIR}/CMakeFiles/${TARGET_UID}.dir")
1362  basis_dump_variables ("${BUILD_DIR}/cache.cmake")
1363  # auto-detect programming language (may be as well UNKNOWN)
1364  if (ARGN_LANGUAGE)
1365  string (TOUPPER "${ARGN_LANGUAGE}" ARGN_LANGUAGE)
1366  else ()
1367  basis_get_source_language (ARGN_LANGUAGE ${SOURCES})
1368  endif ()
1369  # TEST flag
1370  basis_sanitize_for_regex (RE "${PROJECT_TESTING_DIR}")
1371  if (CMAKE_CURRENT_SOURCE_DIR MATCHES "^${RE}")
1372  set (TEST TRUE)
1373  else ()
1374  set (TEST FALSE)
1375  endif ()

1377  if (ARGN_MODULE)
1378  set (TYPE_INFIX "LIBRARY")
1379  elseif (ARGN_LIBEXEC)
1380  set (TYPE_INFIX "LIBEXEC")
1381  else ()
1382  set (TYPE_INFIX "RUNTIME")
1383  endif ()
1384  # output name
1385  string (REGEX REPLACE "\\.in$" "" SOURCE_NAME "${SOURCES}")
1386  if (NOT OUTPUT_NAME)
1387  get_filename_component (OUTPUT_NAME "${SOURCE_NAME}" NAME_WE)
1388  endif ()
1389  if (ARGN_MODULE)
1390  get_filename_component (SUFFIX "${SOURCE_NAME}" EXT)
1391  else ()
1392  if (WIN32)
1393  if (ARGN_LANGUAGE MATCHES "[JP]YTHON|PERL")
1394  set (SUFFIX ".cmd")
1395  else ()
1396  get_filename_component (SUFFIX "${SOURCE_NAME}" EXT)
1397  endif ()
1398  else ()
1399  set (SUFFIX)
1400  endif ()
1401  endif ()
1402  # output directory
1403  if (TEST)
1404  set (OUTPUT_DIRECTORY "${TESTING_${TYPE_INFIX}_DIR}")
1405  else ()
1406  set (OUTPUT_DIRECTORY "${BINARY_${TYPE_INFIX}_DIR}")
1407  endif ()
1408  # installation component
1409  if (NOT ARGN_COMPONENT)
1410  if (ARGN_MODULE)
1411  set (ARGN_COMPONENT "${BASIS_LIBRARY_COMPONENT}")
1412  else ()
1413  set (ARGN_COMPONENT "${BASIS_RUNTIME_COMPONENT}")
1414  endif ()
1415  endif ()
1416  if (NOT ARGN_COMPONENT)
1417  set (ARGN_COMPONENT "Unspecified")
1418  endif ()
1419  # installation directory
1420  if (ARGN_DESTINATION)
1421  if (ARGN_DESTINATION MATCHES "^[nN][oO][nN][eE]$")
1422  set (ARGN_DESTINATION)
1423  elseif (IS_ABSOLUTE "${ARGN_DESTINATION}")
1424  file (RELATIVE_PATH ARGN_DESTINATION "${CMAKE_INSTALL_PREFIX}" "${ARGN_DESTINATION}")
1425  endif ()
1426  elseif (TEST)
1427  set (ARGN_DESTINATION) # do not install
1428  else ()
1429  set (ARGN_DESTINATION "${INSTALL_${TYPE_INFIX}_DIR}")
1430  endif ()
1431  # script configuration ("compile definitions")
1432  if (EXISTS "${BINARY_CONFIG_DIR}/ScriptConfig.cmake")
1433  set (CONFIG_FILE "${BINARY_CONFIG_DIR}/ScriptConfig.cmake")
1434  else ()
1435  set (CONFIG_FILE)
1436  endif ()
1437  # auto-detect use of BASIS utilities
1438  set (LINK_DEPENDS)
1439  if (ARGN_LANGUAGE MATCHES "[JP]YTHON")
1440  set (UTILITIES_LANGUAGE "PYTHON")
1441  else ()
1442  set (UTILITIES_LANGUAGE "${ARGN_LANGUAGE}")
1443  endif ()
1444  if (ARGN_USE_BASIS_UTILITIES)
1445  if (NOT BASIS_UTILITIES_ENABLED MATCHES "${UTILITIES_LANGUAGE}")
1446  message (FATAL_ERROR "Target ${TARGET_UID} requires the BASIS utilities for ${UTILITIES_LANGUAGE}"
1447  " but BASIS was either build without the build of these utilities enabled"
1448  " or no utilities for this programming language are implemented. Remove the"
1449  " USE_BASIS_UTILITIES option if no BASIS utilities are used by the script"
1450  " ${SOURCES} or specify the correct programming language if it was not"
1451  " detected correctly.")
1452  endif ()
1453  set (USES_BASIS_UTILITIES TRUE)
1454  elseif (NOT ARGN_NO_BASIS_UTILITIES AND NOT UTILITIES_LANGUAGE MATCHES "UNKNOWN")
1455  basis_utilities_check (USES_BASIS_UTILITIES ${SOURCES} ${UTILITIES_LANGUAGE})
1456  else ()
1457  set (USES_BASIS_UTILITIES FALSE)
1458  endif ()
1459  if (USES_BASIS_UTILITIES)
1460  basis_set_project_property (PROPERTY PROJECT_USES_${UTILITIES_LANGUAGE}_UTILITIES TRUE)
1461  if (BASIS_DEBUG)
1462  message ("** Target ${TARGET_UID} uses the BASIS utilities for ${UTILITIES_LANGUAGE}.")
1463  endif ()
1464  endif ()
1465  # add custom target
1466  add_custom_target (${TARGET_UID} ALL SOURCES ${SOURCES})
1467  _set_target_properties (
1468  ${TARGET_UID}
1469  PROPERTIES
1470  LANGUAGE ${ARGN_LANGUAGE}
1471  BASIS_TYPE SCRIPT_${TYPE}
1472  BASIS_UTILITIES ${USES_BASIS_UTILITIES}
1473  SOURCE_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}"
1474  BINARY_DIRECTORY "${CMAKE_CURRENT_BINARY_DIR}"
1475  OUTPUT_DIRECTORY "${OUTPUT_DIRECTORY}"
1476  INSTALL_DIRECTORY "${ARGN_DESTINATION}"
1477  COMPONENT "${ARGN_COMPONENT}"
1478  OUTPUT_NAME "${OUTPUT_NAME}"
1479  PREFIX ""
1480  SUFFIX "${SUFFIX}"
1481  SCRIPT_DEFINITIONS ""
1482  SCRIPT_DEFINITIONS_FILE "${CONFIG_FILE}"
1483  LINK_DEPENDS "${LINK_DEPENDS}"
1484  EXPORT ${EXPORT}
1485  COMPILE ${BASIS_COMPILE_SCRIPTS}
1486  TEST ${TEST}
1487  LIBEXEC ${ARGN_LIBEXEC}
1488  )
1489  # add target to list of targets
1490  basis_set_project_property (APPEND PROPERTY TARGETS "${TARGET_UID}")
1491  message (STATUS "Adding ${type} script ${TARGET_UID}... - done")
1492 endfunction ()

1556 #
1557 # By default, the BASIS C++ utilities library is added as link dependency.
1558 # If none of the BASIS C++ utilities are used by this target, the option

1560 # variable @c BASIS_UTILITIES to @c FALSE, best in the <tt>Settings.cmake</tt>
1561 # file located in the @c PROJECT_CONFIG_DIR (add such file if missing).
1562 # If the use of the BASIS C++ utilities is disabled by default, the
1563 # @c USE_BASIS_UTILITIES option can be used to enable them for this target

1565 # would simply not include any of the BASIS utility functions in the final
1566 # binary file if not used. The only advantage of setting @c BASIS_UTILITIES to
1567 # @c FALSE or to always specify @c NO_BASIS_UTILITIES if no target uses the
1568 # utilities is that the BASIS utilities library will not be build in this case.
1569 #
1570 # @param [in] TARGET_NAME Name of build target.
1571 # @param [in] ARGN This argument list is parsed and the following
1572 # arguments are extracted, all other arguments are
1573 # considered to be source code files and simply passed
1574 # on to CMake's add_executable() command.
1575 # @par
1576 # <table border=0>
1577 # <tr>
1578 # @tp @b COMPONENT name @endtp
1579 # <td>Name of component as part of which this executable will be installed

1581 # (default: @c BASIS_RUNTIME_COMPONENT)</td>
1582 # </tr>

1590 # </tr>

1709  endif ()

1713  basis_target_link_libraries (${TARGET_UID} ${BASIS_CXX_UTILITIES_LIBRARY})
1714  # project-specific utilities build as part of this project
1715  basis_target_link_libraries (${TARGET_UID} ${BASIS_UTILITIES_TARGET})
1716  endif ()
1717  _set_target_properties (${TARGET_UID} PROPERTIES BASIS_UTILITIES ${USES_BASIS_UTILITIES})

1727  endif ()

1729  endif ()
1730  endif ()
1731  # installation
1732  if (ARGN_DESTINATION)
1733  if (TEST)
1734  # TODO install (selected?) tests
1735  else ()
1736  install (
1737  TARGETS ${TARGET_UID} ${EXPORT_OPT}
1738  DESTINATION "${ARGN_DESTINATION}"
1739  COMPONENT "${ARGN_COMPONENT}"
1740  )
1741  endif ()
1742  endif ()
1743  # done
1744  message (STATUS "Adding executable ${TARGET_UID}... - done")
