2 # @file shflags.sh
3 # @author Kate Ward <kate.ward at forestent.com>, Andreas Schuh
4 # @brief Advanced command-line flag library for Unix shell scripts.
5 #
6 # @sa http://code.google.com/p/shflags/
7 #
8 # @note The shFlags implementation by Kate Ward (revision 147) has been
9 # considerably modified by Andreas Schuh as part of the BASIS project
10 # to fit the needs of the SBIA Group at The University of Pennsylvania.
11 #
12 # This module implements something like the google-gflags library available
13 # from http://code.google.com/p/google-gflags/.
14 #
15 # FLAG TYPES: This is a list of the DEFINE_*'s that you can do. All flags take
16 # a name, default value, help-string, and optional 'short' name (one-letter
17 # name). Some flags have other arguments, which are described with the flag.
18 #
19 # - DEFINE_string: takes any input, and intreprets it as a string.
20 #
21 # - DEFINE_boolean: typically does not take any argument: say --myflag to set
22 # FLAGS_myflag to true, or --nomyflag to set FLAGS_myflag to false.
23 # Alternatively, you can say
24 # --myflag=true or --myflag=t or --myflag=0 or
25 # --myflag=false or --myflag=f or --myflag=1
26 # Passing an option has the same affect as passing the option once.
27 #
28 # - DEFINE_float: takes an input and intreprets it as a floating point number. As
29 # shell does not support floats per-se, the input is merely validated as
30 # being a valid floating point value.
31 #
32 # - DEFINE_integer: takes an input and intreprets it as an integer.
33 #
34 # - SPECIAL FLAGS: There are a few flags that have special meaning:
35 # --help (or -?) prints a list of all the flags in a human-readable fashion
36 # --flagfile=foo read flags from foo. (not implemented yet)
37 # -- as in getopt(), terminates flag-processing
38 #
39 # EXAMPLE USAGE:
40 #
41 # Example script hello.sh(.in):
42 # @code
43 # #! /bin/sh
44 # . ${BASIS_SOURCE} || exit 1
45 #
46 # DEFINE_string name 'world' "somebody's name" n
47 #
48 # FLAGS "$@" || exit $?
49 # eval set -- "${FLAGS_ARGV}"
50 #
51 # echo "Hello, ${FLAGS_name}."
52 # @endcode
53 #
54 # Usage of example script hello.sh:
55 # @code
56 # $ ./hello.sh -n Kate
57 # Hello, Kate.
58 # @endcode
59 #
60 # CUSTOMIZABLE BEHAVIOR:
61 #
62 # A script can override the default 'getopt' command by providing the path to
63 # an alternate implementation by defining the FLAGS_GETOPT_CMD variable.
64 #
65 # ATTRIBUTES:
66 #
67 # Shared attributes:
68 # flags_error: last error message
69 # flags_return: last return value
70 #
71 # __flags_longNames: list of long names for all flags
72 # __flags_shortNames: list of short names for all flags
73 # __flags_boolNames: list of boolean flag names
74 #
75 # __flags_opts: options parsed by getopt
76 #
77 # Per-flag attributes:
78 # FLAGS_<flag_name>: contains value of flag named 'flag_name'
79 # __flags_<flag_name>_default: the default flag value
80 # __flags_<flag_name>_help: the flag help string
81 # __flags_<flag_name>_short: the flag short name
82 # __flags_<flag_name>_type: the flag type
83 # __flags_<flag_name>_category: category of flag, use special category
84 # 'required' to denote flags that need to be
85 # given on the command line
86 #
87 # NOTES:
88 #
89 # - Not all systems include a getopt version that supports long flags. On these
90 # systems, only short flags are recognized.
91 #
92 # - Lists of strings are space separated, and a null value is the '~' char.

234 # Define a flag.
235 #
236 # Calling this function will define the following info variables for the
237 # specified flag:
238 # FLAGS_flagname - the name for this flag (based upon the long flag name)
239 # __flags_<flag_name>_default - the default value
240 # __flags_<flag_name>_help - the help string
241 # __flags_<flag_name>_short - the single letter alias
242 # __flags_<flag_name>_type - the type of flag (one of __FLAGS_TYPE_*)
243 # __flags_<flag_name>_category - the category of the flag
244 #
245 # Args:

247 # _flags_name: string: long flag name
248 # _flags_default: default flag value
249 # _flags_help: string: help string

251 # _flags_category: string: (optional) category name this flags belongs to
252 # Returns:
253 # integer: success of operation, or error

255 {
256  if [ $# -lt 4 ]; then

258  flags_return=${FLAGS_ERROR}
259  _flags_error "${flags_error}"
260  return ${flags_return}

264  _flags_name_="$2"
265  _flags_default_="$3"
266  _flags_help_="$4"
267  _flags_short_="${5:-${__FLAGS_NULL}}"
268  _flags_category_="${6:-${__FLAGS_NULL}}"
269 
270  _flags_return_=${FLAGS_TRUE}
271  _flags_usName_=`_flags_underscoreName ${_flags_name_}`
272 
273  # check whether the flag name is reserved
274  _flags_itemInList ${_flags_usName_} "${__FLAGS_RESERVED_LIST}"
275  if [ $? -eq ${FLAGS_TRUE} ]; then
276  flags_error="flag name (${_flags_name_}) is reserved"
277  _flags_return_=${FLAGS_ERROR}
278  fi
279 
280  # require short option for getopt that don't support long options
281  if [ ${_flags_return_} -eq ${FLAGS_TRUE} \
282  -a ${__FLAGS_GETOPT_VERS} -ne ${__FLAGS_GETOPT_VERS_ENH} \
283  -a "${_flags_short_}" = "${__FLAGS_NULL}" ]
284  then
285  flags_error="short flag required for (${_flags_name_}) on this platform"
286  _flags_return_=${FLAGS_ERROR}
287  fi
288 
289  # check for existing long name definition
290  if [ ${_flags_return_} -eq ${FLAGS_TRUE} ]; then
291  if _flags_itemInList ${_flags_usName_} ${__flags_definedNames}; then
292  flags_error="definition for ([no]${_flags_name_}) already exists"
293  _flags_warn "${flags_error}"
294  _flags_return_=${FLAGS_FALSE}
295  fi
296  fi
297 
298  # check for existing short name definition
299  if [ ${_flags_return_} -eq ${FLAGS_TRUE} \
300  -a "${_flags_short_}" != "${__FLAGS_NULL}" ]
301  then
302  if _flags_itemInList "${_flags_short_}" ${__flags_shortNames}; then
303  flags_error="flag short name (${_flags_short_}) already defined"
304  _flags_warn "${flags_error}"
305  _flags_return_=${FLAGS_FALSE}
306  fi
307  fi
308 
309  # convert array given as string such as "'a 1' 'b 2' 'c' 'd 4'" to array
310  # ('a 1' 'b 2' 'c' 'd 4') and get type of values for multi-flags
311  if [ ${_flags_type_} -gt 128 -a ${_flags_type_} -ne ${__FLAGS_TYPE_MULTI_BOOLEAN} ]; then
312  _flags_valueType_=`expr "${_flags_type_}" - 128`
313  eval "_flags_default_=(${_flags_default_})"
314  else
315  if [ ${#_flags_default_[@]} -ne 1 ]; then
316  flags_error="${_flags_name_}: invalid default flag value '${_flags_default_}'"
317  _flags_return_=${FLAGS_ERROR}
318  fi
319  _flags_valueType_=${_flags_type_}
320  fi
321 
322  # handle default value. note, on several occasions the 'if' portion of an
323  # if/then/else contains just a ':' which does nothing. a binary reversal via
324  # '!' is not done because it does not work on all shells.
325  if [ ${_flags_return_} -eq ${FLAGS_TRUE} ]; then
326  case ${_flags_type_} in
327  ${__FLAGS_TYPE_BOOLEAN})
328  if _flags_validateBoolean "${_flags_default_}"; then
329  case ${_flags_default_} in
330  true|t|0) _flags_default_=${FLAGS_TRUE} ;;
331  false|f|1) _flags_default_=${FLAGS_FALSE} ;;
332  esac
333  else
334  flags_error="${_flags_name_}: invalid default flag value '${_flags_default_}'"
335  _flags_return_=${FLAGS_ERROR}
336  fi
337  ;;
338 
339  ${__FLAGS_TYPE_MULTI_BOOLEAN})
340  if _flags_validateInteger "${_flags_default_}"; then
341  :
342  else
343  flags_error="${_flags_name_}: invalid default flag value '${_flags_default_}'"
344  _flags_return_=${FLAGS_ERROR}
345  fi
346  ;;
347 
348  *)
349  for _flags_defaultValue_ in "${_flags_default_[@]}"; do
350  case ${_flags_valueType_} in
351  ${__FLAGS_TYPE_FLOAT})
352  if _flags_validateFloat "${_flags_defaultValue_}"; then
353  :
354  else
355  flags_error="${_flags_name_}: invalid default flag value '${_flags_defaultValue_}'"
356  _flags_return_=${FLAGS_ERROR}
357  fi
358  ;;
359 
360  ${__FLAGS_TYPE_INTEGER})
361  if _flags_validateInteger "${_flags_defaultValue_}"; then
362  :
363  else
364  flags_error="${_flags_name_}: invalid default flag value '${_flags_defaultValue_}'"
365  _flags_return_=${FLAGS_ERROR}
366  fi
367  ;;
368 
369  ${__FLAGS_TYPE_STRING}) ;; # everything in shell is a valid string
370 
371  *)
372  flags_error="${_flags_name_}: unrecognized flag type '${_flags_type_}'"
373  _flags_return_=${FLAGS_ERROR}
374  ;;
375  esac
376  done
377  ;;
378  esac
379  fi
380 
381  if [ ${_flags_return_} -eq ${FLAGS_TRUE} ]; then
382  # store flag information
383  if [ ${_flags_type_} -gt 128 ]; then
384  eval "FLAGS_${_flags_usName_}=(\"\${_flags_default_[@]}\")"
385  eval "__flags_${_flags_usName_}_${__FLAGS_INFO_DEFAULT}=(\"\${_flags_default_[@]}\")"
386  else
387  eval "FLAGS_${_flags_usName_}='${_flags_default_}'"
388  eval "__flags_${_flags_usName_}_${__FLAGS_INFO_DEFAULT}='${_flags_default_}'"
389  fi
390  eval "__flags_${_flags_usName_}_${__FLAGS_INFO_TYPE}=${_flags_type_}"
391  eval "__flags_${_flags_usName_}_${__FLAGS_INFO_HELP}=\"${_flags_help_}\""
392  eval "__flags_${_flags_usName_}_${__FLAGS_INFO_SHORT}='${_flags_short_}'"
393  eval "__flags_${_flags_usName_}_${__FLAGS_INFO_CATEGORY}='${_flags_category_}'"
394 
395  # append flag names to name lists
396  __flags_shortNames="${__flags_shortNames}${_flags_short_} "
397  __flags_longNames="${__flags_longNames}${_flags_name_} "
398  [ ${_flags_type_} -eq ${__FLAGS_TYPE_BOOLEAN} -o \
399  ${_flags_type_} -eq ${__FLAGS_TYPE_MULTI_BOOLEAN} ] && \
400  __flags_boolNames="${__flags_boolNames}no${_flags_name_} "
401 
402  # append flag names to defined names for later validation checks
403  __flags_definedNames="${__flags_definedNames}${_flags_usName_} "
404  [ ${_flags_type_} -eq ${__FLAGS_TYPE_BOOLEAN} ] && \
405  __flags_definedNames="${__flags_definedNames}no${_flags_usName_} "
406 
407  # append category name to category names list
408  if [ "${_flags_category_}" != "${__FLAGS_NULL}" ]; then
409  _flags_found_=${FLAGS_FALSE}
410  for _flags_categoryName_ in "${__flags_categoryNames[@]}"; do
411  if [ "${_flags_categoryName_}" = "${_flags_category_}" ]; then
412  _flags_found_=${FLAGS_TRUE}
413  break
414  fi
415  done
416  if [ ${_flags_found_} -eq ${FLAGS_FALSE} ]; then
417  __flags_categoryNames[${#__flags_categoryNames[@]}]="${_flags_category_}"
418  fi
419  fi
420  fi
421 
422  flags_return=${_flags_return_}
423  unset _flags_default_ _flags_help_ _flags_name_ _flags_return_ \
424  _flags_short_ _flags_type_ _flags_usName_ \
425  _flags_category_ _flags_found_
426  [ ${flags_return} -eq ${FLAGS_ERROR} ] && _flags_error "${flags_error}"
427  return ${flags_return}

482 # Returns flag details based on a flag name and flag info.
483 #
484 # Args:
485 # string: underscored flag name
486 # string: flag info (see the _flags_define function for valid info types)
487 # Output:
488 # string: value of dereferenced flag variable
489 # Returns:
490 # integer: one of FLAGS_{TRUE|FALSE|ERROR}
491 _flags_getFlagInfo()
492 {
493  # note: adding gFI to variable names to prevent naming conflicts with calling
494  # functions
495  _flags_gFI_usName_=$1
496  _flags_gFI_info_=$2
497 
498  _flags_infoVar_="__flags_${_flags_gFI_usName_}_${_flags_gFI_info_}"
499  _flags_strToEval_="_flags_infoValue_=\"\${${_flags_infoVar_}:-}\""
500  eval "${_flags_strToEval_}"
501  if [ -n "${_flags_infoValue_}" ]; then
502  flags_return=${FLAGS_TRUE}
503  else
504  # see if the _flags_gFI_usName_ variable is a string as strings can be
505  # empty...
506  # note: the DRY principle would say to have this function call itself for
507  # the next three lines, but doing so results in an infinite loop as an
508  # invalid _flags_name_ will also not have the associated _type variable.
509  # Because it doesn't (it will evaluate to an empty string) the logic will
510  # try to find the _type variable of the _type variable, and so on. Not so
511  # good ;-)
512  _flags_typeVar_="__flags_${_flags_gFI_usName_}_${__FLAGS_INFO_TYPE}"
513  _flags_strToEval_="_flags_typeValue_=\"\${${_flags_typeVar_}:-}\""
514  eval "${_flags_strToEval_}"
515  if [ "${_flags_typeValue_}" = "${__FLAGS_TYPE_STRING}" -o "${_flags_typeValue_}" -gt 128 ]; then
516  flags_return=${FLAGS_TRUE}
517  else
518  flags_return=${FLAGS_ERROR}
519  flags_error="missing flag info variable (${_flags_infoVar_})"
520  fi
521  fi
522 
523  echo "${_flags_infoValue_}"
524  unset _flags_gFI_usName_ _flags_gfI_info_ _flags_infoValue_ _flags_infoVar_ \
525  _flags_strToEval_ _flags_typeValue_ _flags_typeVar_
526  [ ${flags_return} -eq ${FLAGS_ERROR} ] && _flags_error "${flags_error}"
527  return ${flags_return}
528 }

842 ##############################################################################
843 # Dynamically parse a getopt result and set appropriate variables.
844 #
845 # This function does the actual conversion of getopt output and runs it through
846 # the standard case structure for parsing. The case structure is actually quite
847 # dynamic to support any number of flags.
848 #
849 # Args:
850 # argc: int: original command-line argument count
851 # @: varies: output from getopt parsing
852 # Returns:
853 # integer: a FLAGS success condition
854 _flags_parseGetopt()
855 {
856  _flags_argc_=$1
857  shift
858 
859  flags_return=${FLAGS_TRUE}
860 
861  if [ ${__FLAGS_GETOPT_VERS} -ne ${__FLAGS_GETOPT_VERS_ENH} ]; then
862  set -- $@
863  else
864  # note the quotes around the `$@' -- they are essential!
865  eval set -- "$@"
866  fi
867 
868  # provide user with number of arguments to shift by later
869  # NOTE: the FLAGS_ARGC variable is obsolete as of 1.0.3 because it does not
870  # properly give user access to non-flag arguments mixed in between flag
871  # arguments. Its usage was replaced by FLAGS_ARGV, and it is being kept only
872  # for backwards compatibility reasons.
873  FLAGS_ARGC=`expr $# - 1 - ${_flags_argc_}`
874 
875  # handle options. note options with values must do an additional shift
876  while true; do
877  _flags_opt_=$1
878  _flags_arg_=${2:-}
879  _flags_name_=''
880 
881  # determine long flag name
882  case "${_flags_opt_}" in
883  --) shift; break ;; # discontinue option parsing
884 
885  --*) # long option
886  _flags_opt_=`expr -- "${_flags_opt_}" : '--\(.*\)'`
887  _flags_len_=${__FLAGS_LEN_LONG}
888  if _flags_itemInList "${_flags_opt_}" ${__flags_longNames}; then
889  _flags_name_=${_flags_opt_}
890  else
891  # check for negated long boolean version
892  if _flags_itemInList "${_flags_opt_}" ${__flags_boolNames}; then
893  _flags_name_=`expr -- "${_flags_opt_}" : 'no\(.*\)'`
894  _flags_arg_=${__FLAGS_NULL}
895  fi
896  fi
897  ;;
898 
899  -*) # short option
900  _flags_opt_=`expr -- "${_flags_opt_}" : '-\(.*\)'`
901  _flags_len_=${__FLAGS_LEN_SHORT}
902  if _flags_itemInList "${_flags_opt_}" ${__flags_shortNames}; then
903  # yes. match short name to long name. note purposeful off-by-one
904  # (too high) with awk calculations.
905  _flags_pos_=`echo "${__flags_shortNames}" \
906  |awk 'BEGIN{RS=" ";rn=0}$0==e{rn=NR}END{print rn}' \
907  e=${_flags_opt_}`
908  _flags_name_=`echo "${__flags_longNames}" \
909  |awk 'BEGIN{RS=" "}rn==NR{print $0}' rn="${_flags_pos_}"`
910  fi
911  ;;
912  esac
913 
914  # die if the flag was unrecognized
915  if [ -z "${_flags_name_}" ]; then
916  flags_error="unrecognized option (${_flags_opt_})"
917  flags_return=${FLAGS_ERROR}
918  break
919  fi
920 
921  # set new flag value
922  _flags_usName_=`_flags_underscoreName ${_flags_name_}`
923  _flags_type_=`_flags_getFlagInfo "${_flags_usName_}" ${__FLAGS_INFO_TYPE}`
924 
925  case ${_flags_type_} in
926  ${__FLAGS_TYPE_MULTI_BOOLEAN})
927  eval "_flags_val_=\${FLAGS_${_flags_usName_}}"
928  if [ ${_flags_len_} -eq ${__FLAGS_LEN_SHORT} -o \
929  "${_flags_arg_}" != "${__FLAGS_NULL}" ]; then
930  _flags_val_=`expr "${_flags_val_}" + 1`
931  else
932  _flags_val_=`expr "${_flags_val_}" - 1`
933  fi
934  eval "FLAGS_${_flags_usName_}=${_flags_val_}"
935  ;;
936 
937  *)
938  if [ ${_flags_type_} -ge 128 ]; then
939  eval "_flags_idx_=\${#FLAGS_${_flags_usName_}[@]}"
940  _flags_type_=`expr "${_flags_type_}" - 128`
941  else
942  _flags_idx_=0
943  fi
944  case ${_flags_type_} in
945  ${__FLAGS_TYPE_BOOLEAN})
946  if [ ${_flags_len_} -eq ${__FLAGS_LEN_LONG} ]; then
947  if [ "${_flags_arg_}" != "${__FLAGS_NULL}" ]; then
948  eval "FLAGS_${_flags_usName_}[${_flags_idx_}]=${FLAGS_TRUE}"
949  else
950  eval "FLAGS_${_flags_usName_}[${_flags_idx_}]=${FLAGS_FALSE}"
951  fi
952  else
953  _flags_strToEval_="_flags_val_=\${__flags_${_flags_usName_}_${__FLAGS_INFO_DEFAULT}}"
954  eval "${_flags_strToEval_}"
955  if [ ${_flags_val_} -eq ${FLAGS_FALSE} ]; then
956  eval "FLAGS_${_flags_usName_}[${_flags_idx_}]=${FLAGS_TRUE}"
957  else
958  eval "FLAGS_${_flags_usName_}[${_flags_idx_}]=${FLAGS_FALSE}"
959  fi
960  fi
961  ;;
962 
963  ${__FLAGS_TYPE_FLOAT})
964  if _flags_validateFloat "${_flags_arg_}"; then
965  eval "FLAGS_${_flags_usName_}[${_flags_idx_}]='${_flags_arg_}'"
966  else
967  flags_error="${_flags_name_}: invalid float value (${_flags_arg_})"
968  flags_return=${FLAGS_ERROR}
969  break
970  fi
971  ;;
972 
973  ${__FLAGS_TYPE_INTEGER})
974  if _flags_validateInteger "${_flags_arg_}"; then
975  eval "FLAGS_${_flags_usName_}[${_flags_idx_}]='${_flags_arg_}'"
976  else
977  flags_error="${_flags_name_}: invalid integer value (${_flags_arg_})"
978  flags_return=${FLAGS_ERROR}
979  break
980  fi
981  ;;
982 
983  ${__FLAGS_TYPE_STRING})
984  eval "FLAGS_${_flags_usName_}[${_flags_idx_}]='${_flags_arg_}'"
985  ;;
986  esac
987  ;;
988  esac
989 
990  # handle special case help flag
991  if [ "${_flags_usName_}" = 'help' ]; then
992  if [ ${FLAGS_help} -eq ${FLAGS_TRUE} ]; then
993  flags_help
994  flags_error='help requested'
995  flags_return=${FLAGS_TRUE}
996  break
997  fi
998  fi
999 
1000  # handle special case helpman flag
1001  if [ "${_flags_usName_}" = 'helpman' ]; then
1002  if [ ${FLAGS_helpman} -eq ${FLAGS_TRUE} ]; then
1003  flags_error='help requested'
1004  flags_return=${FLAGS_TRUE}
1005  # if man should not be executed dirtly,
1006  # print generated man page to STDOUT instead
1007  if [ -n "${FLAGS_execman}" ] && [ ${FLAGS_execman} -eq ${FLAGS_FALSE} ]; then
1008  flags_helpman
1009  break
1010  fi
1011  # save generated man page to temporary file
1012  flags_manFile_="`mktemp -t \"${0##*/}.1\"`"
1013  if [ $? -ne 0 ]; then
1014  flags_error='failed to create temporary man page file name'
1015  flags_return=${FLAGS_ERROR}
1016  break
1017  fi
1018  flags_man_="`flags_helpman`"
1019  if [ $? -ne ${FLAGS_TRUE} ]; then
1020  flags_error='failed to generate temporary man page file'
1021  flags_return=${FLAGS_ERROR}
1022  break
1023  fi
1024  echo "${flags_man_}" > "${flags_manFile_}"
1025  if [ $? -ne ${FLAGS_TRUE} ]; then
1026  flags_error='failed to write temporary man page file'
1027  flags_return=${FLAGS_ERROR}
1028  break
1029  fi
1030  # execute man to view the man page
1031  man "${flags_manFile_}"
1032  if [ $? -ne ${FLAGS_TRUE} ]; then
1033  flags_error='failed to execute man to view generated man page'
1034  flags_return=${FLAGS_ERROR}
1035  fi
1036  # remove temporary man page file
1037  rm -f "${flags_manFile_}"
1038  if [ ${flags_return} -ne ${FLAGS_ERROR} -a $? -ne ${FLAGS_TRUE} ]; then
1039  flags_error='failed to execute man to view generated man page'
1040  flags_return=${FLAGS_ERROR}
1041  fi
1042  break
1043  fi
1044  fi
1045 
1046  # handle special case helpxml flag
1047  if [ "${_flags_usName_}" = 'helpxml' ]; then
1048  if [ ${FLAGS_helpxml} -eq ${FLAGS_TRUE} ]; then
1049  flags_helpxml
1050  flags_error='help requested'
1051  flags_return=${FLAGS_TRUE}
1052  break
1053  fi
1054  fi
1055 
1056  # handle special case usage flag
1057  if [ "${_flags_usName_}" = 'helpshort' ]; then
1058  if [ ${FLAGS_helpshort} -eq ${FLAGS_TRUE} ]; then
1059  flags_helpshort
1060  flags_error='help requested'
1061  flags_return=${FLAGS_TRUE}
1062  break
1063  fi
1064  fi
1065 
1066  # handle special case version flag
1067  if [ "${_flags_usName_}" = 'version' ]; then
1068  if [ ${FLAGS_version} -eq ${FLAGS_TRUE} ]; then
1069  flags_version
1070  flags_error='version requested'
1071  flags_return=${FLAGS_TRUE}
1072  break
1073  fi
1074  fi
1075 
1076  # shift the option and non-boolean arguments out.
1077  shift
1078  [ ${_flags_type_} != ${__FLAGS_TYPE_BOOLEAN} -a \
1079  ${_flags_type_} != ${__FLAGS_TYPE_MULTI_BOOLEAN} ] && shift
1080  done

1088

1092 }

1349 # A basic boolean flag. Boolean flags do not take any arguments, and their
1350 # value is either 1 (false) or 0 (true). For long flags, the false value is
1351 # specified on the command line by prepending the word 'no'. With short flags,
1352 # the presense of the flag toggles the current value between true and false.
1353 # Specifying a short boolean flag twice on the command results in returning the
1354 # value back to the default value.
1355 #
1356 # A default value is required for boolean flags.
1357 #
1358 # For example, lets say a Boolean flag was created whose long name was 'update'
1359 # and whose short name was 'x', and the default value was 'false'. This flag
1360 # could be explicitly set to 'true' with '--update' or by '-x', and it could be
1361 # explicitly set to 'false' with '--noupdate'.
1362 DEFINE_boolean() { _flags_define ${__FLAGS_TYPE_BOOLEAN} "$@"; }
1363 DEFINE_bool() { _flags_define ${__FLAGS_TYPE_BOOLEAN} "$@"; }
1364

1369 DEFINE_string() { _flags_define ${__FLAGS_TYPE_STRING} "$@"; }

1372 # representing a list of boolean values, this flag's value is a counter which

1374 # otherwise. Note that the value of the counter may hence also be negative.

1376 # An example use case of this flag is the --verbose flag. The more often this
1377 # flag was given, the more verbose the output messages should be.
