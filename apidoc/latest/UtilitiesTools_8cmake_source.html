22 ## @brief Add build target for BASIS C++ utilities library.
23 #
24 # This function is called by the top-level project in order to add the "basis"
25 # build target for the static project-specific BASIS utilities library for C++.
26 # It is called by basis_project_impl() in the root CMakeLists.txt file of the
27 # top-level project.
28 #
29 # The CMake function add_library() checks if the specified source code files
30 # exist. If a source file is not found, an error is raised by CMake. The BASIS
31 # utilities can, however, only be configured at the end of the configuration
32 # step. Therefore, this function simply writes dummy C++ source files in order
33 # to pass the existence check. The actual source files are configured by the
34 # function basis_configure_utilities().
35 #
36 # After writing these dummy source files, a library build target for the
37 # project-specific BASIS C++ utilities is added. This build target is not
38 # being build as part of the ALL target in case it is never used by any of
39 # the build targets of the project. Only if build target links to this
40 # library, it will be build and installed.
41 #
42 # @param [out] UID UID of added build target.
43 function (basis_add_utilities_library UID)
44  # target UID of "basis" library target
45  basis_make_target_uid (TARGET_UID basis)
46  if (NOT TARGET ${TARGET_UID})
47  if (PROJECT_IS_SUBPROJECT)
48  # a subproject has it's own version of the project-specific BASIS utilities
49  # as the targets and functions live in a separate namespace
50  set (CODE_DIR "${BINARY_CODE_DIR}")
51  set (INCLUDE_DIR "${BINARY_INCLUDE_DIR}")
52  set (OUTPUT_DIR "${BINARY_ARCHIVE_DIR}")
53  set (INSTALL_DIR "${INSTALL_ARCHIVE_DIR}")
54  else ()
55  # modules, on the other side, share the library with the top-level project
56  # the addition of the utilities target is in this case only required because
57  # of the install(TARGETS) and install(EXPORT) commands.
58  set (CODE_DIR "${TOPLEVEL_BINARY_CODE_DIR}")
59  set (INCLUDE_DIR "${TOPLEVEL_BINARY_INCLUDE_DIR}")
60  set (OUTPUT_DIR "${TOPLEVEL_BINARY_ARCHIVE_DIR}")
61  set (INSTALL_DIR "${BASIS_INSTALL_ARCHIVE_DIR}")
62  endif ()
63  # write dummy source files
64  basis_library_prefix (PREFIX CXX)
65  foreach (S IN ITEMS basis.h basis.cxx)
66  if (S MATCHES "\\.h$")
67  set (S "${INCLUDE_DIR}/${PREFIX}${S}")
68  else ()
69  set (S "${CODE_DIR}/${S}")
70  endif ()
71  if (NOT EXISTS "${S}")
72  file (WRITE "${S}"
73  "#error This dummy source file should have been replaced by the"
74  " BASIS CMake function basis_configure_utilities()"
75  )
76  endif ()
77  endforeach ()
78  # add library target if not present yet - only build if required
79  add_library (${TARGET_UID} STATIC "${CODE_DIR}/basis.cxx")
80  # define dependency on non-project specific utilities as the order in
81  # which static libraries are listed on the command-line for the linker
82  # matters; this will help CMake to get the order right
83  target_link_libraries (${TARGET_UID} ${BASIS_CXX_UTILITIES_LIBRARY})
84  # set target properties
85  _set_target_properties (
86  ${TARGET_UID}
87  PROPERTIES
88  BASIS_TYPE STATIC_LIBRARY
89  OUTPUT_NAME basis
90  ARCHIVE_OUTPUT_DIRECTORY "${OUTPUT_DIR}"
91  ARCHIVE_INSTALL_DIRECTORY "${INSTALL_DIR}"
92  )
93  # add installation rule
94  install (
95  TARGETS ${TARGET_UID}
96  EXPORT ${PROJECT_NAME}
97  ARCHIVE
98  DESTINATION "${INSTALL_DIR}"
99  COMPONENT "${BASIS_LIBRARY_COMPONENT}"
100  )
101  basis_set_project_property (APPEND PROPERTY EXPORT_TARGETS ${TARGET_UID})
102  basis_set_project_property (APPEND PROPERTY INSTALL_EXPORT_TARGETS ${TARGET_UID})
103  # debug message
104  if (BASIS_DEBUG)
105  message ("** Added BASIS utilities library ${TARGET_UID}")
106  endif ()
107  endif ()
108  # done
109  basis_set_project_property (PROPERTY PROJECT_USES_CXX_UTILITIES TRUE)
110  set (${UID} "${TARGET_UID}" PARENT_SCOPE)
111 endfunction ()

147 # ----------------------------------------------------------------------------
148 ## @brief Check whether the BASIS utilities are used within a given source file.
149 #
150 # This function matches the source code against specific import patterns which
151 # are all valid imports of the BASIS utilities for the respective programming
152 # language of the specified file. If the BASIS utilities are used within the
153 # specified source file, the variable named @p VAR is set to @c TRUE.
154 # Otherwise, it is set to @c FALSE.
155 #
156 # @param [out] VAR Whether the BASIS utilites are used.
157 # @param [in] SOURCE_FILE Path of source file to check.
158 # @param [in] ARGN Source code language. If not specified, the
159 # programming language is determined automatically.
160 function (basis_utilities_check VAR SOURCE_FILE)
161  set (UTILITIES_USED FALSE)
162  if (ARGC EQUAL 2)
163  basis_get_source_language (LANGUAGE "${SOURCE_FILE}")
164  elseif (ARGC EQUAL 3)
165  set (LANGUAGE "${ARGN}")
166  else ()
167  message (FATAL_ERROR "Too many arguments given!")
168  endif ()
169  # --------------------------------------------------------------------------
170  # make file path absolute and append .in suffix if necessary
171  get_filename_component (SOURCE_FILE "${SOURCE_FILE}" ABSOLUTE)
172  if (NOT EXISTS "${SOURCE_FILE}" AND NOT SOURCE_FILE MATCHES "\\.in$" AND EXISTS "${SOURCE_FILE}.in")
173  set (SOURCE_FILE "${SOURCE_FILE}.in")
174  endif ()
175  # --------------------------------------------------------------------------
176  # C++
177  if (LANGUAGE MATCHES "CXX")
178  # read script file
179  file (READ "${SOURCE_FILE}" SOURCE)
180  # match use/require statements
181  basis_library_prefix (PREFIX ${LANGUAGE})
182  set (RE "[ \\t]*#[ \\t]*include[ \\t]+[<\"](${PREFIX})?basis.h[\">]") # e.g., #include "basis.h", #include <pkg/basis.h>
183  if (SCRIPT MATCHES "(^|\n)[ \t]*${RE}([ \t]*//.*|[ \t]*)(\n|$)")
184  set (UTILITIES_USED TRUE)
185  break ()
186  endif ()
187  # --------------------------------------------------------------------------
188  # Python/Jython
189  elseif (LANGUAGE MATCHES "[JP]YTHON")
190  # read script file
191  file (READ "${SOURCE_FILE}" SCRIPT)
192  # deprecated BASIS_PYTHON_UTILITIES macro
193  if (SCRIPT MATCHES "(^|\n|;)[ \t]*\@BASIS_PYTHON_UTILITIES\@")
194  message (FATAL_ERROR "Script ${SOURCE_FILE} uses the deprecated BASIS macro \@BASIS_PYTHON_UTILITIES\@!")
195  endif ()
196  basis_sanitize_for_regex (PYTHON_PACKAGE "${PROJECT_NAMESPACE_PYTHON}")
197  # match use of package-specific utilities
198  if (SCRIPT MATCHES "[^a-zA-Z._]${PYTHON_PACKAGE}.basis([.; \t\n]|$)") # e.g., basis = <package>.basis, <package>.basis.exedir()
199  set (UTILITIES_USED TRUE)
200  else ()
201  # match import statements
202  foreach (RE IN ITEMS
203  "import[ \\t]+basis" # e.g., import basis
204  "import[ \\t]+${PYTHON_PACKAGE}\\.basis" # e.g., import <package>.basis
205  "import[ \\t]+\\.\\.?basis" # e.g., import .basis, import ..basis
206  "from[ \\t]+${PYTHON_PACKAGE}[ \\t]+import[ \\t]+basis" # e.g., from <package> import basis
207  "from[ \\t]+${PYTHON_PACKAGE}.basis[ \\t]+import[ \\t].*" # e.g., from <package>.basis import which
208  "from[ \\t]+\\.\\.?[ \\t]+import[ \\t]+basis" # e.g., from . import basis", "from .. import basis
209  "from[ \\t]+\\.\\.?basis[ \\t]+import[ \\t].*" # e.g., from .basis import which, WhichError, from ..basis import which
210  ) # foreach RE
211  if (SCRIPT MATCHES "(^|\n|;)[ \t]*${RE}([ \t]*as[ \t]+.*)?([ \t]*#.*|[ \t]*)(;|\n|$)")
212  set (UTILITIES_USED TRUE)
213  break ()
214  endif ()
215  endforeach ()
216  endif ()
217  # --------------------------------------------------------------------------
218  # Perl
219  elseif (LANGUAGE MATCHES "PERL")
220  # read script file
221  file (READ "${SOURCE_FILE}" SCRIPT)
222  # deprecated BASIS_PERL_UTILITIES macro
223  if (SCRIPT MATCHES "(^|\n|;)[ \t]*\@BASIS_PERL_UTILITIES\@")
224  message (FATAL_ERROR "Script ${SOURCE_FILE} uses the deprecated BASIS macro \@BASIS_PERL_UTILITIES\@!")
225  endif ()
226  # match use/require statements
227  basis_sanitize_for_regex (PERL_PACKAGE "${PROJECT_NAMESPACE_PERL}")
228  set (RE "(use|require)[ \\t]+${PERL_PACKAGE}::Basis([ \\t]+.*)?") # e.g., use <Package>::Basis qw(:everything);
229  if (SCRIPT MATCHES "(^|\n|;)[ \t]*${RE}([ \t]*#.*|[ \t]*)(;|\n|$)")
230  set (UTILITIES_USED TRUE)
231  break ()
232  endif ()
233  # --------------------------------------------------------------------------
234  # Bash
235  elseif (LANGUAGE MATCHES "BASH")
236  # read script file
237  file (READ "${SOURCE_FILE}" SCRIPT)
238  # deprecated BASIS_BASH_UTILITIES macro
239  if (SCRIPT MATCHES "(^|\n|;)[ \t]*\@BASIS_BASH_UTILITIES\@")
240  message (FATAL_ERROR "Script ${SOURCE_FILE} uses the deprecated BASIS macro \@BASIS_BASH_UTILITIES\@!")
241  endif ()
242  # match source/. built-ins
243  set (RE "(source|\\.)[ \\t]+\\\"?\\\${?BASIS_BASH_UTILITIES}?\\\"?[ \\t]*(\\|\\|.*|&&.*)?(#.*)?") # e.g., . ${BASIS_BASH_UTILITIES} || exit 1
244  if (SCRIPT MATCHES "(^|\n|;)[ \t]*(${RE})[ \t]*(;|\n|$)")
245  set (UTILITIES_USED TRUE)
246  endif ()
247  endif ()
248  # return
249  set (${VAR} "${UTILITIES_USED}" PARENT_SCOPE)
250 endfunction ()

458 # ----------------------------------------------------------------------------
459 ## @brief Generate code for initialization of executable target information.
460 #
461 # This macro generates the initialization code of the executable target
462 # information dictionaries for different supported programming languages.
463 # In case of C++, the source file has been configured and copied to the binary
464 # tree in a first configuration pass such that it could be used in basis_add_*()
465 # commands which check the existence of the arguments immediately.
466 # As the generation of the initialization code requires a complete list of
467 # build targets (cached in @c BASIS_TARGETS), this function has to be called
468 # after all targets have been added and finalized (in case of custom targets).
469 #
470 # @param [in] CXX Request code for C++.
471 # @param [in] PYTHON Request code for Python.
472 # @param [in] PERL Request code for Perl.
473 # @param [in] BASH Request code for Bash.
474 #
475 # @returns Sets the following variables for each requested language.
476 #
477 # @retval EXECUTABLE_TARGET_INFO_CXX C++ code for both build tree and installation.
478 # @retval EXECUTABLE_TARGET_INFO_PYTHON_B Python code for build tree.
479 # @retval EXECUTABLE_TARGET_INFO_PYTHON_I Python code for installation.
480 # @retval EXECUTABLE_TARGET_INFO_PERL_B Perl code for build tree.
481 # @retval EXECUTABLE_TARGET_INFO_PERL_I Perl code for installation.
482 # @retval EXECUTABLE_TARGET_INFO_BASH_B Bash code for build tree.
483 # @retval EXECUTABLE_TARGET_INFO_BASH_I Bash code for installation.
484 # @retval EXECUTABLE_TARGET_INFO_BASH_A Bash code to set aliases.
485 # @retval EXECUTABLE_TARGET_INFO_BASH_S Bash code to set short aliases.
486 function (_basis_generate_executable_target_info CXX PYTHON PERL BASH)
487  # --------------------------------------------------------------------------
488  if (NOT CXX AND NOT PYTHON AND NOT PERL AND NOT BASH)
489  return ()
490  endif ()
491  # --------------------------------------------------------------------------
492  # local constants
493  basis_sanitize_for_regex (PROJECT_NAMESPACE_CMAKE_RE "${PROJECT_NAMESPACE_CMAKE}")
494  # --------------------------------------------------------------------------
495  # lists of executable targets and their location
496  set (EXECUTABLE_TARGETS)
497  set (BUILD_LOCATIONS)
498  set (INSTALL_LOCATIONS)
499  # project targets
500  basis_get_project_property (TARGETS)
501  foreach (TARGET IN LISTS TARGETS)
502  basis_get_target_type (TYPE ${TARGET})
503  if (TYPE MATCHES "EXECUTABLE")
504  basis_get_target_location (BUILD_LOCATION ${TARGET} ABSOLUTE)
505  basis_get_target_location (INSTALL_LOCATION ${TARGET} POST_INSTALL)
506  if (BUILD_LOCATION)
507  list (APPEND EXECUTABLE_TARGETS "${TARGET}")
508  list (APPEND BUILD_LOCATIONS "${BUILD_LOCATION}")
509  list (APPEND INSTALL_LOCATIONS "${INSTALL_LOCATION}")
510  else ()
511  message (FATAL_ERROR "Failed to determine build location of target ${TARGET}!")
512  endif ()
513  endif ()
514  endforeach ()
515  # imported targets
516  basis_get_project_property (IMPORTED_TARGETS)
517  basis_get_project_property (IMPORTED_TYPES)
518  basis_get_project_property (IMPORTED_LOCATIONS)
519  set (I 0)
520  list (LENGTH IMPORTED_TARGETS N)
521  while (I LESS N)
522  list (GET IMPORTED_TARGETS ${I} TARGET)
523  list (GET IMPORTED_TYPES ${I} TYPE)
524  list (GET IMPORTED_LOCATIONS ${I} LOCATION)
525  if (TYPE MATCHES "EXECUTABLE")
526  # get corresponding UID (target may be imported from other module)
527  basis_get_target_uid (UID ${TARGET})
528  # skip already considered executables
529  list (FIND EXECUTABLE_TARGETS ${UID} IDX)
530  if (IDX EQUAL -1)
531  if (LOCATION MATCHES "^NOTFOUND$")
532  message (WARNING "Imported target ${TARGET} has no location property!")
533  else ()
534  list (APPEND EXECUTABLE_TARGETS ${TARGET})
535  list (APPEND BUILD_LOCATIONS "${LOCATION}")
536  list (APPEND INSTALL_LOCATIONS "${LOCATION}")
537  endif ()
538  endif ()
539  endif ()
540  math (EXPR I "${I} + 1")
541  endwhile ()
542  # --------------------------------------------------------------------------
543  # determine maximum length of target alias for prettier output
544  set (MAX_ALIAS_LENGTH 0)
545  foreach (TARGET_UID IN LISTS EXECUTABLE_TARGETS)
546  if (TARGET_UID MATCHES "^\\.")
547  string (LENGTH "${TARGET_UID}" LENGTH)
548  else ()
549  string (LENGTH "${TOPLEVEL_PROJECT_NAMESPACE_CMAKE}.${TARGET_UID}" LENGTH)
550  endif ()
551  if (LENGTH GREATER MAX_ALIAS_LENGTH)
552  set (MAX_ALIAS_LENGTH ${LENGTH})
553  endif ()
554  endforeach ()
555  # --------------------------------------------------------------------------
556  # generate source code
557  set (CC) # C++ - build tree and install tree version, constructor block
558  set (PY_B) # Python - build tree version
559  set (PY_I) # Python - install tree version
560  set (PL_B) # Perl - build tree version, hash entries
561  set (PL_I) # Perl - install tree version, hash entries
562  set (SH_B) # Bash - build tree version
563  set (SH_I) # Bash - install tree version
564  set (SH_A) # Bash - aliases
565  set (SH_S) # Bash - short aliases
566 
567  if (CXX)
568  set (CC "// the following code was automatically generated by the BASIS")
569  set (CC "${CC}\n // CMake function basis_configure_ExecutableTargetInfo()")
570  endif ()
571 
572  set (I 0)
573  list (LENGTH EXECUTABLE_TARGETS N)
574  while (I LESS N)
575  # ------------------------------------------------------------------------
576  # get executable information
577  list (GET EXECUTABLE_TARGETS ${I} TARGET_UID)
578  list (GET BUILD_LOCATIONS ${I} BUILD_LOCATION)
579  list (GET INSTALL_LOCATIONS ${I} INSTALL_LOCATION)
580  get_target_property (BUNDLED ${TARGET_UID} BUNDLED)
581  get_target_property (IMPORTED ${TARGET_UID} IMPORTED)
582  # insert $(IntDir) for Visual Studio build location
583  if (CMAKE_GENERATOR MATCHES "Visual Studio")
584  basis_get_target_type (TYPE ${TARGET_UID})
585  if (TYPE MATCHES "^EXECUTABLE$")
586  get_filename_component (DIRECTORY "${BUILD_LOCATION}" PATH)
587  get_filename_component (FILENAME "${BUILD_LOCATION}" NAME)
588  set (BUILD_LOCATION_WITH_INTDIR "${DIRECTORY}/$(IntDir)/${FILENAME}")
589  else ()
590  set (BUILD_LOCATION_WITH_INTDIR "${BUILD_LOCATION}")
591  endif ()
592  else ()
593  set (BUILD_LOCATION_WITH_INTDIR "${BUILD_LOCATION}")
594  endif ()
595  # installation path (relative) to different library paths
596  foreach (L LIBRARY PYTHON_LIBRARY PERL_LIBRARY BASH_LIBRARY)
597  if (INSTALL_LOCATION AND (NOT IMPORTED OR BUNDLED))
598  file (
599  RELATIVE_PATH INSTALL_LOCATION_REL2${L}
600  "${CMAKE_INSTALL_PREFIX}/${INSTALL_${L}_DIR}/<package>"
601  "${INSTALL_LOCATION}"
602  )
603  else ()
604  set (INSTALL_LOCATION_REL2${L} "${INSTALL_LOCATION}")
605  endif ()
606  endforeach ()
607  # target UID including project namespace
608  if (IMPORTED OR TARGET_UID MATCHES "^\\.")
609  set (ALIAS "${TARGET_UID}")
610  elseif (NOT BASIS_USE_FULLY_QUALIFIED_TARGET_UIDS AND TOPLEVEL_PROJECT_NAMESPACE_CMAKE)
611  set (ALIAS "${TOPLEVEL_PROJECT_NAMESPACE_CMAKE}.${TARGET_UID}")
612  endif ()
613  # indentation after dictionary key, i.e., alias
614  string (LENGTH "${ALIAS}" ALIAS_LENGTH)
615  math (EXPR NUM "${MAX_ALIAS_LENGTH} - ${ALIAS_LENGTH} + 1")
616  if (NUM GREATER 1)
617  string (RANDOM LENGTH ${NUM} ALPHABET " " S)
618  else ()
619  set (S " ")
620  endif ()
621  # ------------------------------------------------------------------------
622  # C++
623  if (CXX)
624  get_filename_component (EXEC_NAME "${BUILD_LOCATION}" NAME)
625  get_filename_component (BUILD_DIR "${BUILD_LOCATION}" PATH)
626  if (INSTALL_LOCATION)
627  get_filename_component (INSTALL_DIR "${INSTALL_LOCATION}" PATH)
628  endif ()
629 
630  set (CC "${CC}\n")
631  set (CC "${CC}\n // ${TARGET_UID}")
632  set (CC "${CC}\n _exec_names [\"${ALIAS}\"]${S}= \"${EXEC_NAME}\";")
633  set (CC "${CC}\n _build_dirs [\"${ALIAS}\"]${S}= \"${BUILD_DIR}\";")
634  set (CC "${CC}\n _install_dirs[\"${ALIAS}\"]${S}= \"${INSTALL_DIR}\";")
635  endif ()
636  # ------------------------------------------------------------------------
637  # Python
638  if (PYTHON)
639  set (PY_B "${PY_B} '${ALIAS}':${S}'${BUILD_LOCATION_WITH_INTDIR}',\n")
640  if (INSTALL_LOCATION)
641  set (PY_I "${PY_I} '${ALIAS}':${S}'${INSTALL_LOCATION_REL2PYTHON_LIBRARY}',\n")
642  else ()
643  set (PY_I "${PY_I} '${ALIAS}':${S}'',\n")
644  endif ()
645  endif ()
646  # ------------------------------------------------------------------------
647  # Perl
648  if (PERL)
649  if (PL_B)
650  set (PL_B "${PL_B},\n")
651  endif ()
652  set (PL_B "${PL_B} '${ALIAS}'${S}=> '${BUILD_LOCATION_WITH_INTDIR}'")
653  if (PL_I)
654  set (PL_I "${PL_I},\n")
655  endif ()
656  if (INSTALL_LOCATION)
657  set (PL_I "${PL_I} '${ALIAS}'${S}=> '${INSTALL_LOCATION_REL2PERL_LIBRARY}'")
658  else ()
659  set (PL_I "${PL_I} '${ALIAS}'${S}=> ''")
660  endif ()
661  endif ()
662  # ------------------------------------------------------------------------
663  # Bash
664  if (BASH)
665  # hash entry
666  set (SH_B "${SH_B}\n _basis_executabletargetinfo_add '${ALIAS}'${S}LOCATION '${BUILD_LOCATION}'")
667  if (INSTALL_LOCATION)
668  set (SH_I "${SH_I}\n _basis_executabletargetinfo_add '${ALIAS}'${S}LOCATION '${INSTALL_LOCATION_REL2BASH_LIBRARY}'")
669  else ()
670  set (SH_I "${SH_I}\n _basis_executabletargetinfo_add '${ALIAS}'${S}LOCATION ''")
671  endif ()
672  # alias
673  set (SH_A "${SH_A}\n alias '${ALIAS}'=`get_executable_path '${ALIAS}'`")
674  # short alias (if target belongs to this project)
675  if (ALIAS MATCHES "^${PROJECT_NAMESPACE_CMAKE_RE}\\.")
676  basis_get_target_name (TARGET_NAME "${ALIAS}")
677  set (SH_S "${SH_S}\n alias '${TARGET_NAME}'='${ALIAS}'")
678  endif ()
679  endif ()
680  # ------------------------------------------------------------------------
681  # next executable target
682  math (EXPR I "${I} + 1")
683  endwhile ()
684  # --------------------------------------------------------------------------
685  # remove unnecessary leading newlines
686  string (STRIP "${CC}" CC)
687  string (STRIP "${PY_B}" PY_B)
688  string (STRIP "${PY_I}" PY_I)
689  string (STRIP "${PL_B}" PL_B)
690  string (STRIP "${PL_I}" PL_I)
691  string (STRIP "${SH_B}" SH_B)
692  string (STRIP "${SH_I}" SH_I)
693  string (STRIP "${SH_A}" SH_A)
694  string (STRIP "${SH_S}" SH_S)
695  # --------------------------------------------------------------------------
696  # return
697  set (EXECUTABLE_TARGET_INFO_CXX "${CC}" PARENT_SCOPE)
698  set (EXECUTABLE_TARGET_INFO_PYTHON_B "${PY_B}" PARENT_SCOPE)
699  set (EXECUTABLE_TARGET_INFO_PYTHON_I "${PY_I}" PARENT_SCOPE)
700  set (EXECUTABLE_TARGET_INFO_PERL_B "${PL_B}" PARENT_SCOPE)
701  set (EXECUTABLE_TARGET_INFO_PERL_I "${PL_I}" PARENT_SCOPE)
702  set (EXECUTABLE_TARGET_INFO_BASH_B "${SH_B}" PARENT_SCOPE)
703  set (EXECUTABLE_TARGET_INFO_BASH_I "${SH_I}" PARENT_SCOPE)
704  set (EXECUTABLE_TARGET_INFO_BASH_A "${SH_A}" PARENT_SCOPE)
705  set (EXECUTABLE_TARGET_INFO_BASH_S "${SH_S}" PARENT_SCOPE)
706 endfunction ()
