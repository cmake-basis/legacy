28 def diff3(yourtext, origtext, theirtext):
29  """Three-way diff based on the GNU diff3.c by R. Smith.
30 
31  @param [in] yourtext Array of lines of your text.
32  @param [in] origtext Array of lines of original text.
33  @param [in] theirtext Array of lines of their text.
34 
35  @returns Array of tuples containing diff results. The tuples consist of
36  (cmd, loA, hiA, loB, hiB), where cmd is either one of
37  '0', '1', '2', or 'A'.
38 
39  """
40  # diff result => [(cmd, loA, hiA, loB, hiB), ...]
41  d2 = (diff(origtext, yourtext), diff(origtext, theirtext))
42  d3 = []
43  r3 = [None, 0, 0, 0, 0, 0, 0]
44  while d2[0] or d2[1]:
45  # find a continual range in origtext lo2..hi2
46  # changed by yourtext or by theirtext.
47  #
48  # d2[0] 222 222222222
49  # origtext ...L!!!!!!!!!!!!!!!!!!!!H...
50  # d2[1] 222222 22 2222222
51  r2 = ([], [])
52  if not d2[0]: i = 1
53  else:
54  if not d2[1]: i = 0
55  else:
56  if d2[0][0][1] <= d2[1][0][1]: i = 0
57  else: i = 1
58  j = i
59  k = xor(i, 1)
60  hi = d2[j][0][2]
61  r2[j].append(d2[j].pop(0))
62  while d2[k] and d2[k][0][1] <= hi + 1:
63  hi_k = d2[k][0][2]
64  r2[k].append(d2[k].pop(0))
65  if hi < hi_k:
66  hi = hi_k
67  j = k
68  k = xor(k, 1)
69  lo2 = r2[i][ 0][1]
70  hi2 = r2[j][-1][2]
71  # take the corresponding ranges in yourtext lo0..hi0
72  # and in theirtext lo1..hi1.
73  #
74  # yourtext ..L!!!!!!!!!!!!!!!!!!!!!!!!!!!!H...
75  # d2[0] 222 222222222
76  # origtext ...00!1111!000!!00!111111...
77  # d2[1] 222222 22 2222222
78  # theirtext ...L!!!!!!!!!!!!!!!!H...
79  if r2[0]:
80  lo0 = r2[0][ 0][3] - r2[0][ 0][1] + lo2
81  hi0 = r2[0][-1][4] - r2[0][-1][2] + hi2
82  else:
83  lo0 = r3[2] - r3[6] + lo2
84  hi0 = r3[2] - r3[6] + hi2
85  if r2[1]:
86  lo1 = r2[1][ 0][3] - r2[1][ 0][1] + lo2
87  hi1 = r2[1][-1][4] - r2[1][-1][2] + hi2
88  else:
89  lo1 = r3[4] - r3[6] + lo2
90  hi1 = r3[4] - r3[6] + hi2
91  # detect type of changes
92  if not r2[0]:
93  cmd = '1'
94  elif not r2[1]:
95  cmd = '0'
96  elif hi0 - lo0 != hi1 - lo1:
97  cmd = 'A'
98  else:
99  cmd = '2'
100  for d in range(0, hi0 - lo0 + 1):
101  (i0, i1) = (lo0 + d - 1, lo1 + d - 1)
102  ok0 = (0 <= i0 and i0 < len(yourtext))
103  ok1 = (0 <= i1 and i1 < len(theirtext))
104  if xor(ok0, ok1) or (ok0 and yourtext[i0] != theirtext[i1]):
105  cmd = 'A'
106  break
107  d3.append((cmd, lo0, hi0, lo1, hi1, lo2, hi2))
108  return d3

182 def _diff_heckel(text_a, text_b):
183  """Two-way diff based on the algorithm by P. Heckel.
184 
185  @param [in] text_a Array of lines of first text.
186  @param [in] text_b Array of lines of second text.
187 
188  @returns TODO
189 
190  """
191  d = [];
192  uniq = [(len(text_a), len(text_b))]
193  (freq, ap, bp) = ({}, {}, {})
194  for i in range(len(text_a)):
195  s = text_a[i]
196  freq[s] = freq.get(s, 0) + 2;
197  ap [s] = i;
198  for i in range(len(text_b)):
199  s = text_b[i]
200  freq[s] = freq.get(s, 0) + 3;
201  bp [s] = i;
202  for s, x in freq.items():
203  if x == 5: uniq.append((ap[s], bp[s]))
204  (freq, ap, bp) = ({}, {}, {})
205  uniq.sort(key=lambda x: x[0])
206  (a1, b1) = (0, 0)
207  while a1 < len(text_a) and b1 < len(text_b):
208  if text_a[a1] != text_b[b1]: break
209  a1 += 1
210  b1 += 1
211  for a_uniq, b_uniq in uniq:
212  if a_uniq < a1 or b_uniq < b1: continue
213  (a0, b0) = (a1, b1)
214  (a1, b1) = (a_uniq - 1, b_uniq - 1)
215  while a0 <= a1 and b0 <= b1:
216  if text_a[a1] != text_b[b1]: break
217  a1 -= 1
218  b1 -= 1
219  if a0 <= a1 and b0 <= b1:
220  d.append(('c', a0 + 1, a1 + 1, b0 + 1, b1 + 1))
221  elif a0 <= a1:
222  d.append(('d', a0 + 1, a1 + 1, b0 + 1, b0))
223  elif b0 <= b1:
224  d.append(('a', a0 + 1, a0, b0 + 1, b1 + 1))
225  (a1, b1) = (a_uniq + 1, b_uniq + 1)
226  while a1 < len(text_a) and b1 < len(text_b):
227  if text_a[a1] != text_b[b1]: break
228  a1 += 1
229  b1 += 1
230  return d
