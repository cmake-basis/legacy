32 ## @brief Import Bash module.
33 #
34 # This function can be used to import, i.e., source, a named module, where the
35 # module is searched for in the directories specified by the @c BASHPATH
36 # (environment) variable. Modules can be prepended by the names of the
37 # subdirectories they are located in relative to a directory listed in the
38 # @c BASHPATH, where dots (.) can be used instead of slashes to separate the
39 # parts of the relative module path. Moreover, the <tt>.sh</tt> extension of
40 # the modules is appended to the module name if not specified.
41 #
42 # Example:
43 # @code
44 # import sbia.basis.core
45 # import -o optional.module
46 # if [ $? -eq 0 ]; then
47 # echo "Module optional.module imported successfully"
48 # else
49 # echo "Module optional.module not found"
50 # fi
51 # @endcode
52 #
53 # @note The directories listed in the @c BASHPATH must be absolute paths
54 # starting with a forward slash (/).
55 #
56 # @param [in] options Option flags. The only possible option is -o. If this
57 # option is given before the module name, the function
58 # returns with return value 1 on error. Otherwise, it
59 # calls the exit function with this exit code.
60 # It can therefore be used for optional modules, which
61 # do not necessarily need to be imported.
62 # @param [in] module Name of the module to import.
63 #
64 # @returns Only on success or if the -o option is given. Otherwise it calls
65 # the exit function if an error occurs.
66 #
67 # @retval 0 if the module was loaded successfully.
68 # @retval 1 on error if -o option is given.
69 import()
70 {
71  if [[ -z "${BASHPATH}" ]]; then
72  echo "import: BASHPATH not specified!" 1>&2
73  exit 1
74  fi
75  local exitonerror='true'
76  while [ $# -gt 0 ]; do
77  case "$1" in
78  -o) exitonerror='false'; ;;
79  *) break; ;;
80  esac
81  shift
82  done
83  local module="${1//.//}" # replace dots (.) by forward slashes (/)
84  if [[ -z "${module}" ]]; then
85  echo "import: missing module name argument!" 1>&2
86  exit 1
87  fi
88  if [[ $# -gt 1 ]]; then
89  echo "import: too many arguments!" 1>&2
90  exit 1
91  fi
92  [[ ${module: -3} == '.sh' ]] || module="${module}.sh"
93  local path="${BASHPATH}:" # ATTENTION: Trailing ':' required to terminate while loop!
94  local root=
95  while [[ -n "${path}" ]]; do
96  root="${path%%:*}"
97  if [[ "${root:0:1}" == '/' && -f "${root}/${module}" ]]; then
98  . "${root}/${module}"
99  return 0
100  fi
101  path="${path#*:}"
102  done
103  if [[ ${exitonerror} == 'true' ]]; then
104  echo "import: module '$1' not found!" 1>&2
105  exit 1
106  fi
107  return 1
108 }

114 # ----------------------------------------------------------------------------
115 ## @brief This function implements a more portable way to do pattern matching.
116 #
117 # Unfortunately, there are significant differences in the way patterns have
118 # to be matched when using different shells. This function considers which
119 # shell is used (at the moment only BASH), and uses the appropriate syntax
120 # for the pattern matching.
121 #
122 # @param [in] value The string to match against pattern.
123 # @param [in] pattern The pattern to match.
124 #
125 # @returns Whether the given string matches the given pattern.
126 #
127 # @retval 0 On match.
128 # @retval 1 Otherwise.
129 match()
130 {
131  [ $# -eq 2 ] || return 1
132 
133  local value=$1
134  local pattern=$2
135 
136  if [ -z "${value}" ]; then
137  [ -z "${pattern}" ]
138  elif [ -z "${pattern}" ]; then
139  [ -z "${value}" ]
140  else
141  if [ ${BASH_VERSION_MAJOR} -gt 2 ]; then
142  # GNU bash, version 3.00.15(1)-release (x86_64-redhat-linux-gnu)
143  # throws an error when a regular expression with groups
144  # such as in '^(a|b|c)' is used. Here, quotes are required.
145  if [ ${BASH_VERSION_MAJOR} -eq 3 -a ${BASH_VERSION_MINOR} -eq 0 ]; then
146  [[ "${value}" =~ "${pattern}" ]]
147  # GNU bash, version 3.2.25(1)-release (x86_64-redhat-linux-gnu)
148  # works with either quotes or not. However, on Mac OS Snow Leopard,
149  # GNU bash, version 3.2.48(1)-release (x86_64-apple-darwin10.0)
150  # requires that no quotes are used. The quotes are otherwise
151  # considered to be part of the pattern.
152  else
153  [[ "${value}" =~ ${pattern} ]]
154  fi
155  else
156  echo "${value}" | egrep -q "${pattern}"
157  fi
158  fi
159 }

165 # ----------------------------------------------------------------------------
166 ## @brief Assign variable one scope above the caller.
167 #
168 # This function can be used inside functions to return values by assigning
169 # them to a variable in the scope of the caller.
170 #
171 # @note For assigning multiple variables, use upvars(). Do NOT use multiple
172 # upvar() calls, since one upvar() call might reassign a variable to
173 # be used by another upvar() call.
174 #
175 # Example:
176 # @code
177 # foo ()
178 # {
179 # local "$1" && upvar $1 "Hello, World!"
180 # }
181 #
182 # foo greeting
183 # echo ${greeting}
184 # @endcode
185 #
186 # @param [in] var Variable name to assign value to
187 # @param [in] values Value(s) to assign. If multiple values, an array is
188 # assigned, otherwise a single value is assigned.
189 #
190 # @returns Nothing.
191 #
192 # @retval 0 On success.
193 # @retval 1 On failure.
194 #
195 # @sa upvars()
196 # @sa http://fvue.nl/wiki/Bash:_Passing_variables_by_reference
197 upvar()
198 {
199  if unset -v "$1"; then # Unset & validate varname
200  if (( $# == 2 )); then
201  eval $1=\"\$2\" # Return single value
202  else
203  eval $1=\(\"\${@:2}\"\) # Return array
204  fi
205  fi
206 }

208 # ----------------------------------------------------------------------------
209 ## @brief Assign variables one scope above the caller.
210 #
211 # @par Synopsis
212 # local varname [varname ...] &&
213 # upvars [-v varname value] | [-aN varname [value ...]] ...
214 #
215 # @par Options:
216 # - -aN Assign next N values to varname as array
217 # - -v Assign single value to varname#
218 #
219 # @retval 0 On success.
220 # @retval 1 On failure.
221 #
222 # @sa http://fvue.nl/wiki/Bash:_Passing_variables_by_reference
223 upvars()
224 {
225  if ! (( $# )); then
226  echo "${FUNCNAME[0]}: usage: ${FUNCNAME[0]} [-v varname"\
227  "value] | [-aN varname [value ...]] ..." 1>&2
228  return 2
229  fi
230  while (( $# )); do
231  case $1 in
232  -a*)
233  # Error checking
234  [[ ${1#-a} ]] || { echo "bash: ${FUNCNAME[0]}: \`$1': missing"\
235  "number specifier" 1>&2; return 1; }
236  printf %d "${1#-a}" &> /dev/null || { echo "bash:"\
237  "${FUNCNAME[0]}: \`$1': invalid number specifier" 1>&2
238  return 1; }
239  # Assign array of -aN elements
240  [[ "$2" ]] && unset -v "$2" && eval $2=\(\"\${@:3:${1#-a}}\"\) &&
241  shift $((${1#-a} + 2)) || { echo "bash: ${FUNCNAME[0]}:"\
242  "\`$1${2+ }$2': missing argument(s)" 1>&2; return 1; }
243  ;;
244  -v)
245  # Assign single value
246  [[ "$2" ]] && unset -v "$2" && eval $2=\"\$3\" &&
247  shift 3 || { echo "bash: ${FUNCNAME[0]}: $1: missing"\
248  "argument(s)" 1>&2; return 1; }
249  ;;
250  --help) echo "\
251 Usage: local varname [varname ...] &&
252  ${FUNCNAME[0]} [-v varname value] | [-aN varname [value ...]] ...
253 Available OPTIONS:
254 -aN VARNAME [value ...] assign next N values to varname as array
255 -v VARNAME value assign single value to varname
256 --help display this help and exit
257 --version output version information and exit"
258  return 0 ;;
259  --version) echo "\
260 ${FUNCNAME[0]}-0.9.dev
261 Copyright (C) 2010 Freddy Vulto
262 License GPLv3+: GNU GPL version 3 or later <http://gnu.org/licenses/gpl.html>
263 This is free software: you are free to change and redistribute it.
264 There is NO WARRANTY, to the extent permitted by law."
265  return 0 ;;
266  *)
267  echo "bash: ${FUNCNAME[0]}: $1: invalid option" 1>&2
268  return 1 ;;
269  esac
270  done
271 }
