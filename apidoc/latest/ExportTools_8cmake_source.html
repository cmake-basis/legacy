28 # ----------------------------------------------------------------------------
29 ## @brief Add target to export set.
30 #
31 # Targets of a subproject are added to the export set named after the subproject
32 # itself. All other targets, including in particular those of other modules,
33 # are added to the export set named after the top-level project.
34 #
35 # @param[out] EXPORT_OPTION Export option for install() command including
36 # the EXPORT option name. Set to an empty string
37 # if target is not installed.
38 # @param[in] TARGET_UID UID of target to be exported.
39 # @param[in] IS_TEST Whether given target is a test executable or library.
40 # @param[in] ARGN Optional installation destinations.
41 function (basis_add_export_target EXPORT_OPTION TARGET_UID IS_TEST)
42  if (PROJECT_IS_SUBPROJECT)
43  set (EXPORT_SET "${PROJECT_NAME}")
44  else ()
45  set (EXPORT_SET "${TOPLEVEL_PROJECT_NAME}")
46  endif ()
47  if (IS_TEST)
48  basis_set_project_property (PROJECT "${EXPORT_SET}" APPEND PROPERTY TEST_EXPORT_TARGETS "${TARGET_UID}")
49  else ()
50  basis_set_project_property (PROJECT "${EXPORT_SET}" APPEND PROPERTY EXPORT_TARGETS "${TARGET_UID}")
51  if (ARGN)
52  basis_set_project_property (PROJECT "${EXPORT_SET}" APPEND PROPERTY INSTALL_EXPORT_TARGETS "${TARGET_UID}")
53  endif ()
54  set (EXPORT_OPT "EXPORT;${EXPORT_SET}" PARENT_SCOPE)
55  endif ()
56 endfunction ()

80 # ----------------------------------------------------------------------------
81 ## @brief Get soname of object file.
82 #
83 # This function extracts the soname from object files in the ELF format on
84 # systems where the objdump command is available. On all other systems,
85 # an empty string is returned.
86 #
87 # @param [out] SONAME The soname of the object file.
88 # @param [in] OBJFILE Object file in ELF format.
89 function (basis_get_soname SONAME OBJFILE)
90  # get absolute path of object file
91  basis_get_target_uid (TARGET_UID ${OBJFILE})
92  if (TARGET TARGET_UID)
93  basis_get_target_location (OBJFILE ${TARGET_UID} ABSOLUTE)
94  string (REPLACE "$<${BASIS_GE_CONFIG}>" "${CMAKE_BUILD_TYPE}" OBJFILE "${OBJFILE}")
95  else ()
96  get_filename_component (OBJFILE "${OBJFILE}" ABSOLUTE)
97  endif ()
98  # usually CMake did this already
99  find_program (CMAKE_OBJDUMP NAMES objdump DOC "The objdump command")
100  # run objdump and extract soname
101  execute_process (
102  COMMAND ${CMAKE_OBJDUMP} -p "${OBJFILE}"
103  COMMAND sed -n "-e's/^[[:space:]]*SONAME[[:space:]]*//p'"
104  RESULT_VARIABLE STATUS
105  OUTPUT_VARIABLE SONAME_OUT
106  ERROR_QUIET
107  )
108  # return
109  if (STATUS EQUAL 0)
110  set (${SONAME} "${SONAME_OUT}" PARENT_SCOPE)
111  else ()
112  set (${SONAME} "" PARENT_SCOPE)
113  endif ()
114 endfunction ()

119  set (C "# Generated by BASIS\n\n")
120  set (C "${C}if (\"\${CMAKE_MAJOR_VERSION}.\${CMAKE_MINOR_VERSION}\" LESS 2.8)\n")
121  set (C "${C} message (FATAL_ERROR \"CMake >= 2.8.4 required\")\n")
122  set (C "${C}endif ()\n")
123  set (C "${C}cmake_policy (PUSH)\n")
124  set (C "${C}cmake_policy (VERSION 2.8.4)\n")
125  set (C "${C}#----------------------------------------------------------------\n")
126  set (C "${C}# Generated CMake target import file.\n")
127  set (C "${C}#----------------------------------------------------------------\n")
128  set (C "${C}\n# Commands may need to know the format version.\n")
129  set (C "${C}set (CMAKE_IMPORT_FILE_VERSION 1)\n")
130  set (${CODE} "${C}" PARENT_SCOPE)

145 # ----------------------------------------------------------------------------
146 ## @brief Add code to add import targets.
147 function (basis_export_import_targets CODE)
148  set (C)
149  foreach (T IN LISTS ARGN)
150  basis_get_fully_qualified_target_uid (UID "${T}")
151  set (C "${C}\n# Create import target \"${UID}\"\n")
152  get_target_property (BASIS_TYPE ${T} "BASIS_TYPE")
153  if (BASIS_TYPE MATCHES "EXECUTABLE")
154  set (C "${C}add_executable (${UID} IMPORTED)\n")
155  elseif (BASIS_TYPE MATCHES "LIBRARY|MODULE|MEX")
156  if (BASIS_TYPE MATCHES "MEX|MCC")
157  set (TYPE SHARED)
158  elseif (BASIS_TYPE MATCHES "SCRIPT")
159  set (TYPE UNKNOWN)
160  else ()
161  string (REGEX REPLACE "_LIBRARY" "" TYPE "${BASIS_TYPE}")
162  endif ()
163  set (C "${C}add_library (${UID} ${TYPE} IMPORTED)\n")
164  else ()
165  message (FATAL_ERROR "Cannot export target ${T} of type ${BASIS_TYPE}! Use NOEXPORT option.")
166  endif ()
167  set (C "${C}set_target_properties (${UID} PROPERTIES BASIS_TYPE \"${BASIS_TYPE}\")\n")
168  endforeach ()
169  set (${CODE} "${${CODE}}${C}" PARENT_SCOPE)
170 endfunction ()

172 # ----------------------------------------------------------------------------
173 ## @brief Add code to set properties of imported targets for build tree.
174 function (basis_export_build_properties CODE)
175  set (C)
176  if (CMAKE_BUILD_TYPE)
177  set (CONFIG "${CMAKE_BUILD_TYPE}")
178  else ()
179  set (CONFIG "noconfig")
180  endif ()
181  string (TOUPPER "${CONFIG}" CONFIG_U)
182  foreach (T IN LISTS ARGN)
183  basis_get_fully_qualified_target_uid (UID "${T}")
184  get_target_property (BASIS_TYPE ${T} BASIS_TYPE)
185  set (C "${C}\n# Import target \"${UID}\" for configuration \"${CONFIG}\"\n")
186  set (C "${C}set_property (TARGET ${UID} APPEND PROPERTY IMPORTED_CONFIGURATIONS ${CONFIG_U})\n")
187  set (C "${C}set_target_properties (${UID} PROPERTIES\n")
188  if (BASIS_TYPE MATCHES "SCRIPT|LIBRARY|MEX")
189  get_target_property (LANGUAGE ${T} LANGUAGE)
190  if (LANGUAGE)
191  set (C "${C} IMPORTED_LINK_INTERFACE_LANGUAGES_${CONFIG_U} \"${LANGUAGE}\"\n")
192  endif ()
193  get_target_property (LINK_DEPENDS ${T} LINK_DEPENDS)
194  set (LINK_UIDS)
195  foreach (LINK_DEPEND IN LISTS LINK_DEPENDS)
196  basis_get_fully_qualified_target_uid (LINK_UID ${LINK_DEPEND})
197  list (APPEND LINK_UIDS "${LINK_UID}")
198  endforeach ()
199  set (C "${C} IMPORTED_LINK_INTERFACE_LIBRARIES_${CONFIG_U} \"${LINK_UIDS}\"\n")
200  endif ()
201  basis_get_target_location (LOCATION ${T} ABSOLUTE)
202  string (REPLACE "$<${BASIS_GE_CONFIG}>" "${CONFIG}" LOCATION "${LOCATION}")
203  set (C "${C} IMPORTED_LOCATION_${CONFIG_U} \"${LOCATION}\"\n")
204  set (C "${C} )\n")
205  endforeach ()
206  set (${CODE} "${${CODE}}${C}" PARENT_SCOPE)
207 endfunction ()

256 # ----------------------------------------------------------------------------
257 ## @brief Export all targets added by basis_add_* commands.
258 function (basis_export_targets)
259  # parse arguments
260  CMAKE_PARSE_ARGUMENTS (ARGN "" "FILE;CUSTOM_FILE" "" ${ARGN})
261 
262  if (NOT ARGN_FILE)
263  message (FATAL_ERROR "basis_export_targets(): FILE option is required!")
264  endif ()
265  if (NOT ARGN_CUSTOM_FILE)
266  message (FATAL_ERROR "basis_export_targets(): CUSTOM_FILE option is required!")
267  endif ()
268 
269  if (IS_ABSOLUTE ${ARGN_FILE})
270  message (FATAL_ERROR "basis_export_targets(): FILE option argument must be a relative path!")
271  endif ()
272  if (IS_ABSOLUTE ${ARGN_CUSTOM_FILE})
273  message (FATAL_ERROR "basis_export_targets(): CUSTOM_FILE option argument must be a relative path!")
274  endif ()
275 
276  # --------------------------------------------------------------------------
277  # export non-custom targets
278  basis_get_project_property (EXPORT_TARGETS)
279  if (EXPORT_TARGETS)
280  if (BASIS_USE_TARGET_UIDS AND BASIS_USE_FULLY_QUALIFIED_UIDS)
281  set (NAMESPACE_OPT)
282  elseif (TOPLEVEL_PROJECT_NAMESPACE_CMAKE)
283  set (NAMESPACE_OPT NAMESPACE "${TOPLEVEL_PROJECT_NAMESPACE_CMAKE}.")
284  endif ()
285  export (
286  TARGETS ${EXPORT_TARGETS}
287  FILE "${PROJECT_BINARY_DIR}/${ARGN_FILE}"
288  ${NAMESPACE_OPT}
289  )
290  basis_get_project_property (INSTALL_EXPORT_TARGETS)
291  if (INSTALL_EXPORT_TARGETS)
292  foreach (COMPONENT "${BASIS_RUNTIME_COMPONENT}" "${BASIS_LIBRARY_COMPONENT}")
293  install (
294  EXPORT "${PROJECT_NAME}"
295  DESTINATION "${INSTALL_CONFIG_DIR}"
296  FILE "${ARGN_FILE}"
297  COMPONENT "${COMPONENT}"
298  ${NAMESPACE_OPT}
299  )
300  endforeach ()
301  endif ()
302  endif ()
303 
304  # --------------------------------------------------------------------------
305  # export custom targets and/or test targets
306  basis_get_project_property (CUSTOM_EXPORT_TARGETS)
307  basis_get_project_property (TEST_EXPORT_TARGETS)
308 
309  if (CUSTOM_EXPORT_TARGETS OR TEST_EXPORT_TARGETS)
310 
311  # write exports for build tree
312  basis_export_header (CONTENT)
313  basis_export_import_targets (CONTENT ${CUSTOM_EXPORT_TARGETS} ${TEST_EXPORT_TARGETS})
314  basis_export_build_properties (CONTENT ${CUSTOM_EXPORT_TARGETS} ${TEST_EXPORT_TARGETS})
315  basis_export_footer (CONTENT)
316 
317  file (WRITE "${PROJECT_BINARY_DIR}/${ARGN_CUSTOM_FILE}" "${CONTENT}")
318  unset (CONTENT)
319 
320  # write exports for installation - excluding test targets
321  if (CUSTOM_EXPORT_TARGETS)
322  set (INSTALL_EXPORT_FILE "${PROJECT_BINARY_DIR}/CMakeFiles/Export/${INSTALL_CONFIG_DIR}/${ARGN_CUSTOM_FILE}")
323 
324  basis_export_header (CONTENT)
325  basis_export_prefix (CONTENT)
326  basis_export_import_targets (CONTENT ${CUSTOM_EXPORT_TARGETS})
327  basis_export_install_properties (CONTENT ${CUSTOM_EXPORT_TARGETS})
328  basis_export_footer (CONTENT)
329 
330  file (WRITE "${INSTALL_EXPORT_FILE}" "${CONTENT}")
331  unset (CONTENT)
332 
333  install (
334  FILES "${INSTALL_EXPORT_FILE}"
335  DESTINATION "${INSTALL_CONFIG_DIR}"
336  )
337  endif ()
338 
339  endif ()
340 endfunction ()
