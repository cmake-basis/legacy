89 # initialize search
90 if (NOT Sphinx_DIR)
91  if (NOT $ENV{Sphinx_DIR} STREQUAL "")
92  set (Sphinx_DIR "$ENV{Sphinx_DIR}" CACHE PATH "Installation prefix of Sphinx (docutils)." FORCE)
93  else ()
94  set (Sphinx_DIR "$ENV{SPHINX_DIR}" CACHE PATH "Installation prefix of Sphinx (docutils)." FORCE)
95  endif ()
96 endif ()

98 # ----------------------------------------------------------------------------
99 # default components to look for
100 if (NOT Sphinx_FIND_COMPONENTS)
101  set (Sphinx_FIND_COMPONENTS "build" "apidoc")
102 elseif (NOT Sphinx_FIND_COMPONENTS MATCHES "^(build|apidoc)$")
103  message (FATAL_ERROR "Invalid Sphinx component in: ${Sphinx_FIND_COMPONENTS}")
104 endif ()
105 
106 # ----------------------------------------------------------------------------
107 # find components, i.e., build tools
108 foreach (_Sphinx_TOOL IN LISTS Sphinx_FIND_COMPONENTS)
109  if (Sphinx_DIR)
110  find_program (
111  Sphinx-${_Sphinx_TOOL}_EXECUTABLE
112  NAMES sphinx-${_Sphinx_TOOL} sphinx-${_Sphinx_TOOL}.py
113  HINTS "${Sphinx_DIR}"
114  PATH_SUFFIXES bin
115  DOC "The sphinx-${_Sphinx_TOOL} Python script."
116  NO_DEFAULT_PATH
117  )
118  else ()
119  find_program (
120  Sphinx-${_Sphinx_TOOL}_EXECUTABLE
121  NAMES sphinx-${_Sphinx_TOOL} sphinx-${_Sphinx_TOOL}.py
122  DOC "The sphinx-${_Sphinx_TOOL} Python script."
123  )
124  endif ()
125  mark_as_advanced (Sphinx-${_Sphinx_TOOL}_EXECUTABLE)
126  list (APPEND _Sphinx_REQUIRED_VARS Sphinx-${_Sphinx_TOOL}_EXECUTABLE)
127 endforeach ()

129 # ----------------------------------------------------------------------------
130 # determine Python executable used by Sphinx
131 if (Sphinx-build_EXECUTABLE)
132  # extract python executable from shebang of sphinx-build
133  find_package (PythonInterp QUIET)
134  set (Sphinx_PYTHON_EXECUTABLE "${PYTHON_EXECUTABLE}")
135  set (Sphinx_PYTHON_OPTIONS)
136  file (STRINGS "${Sphinx-build_EXECUTABLE}" FIRST_LINE LIMIT_COUNT 1)
137  if (FIRST_LINE MATCHES "^#!(.*/python.*)") # does not match "#!/usr/bin/env python" !
138  string (REGEX REPLACE "^ +| +$" "" Sphinx_PYTHON_EXECUTABLE "${CMAKE_MATCH_1}")
139  if (Sphinx_PYTHON_EXECUTABLE MATCHES "([^ ]+) (.*)")
140  set (Sphinx_PYTHON_EXECUTABLE "${CMAKE_MATCH_1}")
141  string (REGEX REPLACE " +" ";" Sphinx_PYTHON_OPTIONS "${CMAKE_MATCH_2}")
142  endif ()
143  endif ()
144  # this is done to avoid problems with multiple Python versions being installed
145  # remember: CMake command if(STR EQUAL STR) is bad and may cause many troubles !
146  string (REGEX REPLACE "([.+*?^$])" "\\\\\\1" _Sphinx_PYTHON_EXECUTABLE_RE "${PYTHON_EXECUTABLE}")
147  list (FIND Sphinx_PYTHON_OPTIONS -E IDX)
148  if (IDX EQUAL -1 AND NOT Sphinx_PYTHON_EXECUTABLE MATCHES "^${_Sphinx_PYTHON_EXECUTABLE_RE}$")
149  list (INSERT Sphinx_PYTHON_OPTIONS 0 -E)
150  endif ()
151  unset (_Sphinx_PYTHON_EXECUTABLE_RE)
152 endif ()
153 
154 # ----------------------------------------------------------------------------
155 # determine Sphinx version
156 if (Sphinx-build_EXECUTABLE)
157  # intentionally use invalid -h option here as the help that is shown then
158  # will include the Sphinx version information
159  if (Sphinx_PYTHON_EXECUTABLE)
160  execute_process (
161  COMMAND "${Sphinx_PYTHON_EXECUTABLE}" ${Sphinx_PYTHON_OPTIONS} "${Sphinx-build_EXECUTABLE}" -h
162  OUTPUT_VARIABLE _Sphinx_VERSION
163  ERROR_VARIABLE _Sphinx_VERSION
164  )
165  elseif (UNIX)
166  execute_process (
167  COMMAND "${Sphinx-build_EXECUTABLE}" -h
168  OUTPUT_VARIABLE _Sphinx_VERSION
169  ERROR_VARIABLE _Sphinx_VERSION
170  )
171  endif ()
172  if (_Sphinx_VERSION MATCHES "Sphinx v([0-9]+\\.[0-9]+\\.[0-9]+)")
173  set (Sphinx_VERSION_STRING "${CMAKE_MATCH_1}")
174  string (REPLACE "." ";" _Sphinx_VERSION "${Sphinx_VERSION_STRING}")
175  list(GET _Sphinx_VERSION 0 Sphinx_VERSION_MAJOR)
176  list(GET _Sphinx_VERSION 1 Sphinx_VERSION_MINOR)
177  list(GET _Sphinx_VERSION 2 Sphinx_VERSION_PATCH)
178  if (Sphinx_VERSION_PATCH EQUAL 0)
179  string (REGEX REPLACE "\\.0$" "" Sphinx_VERSION_STRING "${Sphinx_VERSION_STRING}")
180  endif ()
181  endif()
182 endif ()
