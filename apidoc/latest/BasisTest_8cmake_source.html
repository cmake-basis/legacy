153 # ----------------------------------------------------------------------------
154 ## @brief Set a property of the tests.
155 #
156 # This function replaces CMake's
157 # <a href="http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:set_tests_properties">
158 # set_tests_properties()</a> command.
159 #
160 # @note Due to a bug in CMake (http://www.cmake.org/Bug/view.php?id=12303),
161 # except of the first property given directly after the @c PROPERTIES keyword,
162 # only properties listed in @c BASIS_PROPERTIES_ON_TESTS can be set.
163 #
164 # @param [in] ARGN List of arguments for
165 # <a href="http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:set_tests_properties">
166 # set_tests_properties()</a>.
167 #
168 # @returns Sets the given properties of the specified test.
169 #
170 # @sa http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:set_tests_properties
171 #
172 # @ingroup CMakeAPI
173 function (basis_set_tests_properties)
174  # convert test names to UIDs
175  set (TEST_UIDS)
176  list (GET ARGN 0 ARG)
177  while (ARG AND NOT ARG MATCHES "^PROPERTIES$")
178  basis_get_test_uid (TEST_UID "${ARG}")
179  list (APPEND TEST_UIDS "${TEST_UID}")
180  list (REMOVE_AT ARGN 0)
181  list (GET ARGN 0 ARG)
182  endwhile ()
183  if (NOT ARG MATCHES "^PROPERTIES$")
184  message (FATAL_ERROR "Missing PROPERTIES argument!")
185  elseif (NOT TEST_UIDS)
186  message (FATAL_ERROR "No test specified!")
187  endif ()
188  # remove PROPERTIES keyword
189  list (REMOVE_AT ARGN 0)
190  # set tests properties
191  #
192  # Note: By iterating over the properties, the empty property values
193  # are correctly passed on to CMake's set_tests_properties()
194  # command, while
195  # set_tests_properties(${TEST_UIDS} PROPERTIES ${ARGN})
196  # (erroneously) discards the empty elements in ARGN.
197  if (BASIS_DEBUG)
198  message ("** basis_set_tests_properties:")
199  message ("** Test(s) : ${TEST_UIDS}")
200  message ("** Properties: [${ARGN}]")
201  endif ()
202  list (LENGTH ARGN N)
203  while (N GREATER 1)
204  list (GET ARGN 0 PROPERTY)
205  list (GET ARGN 1 VALUE)
206  list (REMOVE_AT ARGN 0 1)
207  list (LENGTH ARGN N)
208  # The following loop is only required b/c CMake's ARGV and ARGN
209  # lists do not support arguments which are themselves lists.
210  # Therefore, we need a way to decide when the list of values for a
211  # property is terminated. Hence, we only allow known properties
212  # to be set, except for the first property where the name follows
213  # directly after the PROPERTIES keyword.
214  while (N GREATER 0)
215  list (GET ARGN 0 ARG)
216  if (ARG MATCHES "${BASIS_PROPERTIES_ON_TESTS_RE}")
217  break ()
218  endif ()
219  list (APPEND VALUE "${ARG}")
220  list (REMOVE_AT ARGN 0)
221  list (LENGTH ARGN N)
222  endwhile ()
223  # check property name
224  if ("${PROPERTY}" STREQUAL "")
225  message (FATAL_ERROR "Empty property name given!")
226  endif ()
227  # map test names to test UIDs
228  if (PROPERTY MATCHES "^DEPENDS$")
229  set (UIDS)
230  foreach (V IN LISTS VALUE)
231  basis_get_test_uid (UID "${V}")
232  list (APPEND UIDS "${UID}")
233  endforeach ()
234  set (VALUE ${UIDS})
235  endif ()
236  # set target property
237  if (BASIS_DEBUG)
238  message ("** -> ${PROPERTY} = [${VALUE}]")
239  endif ()
240  set_tests_properties (${TEST_UIDS} PROPERTIES ${PROPERTY} "${VALUE}")
241  endwhile ()
242  # make sure that every property had a corresponding value
243  if (NOT N EQUAL 0)
244  message (FATAL_ERROR "No value given for test property ${ARGN}")
245  endif ()
246 endfunction ()

338 # ----------------------------------------------------------------------------
339 ## @brief Add test.
340 #
341 # This command is used similar to CMake's
342 # <a href="http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:add_test">
343 # add_test()</a> command. It adds a test to the
344 # <a href="http://www.cmake.org/cmake/help/ctest-2-8-docs.html">CTest</a>-based
345 # testing system. Unlike CMake's
346 # <a href="http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:add_test">
347 # add_test()</a>, this command can, for convenience, implicitly add
348 # the necessary executable build target to the build system. Therefore,
349 # instead of the name of the executable command, specify the sources of the
350 # test implementation. An executable build target is then added by this
351 # function using basis_add_executable(), and the built executable is used
352 # as test command. If the @p UNITTEST option is given, the necessary unit
353 # testing libraries which are part of the BASIS installation are added as
354 # link dependencies as well as the default implementation of the main()
355 # function if none of the specified source files has the suffix
356 # <tt>-main</tt> or <tt>_main</tt> in the file name.
357 #
358 # Generator expressions as supported by CMake's
359 # <a href="http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:add_test">
360 # add_test()</a> command are also supported by basis_add_test() as arguments of
361 # the test command. For the argument specifying the test command itself, however,
362 # only the generator expression $<TARGET_FILE:tgt> is allowed. Alternatively,
363 # for this special argument, the name of the executable target can be supplied
364 # directly without the use of the $<TARGET_FILE:tgt> generator expression.
365 # See documentation of basis_process_generator_expressions() for details on
366 # the supported generator expressions.
367 #
368 # Example:
369 # @code
370 # basis_add_test (COMMAND $<TARGET_FILE:basis.testdriver> $<TARGET_FILE:myexe> ...)
371 # basis_add_test (COMMAND basis.testdriver $<TARGET_FILE:myexe> ...)
372 # @endcode
373 #
374 # @param [in] TEST_NAME Name of the test. If a source file is given
375 # as first argument, the test name is derived
376 # from the name of this source file and the source
377 # file is added to the list of sources which implement
378 # the test command.
379 # @param [in] ARGN The following parameters are parsed:
380 # @par
381 # <table border="0">
382 # <tr>
383 # @tp @b COMMAND cmd [arg1 [arg2 ...]] @endtp
384 # <td>The command to execute and optionally its arguments. The command
385 # can be the name of an executable target (including imported targets),
386 # or the name or path of an executable. Alternatively,
387 # a test can be build from sources and the build executable
388 # used as command. In this case, specify the sources using the
389 # @p SOURCES argument. The command name @c cmd if given is used as
390 # output name of the built executable. If you do not want to
391 # specify the name of the output executable explicitly, but have
392 # it derived from the @p TEST_NAME, do not specify the @p COMMAND
393 # option and use the @p ARGS option instead to only specify the
394 # arguments of the test command.</td>
395 # </tr>
396 # <tr>
397 # @tp @b ARGS arg1 [arg2 ...] @endtp
398 # <td>Arguments of the test command. If this option is given, the
399 # specified arguments are appended to the arguments specified
400 # already as part of the @p COMMAND option, if any.</td>
401 # </tr>
402 # <tr>
403 # @tp @b WORKING_DIRECTORY dir @endtp
404 # <td>The working directory of the test command. The generator expression
405 # $<TARGET_FILE_DIR:tgt> can be used to specify a working directory
406 # which corresponds to the output directory of a given target file.
407 # Default: @c TESTING_OUTPUT_DIR / @p TEST_NAME. </td>
408 # </tr>
409 # <tr>
410 # @tp @b CONFIGURATIONS @endtp
411 # <td>If a CONFIGURATIONS option is given then the test will be executed
412 # only when testing under one of the named configurations.</td>
413 # </tr>
414 # <tr>
415 # @tp @b SOURCES file1 [file2 ...] @endtp
416 # <td>The source files of the test. Use the @p UNITTEST option to specify
417 # that the sources are an implementation of a unit test. In this case,
418 # the default implementation of the main() function is added to the
419 # build of the test executable. However, if this list contains a
420 # file with the suffix <tt>-main</tt> or <tt>_main</tt> in the name,
421 # the default implementation of the main() function is not used.
422 # See the documentation of the @p UNITTEST option for further details.</td>
423 # </tr>
424 # <tr>
425 # @tp @b LINK_DEPENDS file1|target1 [file2|target2 ...] @endtp
426 # <td>Link dependencies of test executable build from sources.</td>
427 # </tr>
428 # <tr>
429 # @tp @b NO_DEFAULT_MAIN @endtp
430 # <td>Force that the implementation of the default main() function
431 # is not added to unit tests even if neither of the given source
432 # files has the suffix <tt>-main</tt> or <tt>_main</tt> in the
433 # file name.</td>
434 # </tr>
435 # <tr>
436 # @tp @b UNITTEST @endtp
437 # <td>Specifies that the test is a unit test. In this case, the test
438 # implementation is linked to the default unit testing framework
439 # for the used programming language which is part of the BASIS
440 # installation.</td>
441 # </tr>
442 # <tr>
443 # @tp @b WITH_EXT @endtp
444 # <td>Do not strip extension if test name is derived from source file name.</td>
445 # </tr>
446 # <tr>
447 # @tp @b ARGN @endtp
448 # <td>All other arguments are passed on to basis_add_executable() if
449 # an executable target for the test is added.</td>
450 # </tr>
451 # </table>
452 #
453 # @returns Adds build target for test executable if test source files
454 # are given and/or adds a CTest test which executes the given
455 # test command.
456 #
457 # @sa basis_process_generator_expressions()
458 # @sa http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:add_test
459 #
460 # @todo Make use of ExternalData module to fetch remote test data.

619 ## @brief Add tests of default options for given executable.
620 #
621 # @par Default options:
622 # <table border="0">
623 # <tr>
624 # @tp <b>--helpshort</b> @endtp
625 # <td>Short help. The output has to match the regular expression
626 # "[Uu]sage:\n\s*\<executable name\>", where <executable name>
627 # is the name of the tested executable.</td>
628 # </tr>
629 # <tr>
630 # @tp <b>--help, -h</b> @endtp
631 # <td>Help screen. Simply tests if the option is accepted.</td>
632 # </tr>
633 # <tr>
634 # @tp <b>--version</b> @endtp
635 # <td>Version information. Output has to include the project version string.</td>
636 # </tr>
637 # <tr>
638 # @tp <b>--verbose, -v</b> @endtp
639 # <td>Increase verbosity of output messages. Simply tests if the option is accepted.</td>
640 # </tr>
641 # </table>
642 #
643 # @param [in] TARGET_NAME Name of executable or script target.
644 #
645 # @returns Adds tests for the default options of the specified executable.
646 function (basis_add_tests_of_default_options TARGET_NAME)
647  basis_get_target_uid (TARGET_UID "${TARGET_NAME}")
648 
649  message (STATUS "Adding tests of default options for ${TARGET_UID}...")
650 
651  if (NOT TARGET "${TARGET_UID}")
652  message (FATAL_ERROR "Unknown target ${TARGET_UID}.")
653  endif ()
654 
655  # get executable name
656  get_target_property (PREFIX ${TARGET_UID} "PREFIX")
657  get_target_property (OUTPUT_NAME ${TARGET_UID} "OUTPUT_NAME")
658  get_target_property (SUFFIX ${TARGET_UID} "SUFFIX")
659 
660  if (NOT OUTPUT_NAME)
661  set (EXEC_NAME "${TARGET_UID}")
662  endif ()
663  if (PREFIX)
664  set (EXEC_NAME "${PREFIX}${EXEC_NAME}")
665  endif ()
666  if (SUFFIX)
667  set (EXEC_NAME "${EXEC_NAME}${SUFFIX}")
668  endif ()
669 
670  # get absolute path to executable
671  get_target_property (EXEC_DIR ${TARGET_UID} "RUNTIME_OUTPUT_DIRECTORY")
672 
673  # executable command
674  set (EXEC_CMD "${EXEC_DIR}/${EXEC_NAME}")
675 
676  # test option: -v
677  basis_add_test (${EXEC}VersionS "${EXEC_CMD}" "-v")
678 
679  set_tests_properties (
680  ${EXEC}VersionS
681  PROPERTIES
682  PASS_REGULAR_EXPRESSION "${EXEC} ${PROJECT_VERSION}"
683  )
684 
685  # test option: --version
686  basis_add_test (${EXEC}VersionL "${EXEC_CMD}" "--version")
687 
688  set_tests_properties (
689  ${EXEC}VersionL
690  PROPERTIES
691  PASS_REGULAR_EXPRESSION "${EXEC} ${PROJECT_VERSION}"
692  )
693 
694  # test option: -h
695  basis_add_test (${EXEC}HelpS "${EXEC_CMD}" "-h")
696 
697  # test option: --help
698  basis_add_test (${EXEC}HelpL "${EXEC_CMD}" "--help")
699 
700  # test option: --helpshort
701  basis_add_test (${EXEC}UsageL "${EXEC_CMD}" "--helpshort")
702 
703  set_tests_properties (
704  ${EXEC}UsageL
705  PROPERTIES
706  PASS_REGULAR_EXPRESSION "[Uu]sage:(\n)( )*${EXEC_NAME}"
707  )
708 
709  message (STATUS "Adding tests of default options for ${EXEC}... - done")
710 endfunction ()
