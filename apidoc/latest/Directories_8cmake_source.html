121 ## @brief Path of installation directory for CMake package configuration
122 # files relative to @c CMAKE_INSTALL_PREFIX.
123 #
124 # Install configuration files of modules in subdirectories such that
125 # CMake does not find them by default. Their might be a case that
126 # someone is using two projects where the one project is named just
127 # the same as the module of the other project. The configuration file
128 # of the module is anyway supposed to be included by the package
129 # configuration file if this module is requested by the COMPONENTS
130 # argument to the find_package() command.
131 set (INSTALL_CONFIG_DIR "lib/cmake/basis")
132 ## @brief Path of installation directory for runtime executables and shared
133 # libraries on Windows relative to @c CMAKE_INSTALL_PREFIX.
134 set (INSTALL_RUNTIME_DIR "bin")
135 ## @brief Path of installation directory for auxiliary executables
136 # relative to @c CMAKE_INSTALL_PREFIX.
137 set (INSTALL_LIBEXEC_DIR "lib")
138 ## @brief Path of installation directory for public header files
139 # relative to @c CMAKE_INSTALL_PREFIX.
140 #
141 # @note If the @c INCLUDE_PREFIX specifies subdirectories, these will be
142 # created underneath this directory during the installation.
143 set (INSTALL_INCLUDE_DIR "include")
144 ## @brief Path of installation directory for shared libraries on Unix-based
145 # systems and module libraries relative to @c CMAKE_INSTALL_PREFIX.
146 set (INSTALL_LIBRARY_DIR "lib")
147 ## @brief Path of installation directory for static and import libraries
148 # relative to @c CMAKE_INSTALL_PREFIX.
149 set (INSTALL_ARCHIVE_DIR "lib")
150 
151 # Similar to the public header files of C/C++ libraries, the modules written
152 # in Python or Perl are installed with fixed relative directories which
153 # correspond to the packages these modules belong to:
154 #
155 # For example, the Python module <vendor>.<pkg>.<module> is installed
156 # in the subdirectory <vendor>/<pkg>/<module>/ relative to the
157 # installation path for public Python modules, i.e., INSTALL_PYTHON_LIBRARY_DIR.
158 # External users of this module will add the INSTALL_PYTHON_LIBRARY_DIR
159 # to the search path for Python modules and import the module, for example, as
160 # follows:
161 #
162 # from <vendor>.<pkg> import <module>
163
