58 string exepath(const string& name, const IExecutableTargetInfo* targets)
59 {
60  // return path of this executable if no name given
61  if (name.empty()) return os::exepath();
62  // get name of executable and check if target name is known
63  string exec_name = targets != NULL ? targets->basename(name) : "";
64  // if target is not known
65  if (exec_name.empty()) {
66  // return input argument assuming that it is already the path of
67  // an executable file
68  exec_name = name;
69  // try to get absolute path using the which command if path is relative
70  // TODO Replace use of external which command by C++ implementation
71  // of which() even though BASIS includes a Python implementation
72  // that can also be used on Windows. Still, a native C++
73  // implementation is desireable.
74  if (!os::path::isabs(exec_name) && targets->istarget("basis.which")) {
75  vector<string> which(2);
76  which[0] = "basis.which";
77  which[1] = name;
78  ostringstream oss;
79  // attention: this includes a "recursive" call of this function!
80  if (execute(which, true, &oss, true, 0, false, targets) == 0) {
81  exec_name = oss.str();
82  const string::size_type end = exec_name.find_last_not_of(" \t\n\r");
83  if (end == string::npos) {
84  exec_name = "";
85  } else {
86  exec_name.erase(end + 1);
87  }
88  }
89  }
90  return exec_name;
91  }
92  return os::path::join(targets->dirname(name), exec_name);
93 }

141 int execute(vector<string> args, bool quiet, ostream* out,
142  bool allow_fail, int verbose, bool simulate,
143  const IExecutableTargetInfo* targets)
144 {
145  if (args.empty() || args[0].empty()) {
146  BASIS_THROW(SubprocessError, "execute_process(): No command specified");
147  }
148  // map build target name to executable file path
149  string exec_path = exepath(args[0], targets);
150  // prepend absolute path of found executable
151  if (!exec_path.empty()) args[0] = exec_path;
152  // some verbose output
153  if (verbose > 0 || simulate) {
154  cout << "$ " << Subprocess::tostring(args);
155  if (simulate) cout << " (simulated)";
156  cout << endl;
157  }
158  // execute command
159  char buf[1024];
160  int n;
161  int status = 0;
162  Subprocess p;
163  if (!p.popen(args, Subprocess::RM_NONE, Subprocess::RM_PIPE, Subprocess::RM_PIPE)) {
164  BASIS_THROW(SubprocessError, "execute_process(): Failed to create subprocess");
165  }
166  // read child's stdout (blocking)
167  if (!quiet || out != NULL) {
168  while ((n = p.read(buf, 1023)) > 0) {
169  buf[n] = '\0';
170  if (!quiet) {
171  cout << buf;
172  cout.flush();
173  }
174  if (out) *out << buf;
175  }
176  }
177  // wait for child process
178  if (!p.wait()) {
179  BASIS_THROW(SubprocessError, "execute_process(): Failed to wait for subprocess");
180  }
181  // write error messages to stderr of parent process
182  while ((n = p.read(buf, 1023, true)) > 0) {
183  buf[n] = '\0';
184  cerr << buf;
185  }
186  // get exit code
187  status = p.returncode();
188  // if command failed, throw an exception
189  if (status != 0 && !allow_fail) {
190  BASIS_THROW(SubprocessError, "Command " << Subprocess::tostring(args) << " failed");
191  }
192  return status;
193 }
