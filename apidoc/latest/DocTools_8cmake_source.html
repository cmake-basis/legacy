27 ## @brief Add documentation target.
28 #
29 # This function is used to add a software documentation files to the project
30 # which are either just copied to the installation or generated from input
31 # files such as in particular source code files and documentation files
32 # marked up using one of the supported lightweight markup languages.
33 #
34 # The supported generators are:
35 # <table border="0">
36 # <tr>
37 # @tp @b None @endtp
38 # <td>This generator simply installs the given file or all files within
39 # the specified directory.</td>
40 # </tr>
41 # <tr>
42 # @tp @b Doxygen @endtp
43 # <td>Used to generate API documentation from in-source code comments and
44 # other related files marked up using Doxygen comments. See
45 # basis_add_doxygen_doc() for more details.</td>
46 # </tr>
47 # <tr>
48 # @tp @b Sphinx @endtp
49 # <td>Used to generate documentation such as a web site from reStructuredText.
50 # See basis_add_sphinx_doc() for more details.</td>
51 # </tr>
52 # </table>
53 #
54 # @param [in] TARGET_NAME Name of the documentation target or file.
55 # @param [in] ARGN Documentation generator as "GENERATOR generator" option
56 # and additional arguments for the particular generator.
57 # The case of the generator name is ignored, i.e.,
58 # @c Doxygen, @c DOXYGEN, @c doxYgen are all valid arguments
59 # which select the @c Doxygen generator. The default generator
60 # is the @c None generator.</td>
61 #
62 # @returns Adds a custom target @p TARGET_NAME for the generation of the
63 # documentation.
64 #
65 # @sa basis_install_doc()
66 # @sa basis_add_doxygen_doc()
67 # @sa basis_add_sphinx_doc()
68 #
69 # @ingroup CMakeAPI
70 function (basis_add_doc TARGET_NAME)
71  CMAKE_PARSE_ARGUMENTS (ARGN "" "GENERATOR" "" ${ARGN})
72  if (NOT ARGN_GENERATOR)
73  set (ARGN_GENERATOR "NONE")
74  else ()
75  string (TOUPPER "${ARGN_GENERATOR}" ARGN_GENERATOR)
76  endif ()
77  if (ARGN_GENERATOR MATCHES "NONE")
78  basis_install_doc (${TARGET_NAME} ${ARGN_UNPARSED_ARGUMENTS})
79  elseif (ARGN_GENERATOR MATCHES "DOXYGEN")
80  basis_add_doxygen_doc (${TARGET_NAME} ${ARGN_UNPARSED_ARGUMENTS})
81  elseif (ARGN_GENERATOR MATCHES "SPHINX")
82  basis_add_sphinx_doc (${TARGET_NAME} ${ARGN_UNPARSED_ARGUMENTS})
83  else ()
84  message (FATAL_ERROR "Unknown documentation generator: ${ARGN_GENERATOR}.")
85  endif ()
86 endfunction ()

103 # @par
104 # <table border="0">
105 # <tr>
106 # @tp @b COMPONENT component @endtp
107 # <td>Name of the component this documentation belongs to.

109 # </tr>

112 # <td>Installation directory prefix. Defaults to @c INSTALL_DOC_DIR.</td>

115 # @tp @b OUTPUT_NAME name @endtp
116 # <td>Name of file or directory after installation.</td>
117 # </tr>
118 # </table>

121 function (basis_install_doc SOURCE)
122  CMAKE_PARSE_ARGUMENTS (ARGN "" "COMPONENT;DESTINATION;OUTPUT_NAME" "" ${ARGN})
123 
124  if (NOT IS_ABSOLUTE "${SOURCE}")
125  get_filename_component (SOURCE "${SOURCE}" ABSOLUTE)
126  endif ()
127  if (NOT ARGN_DESTINATION)
128  set (ARGN_DESTINATION "${INSTALL_DOC_DIR}")
129  endif ()
130  if (NOT ARGN_COMPONENT)
131  set (ARGN_COMPONENT "${BASIS_RUNTIME_COMPONENT}")
132  endif ()
133  if (NOT ARGN_COMPONENT)
134  set (ARGN_COMPONENT "Unspecified")
135  endif ()
136  if (NOT ARGN_OUTPUT_NAME)
137  basis_get_filename_component (ARGN_OUTPUT_NAME "${SOURCE}" NAME)
138  endif ()
139 
140  basis_get_relative_path (
141  RELPATH
142  "${CMAKE_SOURCE_DIR}"
143  "${CMAKE_CURRENT_SOURCE_DIR}/${ARGN_OUTPUT_NAME}"
144  )
145 
146  message (STATUS "Adding documentation ${RELPATH}...")
147 
148  if (IS_DIRECTORY "${SOURCE}")
149  basis_install_directory (
150  "${SOURCE}" "${ARGN_DESTINATION}/${ARGN_OUTPUT_NAME}"
151  COMPONENT "${ARGN_COMPONENT}"
152  )
153  else ()
154  install (
155  FILES "${SOURCE}"
156  DESTINATION "${ARGN_DESTINATION}"
157  COMPONENT "${ARGN_COMPONENT}"
158  RENAME "${ARGN_OUTPUT_NAME}"
159  )
160  endif ()
161 
162  message (STATUS "Adding documentation ${RELPATH}... - done")
163 endfunction ()
