185 int binary_diff(const char* testfile, const char* baseline)
186 {
187  int retval = 0;
188  ifstream ift(testfile, ios::binary);
189  ifstream ifb(baseline, ios::binary);
190  if (!ift) return -1;
191  if (!ifb) return -2;
192  istream_iterator<unsigned char> eos; // end-of-stream
193  istream_iterator<unsigned char> it(ift);
194  istream_iterator<unsigned char> ib(ifb);
195  while (it != eos && ib != eos) {
196  if (*it != *ib) break;
197  ++it;
198  ++ib;
199  }
200  if (it != eos || ib != eos) retval = 1;
201  ift.close();
202  ifb.close();
203  return retval;
204 }

226 int text_diff_lines(const char* testfile, const char* baseline, unsigned int max_number_of_differences)
227 {
228  int retval = 0;
229  ifstream ift(testfile);
230  ifstream ifb(baseline);
231  if (!ift) return -1;
232  if (!ifb) return -2;
233  string tline, bline;
234  while (getline(ift, tline) && getline(ifb, bline)) {
235  if (tline != bline) retval++;
236  }
237  if (static_cast<unsigned int>(retval) <= max_number_of_differences) retval = 0;
238  if (getline(ift, tline) || getline(ifb, bline)) retval = 1000;
239  ift.close();
240  ifb.close();
241  return retval;
242 }

268 vector<string> get_baseline_filenames(string filename_template)
269 {
270  vector<string> baselines;
271 
272  ifstream ifs(filename_template.c_str());
273  if (ifs) baselines.push_back(filename_template);
274 
275  int x = 0;
276  string::size_type pos = filename_template.rfind(".");
277  string suffix;
278 
279  if (pos != string::npos) {
280  suffix = filename_template.substr(pos);
281  filename_template.erase(pos);
282  }
283  while (++x) {
284  ostringstream filename;
285  filename << filename_template << '.' << x << suffix;
286  ifstream ifs(filename.str().c_str());
287  if (!ifs) break;
288  ifs.close();
289  baselines.push_back(filename.str());
290  }
291  return baselines;
292 }
