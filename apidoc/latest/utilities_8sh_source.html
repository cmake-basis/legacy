11 # @file utilities.sh
12 # @brief Main module of project-independent BASIS utilities.
13 #
14 # This module defines the default BASIS utility functions. These default
15 # implementations are not project-specific, i.e., do not make use of particular
16 # project attributes such as the name or version of the project. The utility
17 # functions defined by this module are intended for use in Bash scripts that
18 # are not build as part of a particular BASIS project. Otherwise, the
19 # project-specific implementations should be used instead, i.e., those defined
20 # by the basis.sh module of the project which is automatically added to the
21 # project during the configuration of the build tree. This basis.sh module and
22 # the submodules used by it are generated from template modules which are
23 # customized for the particular project that is being build.
24 #
25 # Besides the utility functions which are common to all implementations for
26 # the different programming languages, does this module further provide
27 # fundamental functions for the development in Bash.
28 #
29 # @note In Bash, there is no concept of namespaces. Hence, the utility functions
30 # are all defined by the utilities.sh module which is part of the BASIS
31 # installation. By simply setting the constants to the project specific
32 # values, these utility functions are customized for this particular
33 # package. This, however, also means that the BASIS utilities of two
34 # different packages cannot be used within a Bash script at the same
35 # time in general. The order in which the basis.sh modules are sourced
36 # matters. Therefore, in Bash, care must be taken which modules of a
37 # BASIS-based package are being sourced and whether these in turn
38 # source either the utilities.sh module of BASIS or the basis.sh module
39 # which has been configured/customized for this particular package.
40 # If all modules make only use of the utilities.sh module, there are
41 # no conflicts. Thus, Bash should in general only be used for executable
42 # scripts within a project, but not to provide library functions to
43 # other developers. Therefore, consider the use of C++, Python, or Perl,
44 # instead.
45 #

118 ## @brief Print version information including copyright and license notices.
119 #
120 # Example:
121 # @code
122 # print_version 'foo' '1.0'
123 # print_version 'foo' -v '1.0'
124 # print_version -v 1.0 -p BarStool 'foo'
125 # print_version 'foo' -v '1.2' -c '2012 University of Pennsylvania' \
126 # -l 'Apache License, Version 2.0'
127 # @endcode
128 #
129 # @param [in] options Function options as documented below.
130 # @param [in] name Name of executable. Should not be set programmatically
131 # to the first argument of the main script, but a string
132 # literal instead.
133 # @param [in] version Version of executable. Defaults to <tt>${RELEASE}</tt>
134 # if defined, otherwise this argument is required.
135 # @par Options:
136 # <table border="0">
137 # <tr>
138 # @tp @b -v, @b --version &lt;version&gt; @endtp
139 # <td>Version of executable. Can be either given as option or second
140 # positional argument after the @p name.</td>
141 # </tr>
142 # <tr>
143 # @tp @b -p, @b --project &lt;name&gt; @endtp
144 # <td>Name of project this executable belongs to.
145 # Defaults to <tt>${PROJECT}</tt> if defined.
146 # If 'none', no project information is printed.</td>
147 # </tr>
148 # <tr>
149 # @tp @b -c @b --copyright &lt;copyright&gt; @endtp
150 # <td>The copyright notice. Defaults to <tt>${COPYRIGHT}</tt>.
151 # If 'none', no copyright notice is printed.</td>
152 # </tr>
153 # <tr>
154 # @tp @b -l @b --license &lt;license&gt; @endtp
155 # <td>Information regarding licensing. Defaults to <tt>${LICENSE}</tt>.
156 # If 'none', no license information is printed.</td>
157 # </tr>
158 # </table>
159 #
160 # @returns Nothing.
161 print_version()
162 {
163  local _basis_pv_name=''
164  local _basis_pv_version="${RELEASE}"
165  local _basis_pv_project="${PROJECT}"
166  local _basis_pv_copyright="${COPYRIGHT:-}"
167  local _basis_pv_license="${LICENSE:-}"
168  while [ $# -gt 0 ]; do
169  case "$1" in
170  -v|--version)
171  if [ -n "${_basis_pv_version}" ]; then
172  echo "print_version(): Version specified twice!" 1>&2
173  return 1
174  fi
175  if [ $# -gt 1 ]; then
176  _basis_pv_version="$2"
177  else
178  echo "print_version(): Option -v, --version is missing an argument!" 1>&2
179  return 1
180  fi
181  shift
182  ;;
183  -p|--project)
184  if [ $# -gt 1 ]; then
185  _basis_pv_project="$2"
186  else
187  echo "print_version(): Option -p, --project is missing an argument!" 1>&2
188  return 1
189  fi
190  shift
191  ;;
192  -c|--copyright)
193  if [ $# -gt 1 ]; then
194  _basis_pv_copyright="$2"
195  else
196  echo "print_version(): Option -c, --copyright is missing an argument!" 1>&2
197  return 1
198  fi
199  shift
200  ;;
201  -l|--license)
202  if [ $# -gt 1 ]; then
203  _basis_pv_license="$2"
204  else
205  echo "print_version(): Option -l, --license is missing an argument!" 1>&2
206  return 1
207  fi
208  shift
209  ;;
210  *)
211  if [ -z "${_basis_pv_name}" ]; then
212  _basis_pv_name=$1
213  elif [ -z "${_basis_pv_version}" ]; then
214  _basis_pv_version=$1
215  else
216  echo "print_version(): Too many arguments or invalid option: $1" 1>&2
217  return 1
218  fi
219  ;;
220  esac
221  shift
222  done
223  [ -n "${_basis_pv_name}" ] || { echo "print_version(): Missing name argument" 1>&2; return 1; }
224  [ -n "${_basis_pv_version}" ] || { echo "print_version(): Missing version argument" 1>&2; return 1; }
225  echo -n "${_basis_pv_name}"
226  [ -n "${_basis_pv_project}" ] && [ "${_basis_pv_project}" != 'none' ] && {
227  echo -n " (${_basis_pv_project})"
228  }
229  echo " ${_basis_pv_version}"
230  [ -n "${_basis_pv_copyright}" ] && [ "${_basis_pv_copyright}" != 'none' ] && {
231  echo -e "Copyright (c) ${_basis_pv_copyright}. All rights reserved."
232  }
233  [ -n "${_basis_pv_license}" ] && [ "${_basis_pv_license}" != 'none' ] && {
234  echo -e "${_basis_pv_license}"
235  }

238 # ----------------------------------------------------------------------------
239 ## @brief Get UID of build target.
240 #
241 # The UID of a build target is its name prepended by a namespace identifier
242 # which should be unique for each project.
243 #
244 # This function further initializes the dictionary storing the information
245 # about the executable targets upon the first invocation. Reason to do it
246 # here is that every access to the dictionaries first calls this function
247 # to get the UID of a build target. Moreover, also this function needs to
248 # have the already initialized dictionaries to ensure that an already valid
249 # target identifier is not modified. As Bash does not provide hash tables,
250 # dictionary data structures, the imitation of these is necessary which,
251 # however, results in many eval() calls with noticeable impact on running time.
252 # Therefore, to decrease the time required to source the BASIS utilities,
253 # the required dictionary structure is initialized only upon first use.
254 #
255 # @param [out] uid UID of named build target.
256 # @param [in] name Name of build target.
257 #
258 # @returns Nothing.
259 #
260 # @retval 0 On success.
261 # @retval 1 On failure.
262 targetuid()
263 {
264  [ -n "$1" ] && [ $# -eq 2 ] || return 1
265  local _basis_targetuid_target="$2"
266  # initialize module if not done yet - this is only done here because
267  # whenever information is looked up about an executable target, this
268  # function is invoked first
269  if [ "${_BASIS_EXECUTABLETARGETINFO_INITIALIZED}" != 'true' ]; then
270  _basis_executabletargetinfo_initialize || return 1
271  fi
272  # empty string as input remains unchanged
273  [ -z "${_basis_targetuid_target}" ] && local "$1" && upvar $1 '' && return 0
274  # in case of a leading namespace separator, do not modify target name
275  [ "${_basis_targetuid_target:0:1}" == '.' ] && local "$1" && upvar $1 "${_basis_targetuid_target}" && return 0
276  # project namespace
277  local _basis_targetuid_prefix="${_BASIS_TARGET_UID_PREFIX}.DUMMY"
278  # try prepending namespace or parts of it until target is known
279  local _basis_targetuid_path=''
280  while [ "${_basis_targetuid_prefix/\.*/}" != "${_basis_targetuid_prefix}" ]; do
281  _basis_targetuid_prefix="${_basis_targetuid_prefix%\.*}"
282  _basis_executabletargetinfo_get _basis_targetuid_path "${_basis_targetuid_prefix}.${_basis_targetuid_target}" LOCATION
283  if [ -n "${_basis_targetuid_path}" ]; then
284  local "$1" && upvar $1 "${_basis_targetuid_prefix}.${_basis_targetuid_target}"
285  return 0
286  fi
287  done
288  # otherwise, return target name unchanged
289  local "$1" && upvar $1 "${_basis_targetuid_target}"
290 }

306 # ----------------------------------------------------------------------------
307 ## @brief Get absolute path of executable file.
308 #
309 # This function determines the absolute file path of an executable. If no
310 # arguments are given, the absolute path of this executable is returned.
311 # If the given argument is a known build target name, the absolute path
312 # of the executable built by this target is returned. Otherwise, the named
313 # command is searched in the system PATH and it's absolute path returned
314 # if found. If the given argument is neither the name of a known build target
315 # nor an executable found on the PATH, an empty string is returned and
316 # the return value is 1.
317 #
318 # @param [out] path Absolute path of executable file.
319 # @param [in] target Name/UID of build target. If no argument is given,
320 # the file path of the calling executable is returned.
321 #
322 # @returns Nothing.
323 #
324 # @retval 0 On success.
325 # @retval 1 On failure.
326 exepath()
327 {
328  [ -n "$1" ] && [ $# -eq 1 -o $# -eq 2 ] || return 1
329  local _basis_exepath_path=''
330  # if no target name given, get path of this executable
331  if [ -z "$2" ]; then
332  _basis_exepath_path="`realpath "$0"`"
333  # otherwise, get path of executable built by named target
334  else
335  # get UID of target
336  local _basis_exepath_uid && targetuid _basis_exepath_uid "$2"
337  [ "${_basis_exepath_uid:0:1}" == '.' ] && _basis_exepath_uid=${_basis_exepath_uid:1}
338  if [ -n "${_basis_exepath_uid}" ]; then
339  # get path relative to this module
340  _basis_executabletargetinfo_get _basis_exepath_path "${_basis_exepath_uid}" LOCATION
341  if [ -n "${_basis_exepath_path}" ]; then
342  # make path absolute
343  _basis_exepath_path=`abspath "${_BASIS_EXECUTABLE_TARGETS_BASE}" "${_basis_exepath_path}"`
344  [ $? -eq 0 ] || return 1
345  else
346  _basis_exepath_path=`/usr/bin/which "$2" 2> /dev/null`
347  fi
348  else
349  _basis_exepath_path=`/usr/bin/which "$2" 2> /dev/null`
350  fi
351  fi
352  # return path
353  local "$1" && upvar $1 "${_basis_exepath_path}"
354  [ $? -eq 0 ] && [ -n "${_basis_exepath_path}" ]
355 }

401 # ----------------------------------------------------------------------------
402 ## @brief Build quoted string from array.
403 #
404 # Example:
405 # @code
406 # tostring str 'this' "isn't" a 'simple example of "a quoted"' 'string'
407 # echo "${str}"
408 # @endcode
409 #
410 # @param [out] var Name of result variable for quoted string.
411 # @param [in] elements All remaining arguments are considered to be the
412 # elements of the array to convert.
413 #
414 # @returns Nothing.
415 tostring()
416 {
417  local _basis_tostring_str=''
418  local _basis_tostring_element=''
419  # GNU bash, version 3.00.15(1)-release (x86_64-redhat-linux-gnu)
420  # turns the array into a single string value if local is used
421  if [ ${BASH_VERSION_MAJOR} -gt 3 ] || [ ${BASH_VERSION_MAJOR} -eq 3 -a ${BASH_VERSION_MINOR} -gt 0 ]; then
422  local _basis_tostring_args=("$@")
423  else
424  _basis_tostring_args=("$@")
425  fi
426  local _basis_tostring_i=1 # first argument is name of return variable
427  while [ $_basis_tostring_i -lt ${#_basis_tostring_args[@]} ]; do
428  _basis_tostring_element="${_basis_tostring_args[$_basis_tostring_i]}"
429  # escape double quotes
430  _basis_tostring_element=`printf -- "${_basis_tostring_element}" | sed 's/\\"/\\\\"/g'`
431  # surround element by double quotes if necessary
432  match "${_basis_tostring_element}" "[' ]|^$" && _basis_tostring_element="\"${_basis_tostring_element}\""
433  # append element
434  [ -n "${_basis_tostring_str}" ] && _basis_tostring_str="${_basis_tostring_str} "
435  _basis_tostring_str="${_basis_tostring_str}${_basis_tostring_element}"
436  # next argument
437  let _basis_tostring_i++
438  done
439  local "$1" && upvar $1 "${_basis_tostring_str}"
440 }

442 # ----------------------------------------------------------------------------
443 ## @brief Split (quoted) string.
444 #
445 # This function can be used to split a (quoted) string into its elements.
446 #
447 # Example:
448 # @code
449 # str="'this' 'isn\'t' a \"simple example of \\\"a quoted\\\"\" 'string'"
450 # qsplit array "${str}"
451 # echo ${#array[@]} # 5
452 # echo "${array[3]}" # simple example of "a quoted"
453 # @endcode
454 #
455 # @param [out] var Result variable for array.
456 # @param [in] str Quoted string.
457 #
458 # @returns Nothing.
459 qsplit()
460 {
461  [ $# -eq 2 ] || return 1
462  # GNU bash, version 3.00.15(1)-release (x86_64-redhat-linux-gnu)
463  # turns the array into a single string value if local is used
464  if [ ${BASH_VERSION_MAJOR} -gt 3 ] || [ ${BASH_VERSION_MAJOR} -eq 3 -a ${BASH_VERSION_MINOR} -gt 0 ]; then
465  local _basis_qsplit_array=()
466  else
467  _basis_qsplit_array=()
468  fi
469  local _basis_qsplit_str=$2
470  # match arguments from left to right
471  while match "${_basis_qsplit_str}" "[ ]*('([^']|\\\')*[^\\]'|\"([^\"]|\\\")*[^\\]\"|[^ ]+)(.*)"; do
472  # matched element including quotes
473  _basis_qsplit_element="${BASH_REMATCH[1]}"
474  # remove quotes
475  if [[ ${_basis_qsplit_element:0:1} == '"' && ${_basis_qsplit_element: -1} == '"' ]]; then
476  _basis_qsplit_element="${_basis_qsplit_element:1}"
477  _basis_qsplit_element="${_basis_qsplit_element%\"}"
478  elif [[ ${_basis_qsplit_element:0:1} == "'" && ${_basis_qsplit_element: -1} == "'" ]]; then
479  _basis_qsplit_element="${_basis_qsplit_element:1}"
480  _basis_qsplit_element="${_basis_qsplit_element%\'}"
481  fi
482  # replace quoted quotes within argument by quotes
483  _basis_qsplit_element=`printf -- "${_basis_qsplit_element}" | sed "s/[\\]'/'/g;s/[\\]\"/\"/g"`
484  # add to resulting array
485  _basis_qsplit_array[${#_basis_qsplit_array[@]}]="${_basis_qsplit_element}"
486  # continue with residual command-line
487  _basis_qsplit_str="${BASH_REMATCH[4]}"
488  done
489  # return
490  local "$1" && upvar $1 "${_basis_qsplit_array[@]}"
491 }

493 # ----------------------------------------------------------------------------
494 ## @brief Execute command as subprocess.
495 #
496 # This function is used to execute a subprocess within a Bash script.
497 #
498 # Example:
499 # @code
500 # # the next command will exit the current shell if it fails
501 # execute ls /not/existing
502 # # to prevent this, provide the --allow_fail option
503 # execute --allow_fail ls /not/existing
504 # # to make it explicit where the command-line to execute starts, use --
505 # execute --allow_fail -- ls /not/existing
506 # @endcode
507 #
508 # Note that the output of the command is not redirected by this function.
509 # In order to execute the command quietly, use this function as follows:
510 # @code
511 # execute ls / &> /dev/null
512 # @endcode
513 # Or to store the command output in a variable including error messages
514 # use it as follows:
515 # @code
516 # output=`execute ls / 2>&1`
517 # @endcode
518 # Note that in this case, the option --allow_fail has no effect as the
519 # calling shell will never be terminated. Only the subshell in which the
520 # command is executed will be terminated. Checking the exit code $? is
521 # in this case required.
522 #
523 # @param [in] options Function options as documented below.
524 # @param [in] cmd Executable of command to run or corresponding build
525 # target name. This is assumed to be the first
526 # non-option argument or the argument that follows the
527 # special '--' argument.
528 # @param [in] args All remaining arguments are passed as arguments to
529 # the given command.
530 # @par Options:
531 # <table border="0">
532 # <tr>
533 # @tp <b>-f, --allow_fail</b> @endtp
534 # <td>Allows the command to fail. By default, if the command
535 # returns a non-zero exit code, the exit() function is
536 # called to terminate the current shell.</td>
537 # </tr>
538 # <tr>
539 # @tp <b>-v, --verbose</b> [int] @endtp
540 # <td>Print command-line to stdout before execution. Optionally, as it is
541 # sometimes more convenient to pass in the value of another variable
542 # which controls the verbosity of the parent process itself, a verbosity
543 # value can be specified following the option flag. If this verbosity
544 # less or equal to zero, the command-line of the subprocess is not
545 # printed to stdout, otherwise it is.</td>
546 # </tr>
547 # <tr>
548 # @tp <b>-s, --simulate</b> @endtp
549 # <td>If this option is given, the command is not actually
550 # executed, but the command-line printed to stdout only.</td>
551 # </tr>
552 # </table>
553 #
554 # @returns Exit code of subprocess.
555 execute()
556 {
557  # parse arguments
558  local _basis_execute_allow_fail='false'
559  local _basis_execute_simulate='false'
560  local _basis_execute_verbose=0
561  local _basis_execute_args=''
562  while [ $# -gt 0 ]; do
563  case "$1" in
564  -f|--allow_fail) _basis_execute_allow_fail='true'; ;;
565  -s|--simulate) _basis_execute_simulate='true'; ;;
566  -v|--verbose)
567  match "$2" '^-?[0-9]+$'
568  if [ $? -eq 0 ]; then
569  _basis_execute_verbose=$2
570  shift
571  else
572  let _basis_execute_verbose++
573  fi
574  ;;
575  --) shift; break; ;;
576  *) break; ;;
577  esac
578  shift
