34 void get_terminal_size(int& lines, int& columns)
35 {
36  lines = 0;
37  columns = 0;
38  #if WINDOWS
39  CONSOLE_SCREEN_BUFFER_INFO csbi;
40  if (GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &csbi)) {
41  columns = csbi.dwSize.X;
42  lines = csbi.dwSize.Y;
43  }
44  #else
45  struct winsize w;
46  if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &w) == 0) {
47  columns = w.ws_col;
48  lines = w.ws_row;
49  }
50  #endif
51  if (columns == 0) {
52  const char* COLUMNS = getenv("COLUMNS");
53  if (COLUMNS) columns = atoi(COLUMNS);
54  }
55  if (lines == 0) {
56  const char* LINES = getenv("LINES");
57  if (LINES) columns = atoi(LINES);
58  }
59 }

78 ostream& print_wrapped(ostream& os,
79  const string& text,
80  int width,
81  int indent,
82  int offset)
83 {
84  assert(indent + offset < width); // such that allowed_length > 0
85 
86  int text_length = static_cast<int>(text.length());
87  int allowed_length = width - indent;
88  int start = 0;
89 
90  // Note: Despite of the TCLAP::StdOutput::spacePrint() implementation,
91  // the following while loop is always performed even if the given
92  // text seems to fit on one line. Reason is that the text can
93  // include newline characters itself. In this case, we still need
94  // to take care of the proper indentation of the consecutive lines.
95 
96  while (start < text_length) {
97  // determine length of next line to be printed
98  int line_length = min<int>(text_length - start, allowed_length);
99  if (line_length == allowed_length) {
100  while (line_length >= 0 &&
101  text[start + line_length] != ' ' &&
102  text[start + line_length] != ',' &&
103  text[start + line_length] != '|' ) {
104  line_length--;
105  }
106  }
107  if (line_length <= 0) line_length = allowed_length;
108  // truncate line at already present newline (including the newline)
109  for (int i = 0; i < line_length; i++) {
110  if (text[start + i] == '\n') line_length = i + 1;
111  }
112  // print the line and add a newline
113  for (int i = 0; i < indent; i++ ) os << " ";
114  os << text.substr(start, line_length) << endl;
115  // adjust indent for lines after the first one
116  if (start == 0) {
117  indent += offset;
118  allowed_length -= offset;
119  }
120  // next line
121  start += line_length;
122  // skip space characters so next line does not start with
123  // a further indentation besides the one specified by the indent
124  while (text[start] == ' ' && start < text_length) start++;
125  }
126 
127  return os;
128 }
