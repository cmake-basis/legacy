97 ## @brief Print version information including copyright and license notices.
98 #
99 # @note This function can be customized when importing it in order to set
100 # default values for its parameters, which is in particular done by
101 # the Basis module.
102 #
103 # Example:
104 # @code
105 # use BASIS::Utilities qw(print_version);
106 # print_version('foo', '1.0');
107 # print_version('foo', version => '1.0');
108 # print_version(name => 'foo', version => '1.0');
109 # @endcode
110 #
111 # Example:
112 # @code
113 # use BASIS::Utilities
114 # print_version => {project => 'FooBar',
115 # version => '1.0',
116 # copyright => '2012 Andreas Schuh',
117 # license => 'Licensed under the Apache License, Version 2.0'};
118 # print_version('foo');
119 # @endcode
120 # which results in the output
121 # @verbatim
122 # foo (FooBar) 1.0
123 # Copyright (c) 2012 Andreas Schuh. All rights reserved.
124 # Licensed under the Apache License, Version 2.0
125 # @endverbatim
126 #
127 # @param [in] $name Name of executable. Should not be set programmatically
128 # to the first argument of the main script, but a string
129 # literal instead. This argument is required if no default
130 # has been set during customization. The argument can be
131 # either given as first argument or as keyword argument
132 # as in "name => 'foo'".
133 # @param [in] $version Version of executable, e.g., release of project this
134 # executable belongs to. This argument is required if no
135 # default has been set during customization. The argument
136 # can be either given as second argument or as keyword
137 # argument as in "version => '1.0'".
138 # @param [in] $project Name of project this executable belongs to.
139 # If @c undef or an empty string is given, no project
140 # information is included in output.
141 # @param [in] $copyright The copyright notice. If @c undef, the default copyright
142 # is used. If an empty string is given, no copyright notice
143 # is printed.
144 # @param [in] $license Information regarding licensing. If @c undef, the default
145 # software license is used. If an empty string is given,
146 # no license information is printed.
147 sub print_version
148 {
149  my $name = undef;
150  my $version = undef;
151  if (@_ != 0 and (not defined $_[0] or $_[0] !~ /^(name|version|project|copyright|license)$/)) {
152  $name = $_[0];
153  shift;
154  }
155  if (@_ != 0 and (not defined $_[0] or $_[0] !~ /^(name|version|project|copyright|license)$/)) {
156  $version = $_[0];
157  shift;
158  }
159  die "print_version(): Invalid number of arguments" if scalar(@_) % 2 == 1;
160  my %defaults = (name => undef, version => undef, project => undef, copyright => COPYRIGHT, license => LICENSE);
161  my %options = (%defaults, @_);
162  die "print_version(): Name argument given twice" if defined $options{'name'} and defined $name;
163  die "print_version(): Version argument given twice" if defined $options{'version'} and defined $version;
164  $name = $options{'name'} unless $name;
165  $version = $options{'version'} unless $version;
166  die "print_version(): Missing name argument" unless $name;
167  die "print_version(): Missing version argument" unless $version;
168  # program identification
169  print $name;
170  print " ($options{'project'})" if $options{'project'};
171  print " ", $version, "\n";
172  # copyright notice
173  print "Copyright (c) ", $options{'copyright'}, ". All rights reserved.\n" if $options{'copyright'};
174  # license information
175  print $options{'license'}, "\n" if $options{'license'};
176 }

179 ## @brief Get UID of build target.
180 #
181 # @note This function can be customized when importing it in order to set
182 # default values for @p prefix and @p targets, which is in particular
183 # done by the Basis module.
184 #
185 # This function prepends the default namespace used for targets build as
186 # part of the project this module belongs to if the given target name is yet
187 # neither known nor fully-qualified, i.e., in the form "<namespace>::<target>".
188 #
189 # @param [in] $target Name of build target.
190 # @param [in] $prefix Common target name prefix. If @c undef, the given
191 # target name must match excactly. Otherwise, targets
192 # within the specified namespace are considered.
193 # @param [in] %$targets Reference to hash which maps known build targets to
194 # executable file paths. If not specified, this function
195 # always returns the input target name unchanged.
196 #
197 # @returns Fully-qualified build target name or @c undef if @p target is
198 # @c undef or an empty string.
199 sub targetuid
200 {
201  my $target = shift;
202  my $prefix = shift;
203  my $targets = shift;
204  # handle invalid arguments
205  return undef unless defined $target and length($target) > 0;
206  # in case of a leading namespace separator or if no lookup table
207  # of executable build target is provided, do not modify target name
208  return $target if $target =~ /^\./ or not defined $targets;
209  # project namespace
210  $prefix = '' unless defined $prefix;
211  $prefix = $prefix . '.DUMMY'; # simplifies while loop
212  # try prepending namespace or parts of it until target is known
213  while ($prefix =~ s/(.*)\.[^.]*/$1/) {
214  if (exists $targets->{$prefix . '.' . $target}) {
215  return $prefix . '.' . $target;
216  }
217  }
218  # otherwise, return target name unchanged
219  return $target;
220 }

253 # ----------------------------------------------------------------------------
254 ## @brief Get absolute path of executable file.
255 #
256 # @note This function can be customized when importing it in order to set
257 # default values for @p prefix and @p targets, which is in particular
258 # done by the Basis module.
259 #
260 # This function determines the absolute file path of an executable. If no
261 # arguments are given, the absolute path of this executable is returned.
262 # Otherwise, the named command is searched in the system PATH and its
263 # absolute path returned if found. If the executable is not found, @c undef
264 # is returned.
265 #
266 # @param [in] $name Name of command or @c undef.
267 # @param [in] $prefix Common target name prefix. If @c undef, the given
268 # target name must match excactly. Otherwise, targets
269 # within the specified namespace are considered.
270 # @param [in] %$targets Reference to hash which maps known build targets to
271 # executable file paths. If not specified, this function
272 # always returns false.
273 # @param [in] $base Base directory used for relative paths in @p %$targets.
274 #
275 # @returns Absolute path of executable or @c undef if not found.
276 # If @p name is @c undef, the path of this executable is returned.
277 sub exepath
278 {
279  my $name = shift;
280  my $prefix = shift;
281  my $targets = shift;
282  my $base = shift || '.';
283  my $path = undef;
284  if (not defined $name) {
285  $path = realpath($0);
286  } elsif (defined $targets) {
287  my $uid = targetuid($name, $prefix, $targets);
288  defined $uid and $uid =~ s/^[.]?(.*)/$1/;
289  if (defined $uid and exists $targets->{$uid}) {
290  $path = $targets->{$uid};
291  if ($path =~ m/\$\(IntDir\)/) {
292  my $tmppath = '';
293  my $intdir = '';
294  foreach $intdir ('Release', 'Debug', 'RelWithDebInfo', 'MinSizeRel') {
295  $tmppath = $path;
296  $tmppath =~ s/\$\(IntDir\)/$intdir/g;
297  if (-e $tmppath) {
298  $path = $tmppath;
299  last;
300  }
301  }
302  $path =~ s/\$\(IntDir\)//g;
303  }
304  $path = File::Spec->rel2abs($path, File::Spec->rel2abs($base, dirname(__FILE__)));
305  # the realpath() function only works for existing paths
306  $path = realpath($path) if -e $path;
307  }
308  }
309  $path = which($name) unless defined $path;
310  return $path;
311 }

368 ## @brief Convert list to double quoted string.
369 #
370 # @param [in] @$args Array of arguments.
371 #
372 # @returns Double quoted string, i.e., string where array elements are separated
373 # by a space character and surrounded by double quotes if necessary.
374 # Double quotes within an array element are escaped with a backslash.
375 sub tostring
376 {
377  my $str = '';
378  if (ref($_[0]) eq 'ARRAY') {
379  foreach my $arg (@{$_[0]}) {
380  $arg =~ s/"/\\"/g; # escape double quotes
381  $arg = '"' . $arg . '"' if $arg =~ m/'|\s|^$/; # quote if necessary
382  $str .= ' ' if $str ne '';
383  $str .= $arg;
384  }
385  } else {
386  $str = $_[0];
387  $str =~ s/"/\\"/g; # escape double quotes
388  $str = '"' . $str . '"' if $str =~ m/'|\s|^$/; # quote if necessary
389  }
390  return $str;
391 }

393 # ----------------------------------------------------------------------------
394 ## @brief Split quoted string.
395 #
396 # @param [in] $str Quoted string.
397 sub qsplit
398 {
399  my $str = shift;
400  my $max = shift;
401  my $arg = '';
402  my @args = ();
403  LOOP: {
404  while ($str =~ /[ ]*('([^']|\\\')*[^\\]'|\"([^\"]|\\\")*[^\\]\"|[^ ]+)(.*)/) {
405  $arg = $1; # matched element including quotes
406  $str = $4; # continue with residual command-line
407  $arg =~ s/^['\"]|(^|[^\\])['\"]$//g; # remove quotes
408  $arg =~ s/[\\]('|\")/$1/g; # unescape quotes
409  push @args, $arg; # add to resulting array
410  last LOOP if defined $max and scalar(@args) >= $max;
411  }
412  }
413  if (defined $max) {
414  if ($max eq 1) { return ($args[0], $str); }
415  else { return (@args, $str); }
416  } else { return @args; }
417 }

443 # ----------------------------------------------------------------------------
444 ## @brief Execute command as subprocess.
445 #
446 # @note This function can be customized when importing it in order to set
447 # default values for @p prefix and @p targets, which is in particular
448 # done by the Basis module.
449 #
450 # This command takes either an array reference or a string as first argument.
451 # All other arguments are keyword arguments using hash notation.
452 #
453 # Example:
454 # @code
455 # # only returns exit code of command but does not output anything
456 # my $status = execute(['ls', '/'], quiet => 1);
457 # # returns exit code of command and returns command output w/o printing to stdout
458 # my ($status, $stdout) = execute('ls /', quiet => 1, stdout => 1);
459 # @endcode
460 #
461 # @param [in] $args Command with arguments given either as single quoted
462 # string or array of command name and arguments.
463 # @param [in] $quiet Turns off output of @c stdout of child process to
464 # @c stdout of parent process.
465 # @param [in] $stdout Whether to return the command output.
466 # @param [in] $allow_fail If true, does not raise an exception if return
467 # value is non-zero. Otherwise, an exception is
468 # raised by this function using die.
469 # @param [in] $verbose Verbosity of output messages.
470 # Does not affect verbosity of executed command.
471 # @param [in] $simulate Whether to simulate command execution only.
472 # @param [in] $prefix Common target name prefix. If @c undef, the given
473 # target name must match excactly. Otherwise, targets
474 # within the specified namespace are considered.
475 # @param [in] %$targets Reference to hash which maps known build targets to
476 # executable file paths. If not specified, this function
477 # always returns false.
478 # @param [in] $base Base directory used for relative paths in @p %$targets.
479 #
480 # @returns The exit code of the subprocess if @p stdout is false (the default).
481 # Otherwise, if @p stdout is true, a tuple consisting of exit code
482 # command output is returned. Note that if @p allow_fail is false,
483 # the returned exit code will always be 0.
484 #
485 # @throws die If command execution failed. This exception is not raised
486 # if the command executed with non-zero exit code but
487 # @p allow_fail is true.
488 sub execute
489 {
490  # arguments
491  my $args = shift or die "execute(): No command specified for execution";
492  if ($args =~ m/^(quiet|stdout|allow_fail|verbose|simulate|prefix|targets|base)$/) {
493  warn "First argument matches option name. Missing args argument?";
494  }
495  my %defaults = (quiet => 0, stdout => 0, allow_fail => 0,
496  verbose => 0, simulate => 0,
497  prefix => undef, targets => undef, base => '.');
498  my %options = (%defaults, @_);
499  # get absolute path of executable
500  my ($command, $arguments) = _split_command_and_arguments($args);
501  my $exec_path = exepath($command, $options{'prefix'}, $options{'targets'}, $options{'base'});
502  defined $exec_path or die "$command: Command not found";
503  $exec_path = '"' . $exec_path . '"' if $exec_path =~ m/'|\s/; # quote if necessary
504  $args = "$exec_path $arguments";
505  # some verbose output
506  if ($options{'verbose'} gt 0 or $options{'simulate'}) {
507  print "\$ ", $args;
508  $options{'simulate'} and print " (simulated)";
509  print "\n";
510  }
511  # execute command
512  my $status = 0;
513  my $output = '';
514  if (not $options{'simulate'}) {
515  open CMD, "$args |" or die "$command: Failed to open subprocess";
516  my $ofh = select STDOUT;
517  $|++;
518  while (<CMD>) {
519  print $_ unless $options{'quiet'};
520  $output .= $_ if $options{'stdout'};
521  }
522  $|--;
523  select $ofh;
524  close CMD;
525  $status = $?;
526  }
527  # if command failed, throw an exception
528  if ($status != 0 and not $options{'allow_fail'}) {
529  die "Command $args failed";
530  }
531  # return
532  if ($options{'stdout'}) { return ($status, $output); }
533  else { return $status; }
534 }
