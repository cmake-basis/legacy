106 /**
107  * @brief Positional argument.
108  *
109  * Processes only one positional argument. Add the positional arguments in
110  * the right order to the command-line.
111  *
112  * @note The other unlabeled arguments as supported by TCLAP are intentionally
113  * not available through BASIS itself. Any non-string argument should
114  * more likely be a labeled argument, i.e., one with an option flag.
115  */
116 typedef TCLAP::UnlabeledValueArg<std::string> PositionalArg;
117 
118 /**
119  * @brief Positional arguments.
120  *
121  * Use only one positional argument per command-line. Must be the last argument
122  * added to the command-line as it is greedy and will aggregate all remaining
123  * command-line arguments.
124  *
125  * @note The other unlabeled arguments as supported by TCLAP are intentionally
126  * not available through BASIS itself. Any non-string argument should
127  * more likely be a labeled argument, i.e., one with an option flag.
128  */
129 typedef TCLAP::UnlabeledMultiArg<std::string> PositionalArgs;
