2 # @file BasisPack.cmake
3 # @brief CPack configuration. Include this module instead of CPack.
4 #
5 # @note This module is included by basis_project_finalize().
6 #
7 # This module implements the packaging of BASIS projects.
8 #
9 # Overwrite the package information set by this module either in a file
10 # Package.cmake or a file Package.cmake.in located in the directory
11 # specified by PROJECT_CONFIG_DIR. The latter is configured and copied to the
12 # binary tree before included by this module. Further, to enable a
13 # component-based installation, provide either a file Components.cmake or
14 # Components.cmake.in again in the directory specified by PROJECT_CONFIG_DIR.
15 # Also in this case, the latter is configured via CMake's configure_file()
16 # before use. This file is referred to as components definition (file).
17 #
18 # Components can be added in the components definition using the command
19 # basis_add_component(). Several components can be grouped together and a
20 # group description be added using the command basis_add_component_group().
21 # Different pre-configured install types which define a certain selection of
22 # components to install can be added using basis_add_install_type().
23 # Note that all these BASIS functions are wrappers around the corresponding
24 # CPack functions.
25 #
26 # @sa CPack.cmake
27 # @sa http://www.vtk.org/Wiki/CMake:Component_Install_With_CPack#Component-Based_Installers_with_CPack

216 # @sa http://www.cmake.org/pipermail/cmake/2008-August/023336.html
217 # @sa cpack_add_component_group()
218 function (basis_add_component_group GRPNAME)
219  set (OPTION_NAME)
220  set (PARENT_GROUP)
221 
222  foreach (ARG ${ARGN})
223  if (OPTION_NAME)
224  set (${OPTION_NAME} "${ARG}")
225  set (OPTION_NAME)
226  break ()
227  else ()
228  if (ARG MATCHES "^PARENT_GROUP$")
229  set (OPTION_NAME "PARENT_GROUP")
230  endif ()
231  endif ()
232  endforeach ()
233 
234  cpack_add_component_group (${GRPNAME} ${ARGN})
235 
236  add_custom_target (install_${GRPNAME})
237 
238  if (PARENT_GROUP)
239  add_dependencies (install_${PARENT_GROUP} install_${GRPNAME})
240  endif ()
241 endfunction ()
